19use darkfi::{geode::hash_to_string, rpc::util::json_map};
20use tinyjson::JsonValue;
21
22#[derive(Clone, Debug)]
23pub enum ResourceStatus {
24    Downloading,
25    Seeding,
26    Discovering,
27    Incomplete,
28}
29
30#[derive(Clone, Debug)]
31pub struct Resource {
32    pub hash: blake3::Hash,
33    pub status: ResourceStatus,
34    pub chunks_total: u64,
35    pub chunks_downloaded: u64,
36}
37
38impl From<Resource> for JsonValue {
39    fn from(rs: Resource) -> JsonValue {
40        json_map([
41            ("hash", JsonValue::String(hash_to_string(&rs.hash))),
42            (
43                "status",
44                JsonValue::String(
45                    match rs.status {
46                        ResourceStatus::Downloading => "downloading",
47                        ResourceStatus::Seeding => "seeding",
48                        ResourceStatus::Discovering => "discovering",
49                        ResourceStatus::Incomplete => "incomplete",
50                    }
51                    .to_string(),
52                ),
53            ),
54            ("chunks_total", JsonValue::Number(rs.chunks_total as f64)),
55            ("chunks_downloaded", JsonValue::Number(rs.chunks_downloaded as f64)),
56        ])
57    }
58}
