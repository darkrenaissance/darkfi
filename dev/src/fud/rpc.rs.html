51#[async_trait]
52impl RequestHandler<()> for Fud {
53    async fn handle_request(&self, req: JsonRequest) -> JsonResult {
54        return match req.method.as_str() {
55            "ping" => self.pong(req.id, req.params).await,
56
57            "put" => self.put(req.id, req.params).await,
58            "get" => self.get(req.id, req.params).await,
59            "subscribe" => self.subscribe(req.id, req.params).await,
60            "remove" => self.remove_resource(req.id, req.params).await,
61            "list_resources" => self.list_resources(req.id, req.params).await,
62            "list_buckets" => self.list_buckets(req.id, req.params).await,
63            "list_seeders" => self.list_seeders(req.id, req.params).await,
64            "verify" => self.verify(req.id, req.params).await,
65
66            "dnet.switch" => self.dnet_switch(req.id, req.params).await,
67            "dnet.subscribe_events" => self.dnet_subscribe_events(req.id, req.params).await,
68            "p2p.get_info" => self.p2p_get_info(req.id, req.params).await,
69            _ => JsonError::new(ErrorCode::MethodNotFound, None, req.id).into(),
70        }
71    }
72
73    async fn connections_mut(&self) -> MutexGuard<'_, HashSet<StoppableTaskPtr>> {
74        self.rpc_connections.lock().await
75    }
76}

78/// Fud RPC methods
79impl Fud {
80    // RPCAPI:
81    // Put a file onto the network. Takes a local filesystem path as a parameter.
82    // Returns the file hash that serves as a pointer to the uploaded file.
83    //
84    // --> {"jsonrpc": "2.0", "method": "put", "params": ["/foo.txt"], "id": 42}
85    // <-- {"jsonrpc": "2.0", "result: "df4...3db7", "id": 42}
86    async fn put(&self, id: u16, params: JsonValue) -> JsonResult {
87        let self_node = self.dht.node().await;
88
89        if self_node.addresses.is_empty() {
90            error!(target: "fud::put()", "Cannot put file, you don't have any external address");
91            return JsonError::new(
92                ErrorCode::InternalError,
93                Some("You don't have any external address".to_string()),
94                id,
95            )
96            .into()
97        }
98
99        let params = params.get::<Vec<JsonValue>>().unwrap();
100        if params.len() != 1 || !params[0].is_string() {
101            return JsonError::new(ErrorCode::InvalidParams, None, id).into()
102        }
103
104        let path = params[0].get::<String>().unwrap();
105        let path = match expand_path(path.as_str()) {
106            Ok(v) => v,
107            Err(_) => return JsonError::new(ErrorCode::InvalidParams, None, id).into(),
108        };
109
110        // A valid path was passed. Let's see if we can read it, and if so,
111        // add it to Geode.
112        let fd = match File::open(&path).await {
113            Ok(v) => v,
114            Err(e) => {
115                error!(target: "fud::put()", "Failed to open {:?}: {}", path, e);
116                return JsonError::new(ErrorCode::InvalidParams, None, id).into()
117            }
118        };
119
120        let (file_hash, chunk_hashes) = match self.geode.insert(fd).await {
121            Ok(v) => v,
122            Err(e) => {
123                let error_str = format!("Failed inserting file {:?} to geode: {}", path, e);
124                error!(target: "fud::put()", "{}", error_str);
125                return JsonError::new(ErrorCode::InternalError, Some(error_str), id).into()
126            }
127        };
128
129        // Add path to the sled db
130        if let Err(e) = self
131            .path_tree
132            .insert(file_hash.as_bytes(), path.to_string_lossy().to_string().as_bytes())
133        {
134            error!(target: "fud::put()", "Failed inserting new file into sled: {}", e);
135            return JsonError::new(ErrorCode::InternalError, None, id).into()
136        }
137
138        // Add resource
139        let mut resources_write = self.resources.write().await;
140        resources_write.insert(
141            file_hash,
142            Resource {
143                hash: file_hash,
144                path,
145                status: ResourceStatus::Seeding,
146                chunks_total: chunk_hashes.len() as u64,
147                chunks_downloaded: chunk_hashes.len() as u64,
148            },
149        );
150        drop(resources_write);
151
152        // Announce file
153        let fud_announce = FudAnnounce { key: file_hash, seeders: vec![self_node.into()] };
154        let _ = self.announce(&file_hash, &fud_announce, self.seeders_router.clone()).await;
155
156        JsonResponse::new(JsonValue::String(hash_to_string(&file_hash)), id).into()
157    }
158
159    // RPCAPI:
160    // Fetch a file from the network. Takes a file hash and file path (absolute or relative) as parameters.
161    // Returns the path where the file will be located once downloaded.
162    //
163    // --> {"jsonrpc": "2.0", "method": "get", "params": ["1211...abfd", "~/myfile.jpg"], "id": 42}
164    // <-- {"jsonrpc": "2.0", "result": "/home/user/myfile.jpg", "id": 42}
165    async fn get(&self, id: u16, params: JsonValue) -> JsonResult {
166        let params = params.get::<Vec<JsonValue>>().unwrap();
167        if params.len() != 2 || !params[0].is_string() || !params[1].is_string() {
168            return JsonError::new(ErrorCode::InvalidParams, None, id).into()
169        }
170
171        let mut hash_buf = vec![];
172        match bs58::decode(params[0].get::<String>().unwrap().as_str()).onto(&mut hash_buf) {
173            Ok(_) => {}
174            Err(_) => return JsonError::new(ErrorCode::InvalidParams, None, id).into(),
175        }
176
177        if hash_buf.len() != 32 {
178            return JsonError::new(ErrorCode::InvalidParams, None, id).into()
179        }
180
181        let mut hash_buf_arr = [0u8; 32];
182        hash_buf_arr.copy_from_slice(&hash_buf);
183
184        let file_hash = blake3::Hash::from_bytes(hash_buf_arr);
185        let file_hash_str = hash_to_string(&file_hash);
186
187        let file_path = match params[1].get::<String>() {
188            Some(path) => match path.is_empty() {
189                true => self.downloads_path.join(&file_hash_str).join(&file_hash_str),
190                false => match PathBuf::from(path).is_absolute() {
191                    true => PathBuf::from(path),
192                    false => self.downloads_path.join(&file_hash_str).join(path),
193                },
194            },
195            None => self.downloads_path.join(&file_hash_str).join(&file_hash_str),
196        };
197
198        // Get the parent directory of the file
199        if let Some(parent) = file_path.parent() {
200            // Create all directories leading up to the file
201            let _ = fs::create_dir_all(parent).await;
202        }
203
204        let _ = self.get_tx.send((id, file_hash, file_path.clone(), Ok(()))).await;
205
206        JsonResponse::new(JsonValue::String(file_path.to_string_lossy().to_string()), id).into()
207    }

218    // RPCAPI:
219    // Activate or deactivate dnet in the P2P stack.
220    // By sending `true`, dnet will be activated, and by sending `false` dnet
221    // will be deactivated. Returns `true` on success.
222    //
223    // --> {"jsonrpc": "2.0", "method": "dnet_switch", "params": [true], "id": 42}
224    // <-- {"jsonrpc": "2.0", "result": true, "id": 42}
225    async fn dnet_switch(&self, id: u16, params: JsonValue) -> JsonResult {
226        let params = params.get::<Vec<JsonValue>>().unwrap();
227        if params.len() != 1 || !params[0].is_bool() {
228            return JsonError::new(ErrorCode::InvalidParams, None, id).into()
229        }
230
231        let switch = params[0].get::<bool>().unwrap();
232
233        if *switch {
234            self.p2p.dnet_enable();
235        } else {
236            self.p2p.dnet_disable();
237        }
238
239        JsonResponse::new(JsonValue::Boolean(true), id).into()
240    }

258    // RPCAPI:
259    // Returns resources.
260    //
261    // --> {"jsonrpc": "2.0", "method": "list_buckets", "params": [], "id": 1}
262    // <-- {"jsonrpc": "2.0", "result": [[["abcdef", ["tcp://127.0.0.1:13337"]]]], "id": 1}
263    pub async fn list_resources(&self, id: u16, params: JsonValue) -> JsonResult {
264        let params = params.get::<Vec<JsonValue>>().unwrap();
265        if !params.is_empty() {
266            return JsonError::new(ErrorCode::InvalidParams, None, id).into()
267        }
268
269        let resources_read = self.resources.read().await;
270        let mut resources: Vec<JsonValue> = vec![];
271        for (_, resource) in resources_read.iter() {
272            resources.push(resource.clone().into());
273        }
274
275        JsonResponse::new(JsonValue::Array(resources), id).into()
276    }

278    // RPCAPI:
279    // Returns the current buckets.
280    //
281    // --> {"jsonrpc": "2.0", "method": "list_buckets", "params": [], "id": 1}
282    // <-- {"jsonrpc": "2.0", "result": [[["abcdef", ["tcp://127.0.0.1:13337"]]]], "id": 1}
283    pub async fn list_buckets(&self, id: u16, params: JsonValue) -> JsonResult {
284        let params = params.get::<Vec<JsonValue>>().unwrap();
285        if !params.is_empty() {
286            return JsonError::new(ErrorCode::InvalidParams, None, id).into()
287        }
288        let mut buckets = vec![];
289        for bucket in self.dht.buckets.read().await.iter() {
290            let mut nodes = vec![];
291            for node in bucket.nodes.clone() {
292                let mut addresses = vec![];
293                for addr in node.addresses {
294                    addresses.push(JsonValue::String(addr.to_string()));
295                }
296                nodes.push(JsonValue::Array(vec![
297                    JsonValue::String(hash_to_string(&node.id)),
298                    JsonValue::Array(addresses),
299                ]));
300            }
301            buckets.push(JsonValue::Array(nodes));
302        }
303
304        JsonResponse::new(JsonValue::Array(buckets), id).into()
305    }

307    // RPCAPI:
308    // Returns the content of the seeders router.
309    //
310    // --> {"jsonrpc": "2.0", "method": "list_seeders", "params": [], "id": 1}
311    // <-- {"jsonrpc": "2.0", "result": {"seeders": {"abcdef": ["ghijkl"]}}, "id": 1}
312    pub async fn list_seeders(&self, id: u16, params: JsonValue) -> JsonResult {
313        let params = params.get::<Vec<JsonValue>>().unwrap();
314        if !params.is_empty() {
315            return JsonError::new(ErrorCode::InvalidParams, None, id).into()
316        }
317        let mut seeders_router: HashMap<String, JsonValue> = HashMap::new();
318        for (hash, items) in self.seeders_router.read().await.iter() {
319            let mut node_ids = vec![];
320            for item in items {
321                node_ids.push(JsonValue::String(hash_to_string(&item.node.id)));
322            }
323            seeders_router.insert(hash_to_string(hash), JsonValue::Array(node_ids));
324        }
325        let mut res: HashMap<String, JsonValue> = HashMap::new();
326        res.insert("seeders".to_string(), JsonValue::Object(seeders_router));
327
328        JsonResponse::new(JsonValue::Object(res), id).into()
329    }

331    // RPCAPI:
332    // Removes a resource.
333    //
334    // --> {"jsonrpc": "2.0", "method": "remove", "params": ["1211...abfd"], "id": 1}
335    // <-- {"jsonrpc": "2.0", "result": [], "id": 1}
336    pub async fn remove_resource(&self, id: u16, params: JsonValue) -> JsonResult {
337        let params = params.get::<Vec<JsonValue>>().unwrap();
338        if params.len() != 1 || !params[0].is_string() {
339            return JsonError::new(ErrorCode::InvalidParams, None, id).into()
340        }
341        let mut hash_buf = [0u8; 32];
342        match bs58::decode(params[0].get::<String>().unwrap().as_str()).onto(&mut hash_buf) {
343            Ok(_) => {}
344            Err(_) => return JsonError::new(ErrorCode::InvalidParams, None, id).into(),
345        }
346
347        let hash = blake3::Hash::from_bytes(hash_buf);
348        let mut resources_write = self.resources.write().await;
349        resources_write.remove(&hash);
350        drop(resources_write);
351
352        self.event_publisher
353            .notify(FudEvent::ResourceRemoved(event::ResourceRemoved { hash }))
354            .await;
355
356        JsonResponse::new(JsonValue::Array(vec![]), id).into()
357    }

359    // RPCAPI:
360    // Verifies local files. Takes a list of file hashes as parameters.
361    // An empty list means all known files.
362    // Returns the path where the file will be located once downloaded.
363    //
364    // --> {"jsonrpc": "2.0", "method": "verify", "params": ["1211...abfd"], "id": 42}
365    // <-- {"jsonrpc": "2.0", "result": [], "id": 1}
366    async fn verify(&self, id: u16, params: JsonValue) -> JsonResult {
367        let params = params.get::<Vec<JsonValue>>().unwrap();
368        if !params.iter().all(|param| param.is_string()) {
369            return JsonError::new(ErrorCode::InvalidParams, None, id).into()
370        }
371        let hashes = if params.is_empty() {
372            None
373        } else {
374            let hashes_str: Vec<String> =
375                params.iter().map(|param| param.get::<String>().unwrap().clone()).collect();
376            let hashes: Result<Vec<blake3::Hash>> = hashes_str
377                .into_iter()
378                .map(|hash_str| {
379                    let mut buf = [0u8; 32];
380                    bs58::decode(hash_str).onto(&mut buf)?;
381                    Ok(blake3::Hash::from_bytes(buf))
382                })
383                .collect();
384            if hashes.is_err() {
385                return JsonError::new(ErrorCode::InvalidParams, None, id).into();
386            }
387            Some(hashes.unwrap())
388        };
389
390        if let Err(e) = self.verify_resources(hashes).await {
391            error!(target: "fud::verify()", "Could not verify resources: {}", e);
392            return JsonError::new(ErrorCode::InternalError, None, id).into();
393        }
394
395        JsonResponse::new(JsonValue::Array(vec![]), id).into()
396    }

399impl Fud {
400    /// Handle `get` RPC request
401    pub async fn handle_get(&self, file_hash: &blake3::Hash, file_path: &PathBuf) -> Result<()> {
402        let self_node = self.dht().node().await;
403        let mut closest_nodes = vec![];
404
405        // Add path to the sled db
406        self.path_tree
407            .insert(file_hash.as_bytes(), file_path.to_string_lossy().to_string().as_bytes())?;
408
409        // Add resource to `self.resources`
410        let resource = Resource {
411            hash: *file_hash,
412            path: file_path.clone(),
413            status: ResourceStatus::Discovering,
414            chunks_total: 0,
415            chunks_downloaded: 0,
416        };
417        let mut resources_write = self.resources.write().await;
418        resources_write.insert(*file_hash, resource.clone());
419        drop(resources_write);
420
421        // Send a DownloadStarted event
422        self.event_publisher
423            .notify(FudEvent::DownloadStarted(event::DownloadStarted {
424                hash: *file_hash,
425                resource,
426            }))
427            .await;
428
429        // Try to get the chunked file from geode
430        let chunked_file = match self.geode.get(file_hash, file_path).await {
431            // We already know the list of chunk hashes for this file
432            Ok(v) => v,
433            // The metadata in geode is invalid or corrupted
434            Err(Error::GeodeNeedsGc) => todo!(),
435            // If we could not find the file in geode, get the file metadata from the network
436            Err(Error::GeodeFileNotFound) => {
437                // Find nodes close to the file hash
438                info!(target: "self::get()", "Requested file {} not found in Geode, triggering fetch", hash_to_string(file_hash));
439                closest_nodes = self.lookup_nodes(file_hash).await.unwrap_or_default();
440
441                // Fetch file metadata (list of chunk hashes)
442                self.file_fetch_tx
443                    .send((closest_nodes.clone(), *file_hash, file_path.clone(), Ok(())))
444                    .await
445                    .unwrap();
446                info!(target: "self::get()", "Waiting for background file fetch task...");
447                let (i_file_hash, status) = self.file_fetch_end_rx.recv().await.unwrap();
448                match status {
449                    // The file metadata was found and inserted into geode
450                    Ok(()) => self.geode.get(&i_file_hash, file_path).await.unwrap(),
451                    // We could not find the file metadata
452                    Err(Error::GeodeFileRouteNotFound) => {
453                        // Set resource status to `Incomplete` and send FudEvent::FileNotFound
454                        let mut resources_write = self.resources.write().await;
455                        if let Some(resource) = resources_write.get_mut(file_hash) {
456                            resource.status = ResourceStatus::Incomplete;
457
458                            self.event_publisher
459                                .notify(FudEvent::FileNotFound(event::FileNotFound {
460                                    hash: *file_hash,
461                                    resource: resource.clone(),
462                                }))
463                                .await;
464                        }
465                        drop(resources_write);
466                        return Err(Error::GeodeFileRouteNotFound);
467                    }
468
469                    Err(e) => {
470                        error!(target: "fud::handle_get()", "{}", e);
471                        return Err(e);
472                    }
473                }
474            }
475
476            Err(e) => {
477                error!(target: "fud::handle_get()", "{}", e);
478                return Err(e);
479            }
480        };
481
482        // Set resource status to `Downloading`
483        let mut resources_write = self.resources.write().await;
484        let resource = match resources_write.get_mut(file_hash) {
485            Some(resource) => {
486                resource.status = ResourceStatus::Downloading;
487                resource.chunks_downloaded = chunked_file.local_chunks() as u64;
488                resource.chunks_total = chunked_file.len() as u64;
489                resource.clone()
490            }
491            None => return Ok(()), // Resource was removed, abort
492        };
493        drop(resources_write);
494
495        // Send a FileDownloadCompleted event
496        self.event_publisher
497            .notify(FudEvent::FileDownloadCompleted(event::FileDownloadCompleted {
498                hash: *file_hash,
499                resource: resource.clone(),
500            }))
501            .await;
502
503        // If the file is already complete, we don't need to download any chunk
504        if chunked_file.is_complete() {
505            // Announce the file
506            let self_announce =
507                FudAnnounce { key: *file_hash, seeders: vec![self_node.clone().into()] };
508            let _ = self.announce(file_hash, &self_announce, self.seeders_router.clone()).await;
509
510            // Set resource status to `Seeding`
511            let mut resources_write = self.resources.write().await;
512            let resource = match resources_write.get_mut(file_hash) {
513                Some(resource) => {
514                    resource.status = ResourceStatus::Seeding;
515                    resource.chunks_downloaded = chunked_file.len() as u64;
516                    resource.clone()
517                }
518                None => return Ok(()), // Resource was removed, abort
519            };
520            drop(resources_write);
521
522            // Send a DownloadCompleted event
523            self.event_publisher
524                .notify(FudEvent::DownloadCompleted(event::DownloadCompleted {
525                    hash: *file_hash,
526                    resource,
527                }))
528                .await;
529
530            return Ok(());
531        }
532
533        // Find nodes close to the file hash if we didn't previously fetched them
534        if closest_nodes.is_empty() {
535            closest_nodes = self.lookup_nodes(file_hash).await.unwrap_or_default();
536        }
537
538        // Find seeders and remove ourselves from the result
539        let seeders = self
540            .fetch_seeders(&closest_nodes, file_hash)
541            .await
542            .iter()
543            .filter(|seeder| seeder.node.id != self_node.id)
544            .cloned()
545            .collect();
546
547        // List missing chunks
548        let mut missing_chunks = HashSet::new();
549        for (chunk, path) in chunked_file.iter() {
550            if path.is_none() {
551                missing_chunks.insert(*chunk);
552            }
553        }
554
555        // Fetch missing chunks from seeders
556        self.fetch_chunks(file_path, file_hash, &missing_chunks, &seeders).await?;
557
558        // Get chunked file from geode
559        let chunked_file = match self.geode.get(file_hash, file_path).await {
560            Ok(v) => v,
561            Err(e) => {
562                error!(target: "fud::handle_get()", "{}", e);
563                return Err(e);
564            }
565        };
566
567        // We fetched all chunks, but the file is not complete
568        // (some chunks were missing from all seeders)
569        if !chunked_file.is_complete() {
570            // Set resource status to `Incomplete`
571            let mut resources_write = self.resources.write().await;
572            let resource = match resources_write.get_mut(file_hash) {
573                Some(resource) => {
574                    resource.status = ResourceStatus::Incomplete;
575                    resource.clone()
576                }
577                None => return Ok(()), // Resource was removed, abort
578            };
579            drop(resources_write);
580
581            // Send a MissingChunks event
582            self.event_publisher
583                .notify(FudEvent::MissingChunks(event::MissingChunks {
584                    hash: *file_hash,
585                    resource,
586                }))
587                .await;
588            return Ok(());
589        }
590
591        // Announce the file
592        let self_announce =
593            FudAnnounce { key: *file_hash, seeders: vec![self_node.clone().into()] };
594        let _ = self.announce(file_hash, &self_announce, self.seeders_router.clone()).await;
595
596        // Set resource status to `Seeding`
597        let mut resources_write = self.resources.write().await;
598        let resource = match resources_write.get_mut(file_hash) {
599            Some(resource) => {
600                resource.status = ResourceStatus::Seeding;
601                resource.chunks_downloaded = chunked_file.len() as u64;
602                resource.clone()
603            }
604            None => return Ok(()), // Resource was removed, abort
605        };
606        drop(resources_write);
607
608        // Send a DownloadCompleted event
609        self.event_publisher
610            .notify(FudEvent::DownloadCompleted(event::DownloadCompleted {
611                hash: *file_hash,
612                resource,
613            }))
614            .await;
615
616        Ok(())
617    }
