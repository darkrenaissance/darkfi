43/// Triggered when calling the `fud.get()` method.
44/// It creates a new StoppableTask (running `fud.fetch_resource()`) and inserts
45/// it into the `fud.fetch_tasks` hashmap. When the task is stopped it's
46/// removed from the hashmap.
47pub async fn get_task(fud: Arc<Fud>, executor: ExecutorPtr) -> Result<()> {
48    loop {
49        let (hash, path, files) = fud.get_rx.recv().await.unwrap();
50
51        // Create the new task
52        let mut fetch_tasks = fud.fetch_tasks.write().await;
53        let task = StoppableTask::new();
54        fetch_tasks.insert(hash, task.clone());
55        drop(fetch_tasks);
56
57        // Start the new task
58        let fud_1 = fud.clone();
59        let fud_2 = fud.clone();
60        task.start(
61            async move { fud_1.fetch_resource(&hash, &path, &files).await },
62            move |res| async move {
63                // Remove the task from the `fud.fetch_tasks` hashmap once it is
64                // stopped (error, manually, or just done).
65                let mut fetch_tasks = fud_2.fetch_tasks.write().await;
66                fetch_tasks.remove(&hash);
67                match res {
68                    Ok(()) | Err(Error::DetachedTaskStopped) => { /* Do nothing */ }
69                    Err(e) => {
70                        error!(target: "fud::get_task()", "Error while fetching resource: {e}");
71
72                        // Send a DownloadError for any error that stopped the fetch task
73                        notify_event!(fud_2, DownloadError, {
74                            hash,
75                            error: e.to_string(),
76                        });
77                    }
78                }
79            },
80            Error::DetachedTaskStopped,
81            executor.clone(),
82        );
83    }
84}

86/// Background task that announces our files once every hour.
87/// Also removes seeders that did not announce for too long.
88pub async fn announce_seed_task(fud: Arc<Fud>) -> Result<()> {
89    let interval = 3600; // TODO: Make a setting
90
91    loop {
92        sleep(interval).await;
93
94        let seeders = vec![fud.node().await.into()];
95
96        info!(target: "fud::announce_seed_task()", "Verifying seeds...");
97        let seeding_resources = match fud.verify_resources(None).await {
98            Ok(resources) => resources,
99            Err(e) => {
100                error!(target: "fud::announce_seed_task()", "Error while verifying seeding resources: {e}");
101                continue;
102            }
103        };
104
105        info!(target: "fud::announce_seed_task()", "Announcing files...");
106        for resource in seeding_resources {
107            let _ = fud
108                .announce(
109                    &resource.hash,
110                    &FudAnnounce { key: resource.hash, seeders: seeders.clone() },
111                    fud.seeders_router.clone(),
112                )
113                .await;
114        }
115
116        info!(target: "fud::announce_seed_task()", "Pruning seeders...");
117        fud.dht().prune_router(fud.seeders_router.clone(), interval.try_into().unwrap()).await;
118    }
119}

121/// Background task that:
122/// 1. Updates the [`crate::bitcoin::BitcoinHashCache`]
123/// 2. Removes old nodes from the DHT
124/// 3. Removes old nodes from the seeders router
125/// 4. If the Bitcoin block hash we currently use in our `fud.node_data` is too old, we update it and reset our DHT
126pub async fn node_id_task(fud: Arc<Fud>) -> Result<()> {
127    let interval = 600; // TODO: Make a setting
128
129    loop {
130        sleep(interval).await;
131
132        let mut pow = fud.pow.write().await;
133        let btc = &mut pow.bitcoin_hash_cache;
134
135        if btc.update().await.is_err() {
136            continue
137        }
138
139        let block = fud.node_data.read().await.btc_block_hash;
140        let needs_dht_reset = match btc.block_hashes.iter().position(|b| *b == block) {
141            Some(i) => i < 6,
142            None => true,
143        };
144
145        if !needs_dht_reset {
146            // Removes nodes in the DHT with unknown BTC block hashes.
147            let dht = fud.dht();
148            let mut buckets = dht.buckets.write().await;
149            for bucket in buckets.iter_mut() {
150                for (i, node) in bucket.nodes.clone().iter().enumerate().rev() {
151                    // If this node's BTC block hash is unknown, remove it from the bucket
152                    if !btc.block_hashes.contains(&node.data.btc_block_hash) {
153                        bucket.nodes.remove(i);
154                        info!(target: "fud::node_id_task()", "Removed node {} from the DHT (BTC block hash too old or unknown)", hash_to_string(&node.id()));
155                    }
156                }
157            }
158            drop(buckets);
159
160            // Removes nodes in the seeders router with unknown BTC block hashes
161            let mut seeders_router = fud.seeders_router.write().await;
162            for (key, seeders) in seeders_router.iter_mut() {
163                for seeder in seeders.clone().iter() {
164                    if !btc.block_hashes.contains(&seeder.node.data.btc_block_hash) {
165                        seeders.remove(seeder);
166                        info!(target: "fud::node_id_task()", "Removed node {} from the seeders of key {} (BTC block hash too old or unknown)", hash_to_string(&seeder.node.id()), hash_to_string(key));
167                    }
168                }
169            }
170
171            continue
172        }
173
174        info!(target: "fud::node_id_task()", "Creating a new node id...");
175        let (node_data, secret_key) = match pow.generate_node().await {
176            Ok(res) => res,
177            Err(e) => {
178                warn!(target: "fud::node_id_task()", "Error creating a new node id: {e}");
179                continue
180            }
181        };
182        drop(pow);
183        info!(target: "fud::node_id_task()", "New node id: {}", hash_to_string(&node_data.id()));
184
185        // Close all channels
186        let dht = fud.dht();
187        let mut channel_cache = dht.channel_cache.write().await;
188        for channel in dht.p2p.hosts().channels().clone() {
189            channel.stop().await;
190            channel_cache.remove(&channel.info.id);
191        }
192        drop(channel_cache);
193
194        // Reset the DHT
195        dht.reset().await;
196
197        // Reset the seeders router
198        *fud.seeders_router.write().await = HashMap::new();
199
200        // Update our node data and our secret key
201        *fud.node_data.write().await = node_data;
202        *fud.secret_key.write().await = secret_key;
203
204        // DHT will be bootstrapped on the next channel connection
205    }
