173    async fn ping(&self, channel: ChannelPtr) -> Result<DhtNode> {
174        debug!(target: "fud::DhtHandler::ping()", "Sending ping to channel {}", channel.info.id);
175        let msg_subsystem = channel.message_subsystem();
176        msg_subsystem.add_dispatch::<FudPingReply>().await;
177        let msg_subscriber = channel.subscribe_msg::<FudPingReply>().await.unwrap();
178        let request = FudPingRequest {};
179
180        channel.send(&request).await?;
181
182        let reply = msg_subscriber.receive_with_timeout(self.dht().settings.timeout).await?;
183
184        msg_subscriber.unsubscribe().await;
185
186        Ok(reply.node.clone())
187    }
188
189    // TODO: Optimize this
190    async fn on_new_node(&self, node: &DhtNode) -> Result<()> {
191        debug!(target: "fud::DhtHandler::on_new_node()", "New node {}", hash_to_string(&node.id));
192
193        // If this is the first node we know about, then bootstrap
194        if !self.dht().is_bootstrapped().await {
195            self.dht().set_bootstrapped().await;
196
197            // Lookup our own node id
198            debug!(target: "fud::DhtHandler::on_new_node()", "DHT bootstrapping {}", hash_to_string(&self.dht().node_id));
199            let _ = self.lookup_nodes(&self.dht().node_id).await;
200        }
201
202        // Send keys that are closer to this node than we are
203        let self_id = self.dht().node_id;
204        let channel = self.get_channel(node).await?;
205        for (key, seeders) in self.seeders_router.read().await.iter() {
206            let node_distance = BigUint::from_bytes_be(&self.dht().distance(key, &node.id));
207            let self_distance = BigUint::from_bytes_be(&self.dht().distance(key, &self_id));
208            if node_distance <= self_distance {
209                let _ = channel
210                    .send(&FudAnnounce {
211                        key: *key,
212                        seeders: seeders.clone().into_iter().collect(),
213                    })
214                    .await;
215            }
216        }
217
218        Ok(())
219    }

221    async fn fetch_nodes(&self, node: &DhtNode, key: &blake3::Hash) -> Result<Vec<DhtNode>> {
222        debug!(target: "fud::DhtHandler::fetch_value()", "Fetching nodes close to {} from node {}", hash_to_string(key), hash_to_string(&node.id));

224        let channel = self.get_channel(node).await?;
225        let msg_subsystem = channel.message_subsystem();
226        msg_subsystem.add_dispatch::<FudFindNodesReply>().await;
227        let msg_subscriber_nodes = channel.subscribe_msg::<FudFindNodesReply>().await.unwrap();
228
229        let request = FudFindNodesRequest { key: *key };
230        channel.send(&request).await?;
231
232        let reply = msg_subscriber_nodes.receive_with_timeout(self.dht().settings.timeout).await?;
233
234        msg_subscriber_nodes.unsubscribe().await;
235
236        Ok(reply.nodes.clone())
237    }

241    /// Add ourselves to `seeders_router` for the files we already have.
242    /// Skipped if we have no external address.
243    async fn init(&self) -> Result<()> {
244        info!(target: "fud::init()", "Finding resources...");
245        let hashes = self.geode.list_files().await?;
246        let mut resources_write = self.resources.write().await;
247        for hash in hashes {
248            resources_write.insert(
249                hash,
250                Resource {
251                    hash,
252                    status: ResourceStatus::Incomplete,
253                    chunks_total: 0,
254                    chunks_downloaded: 0,
255                },
256            );
257        }
258        drop(resources_write);
259
260        info!(target: "fud::init()", "Verifying resources...");
261        let resources = self.get_seeding_resources().await?;
262
263        let self_node = self.dht().node().await;
264
265        if self_node.addresses.is_empty() {
266            return Ok(());
267        }
268
269        info!(target: "fud::init()", "Start seeding...");
270        let self_router_items: Vec<DhtRouterItem> = vec![self_node.into()];
271
272        for resource in resources {
273            self.add_to_router(
274                self.seeders_router.clone(),
275                &resource.hash,
276                self_router_items.clone(),
277            )
278            .await;
279        }
280
281        Ok(())
282    }

284    /// Verify if resources are complete and uncorrupted.
285    /// If a resource is incomplete or corrupted, its status is changed to Incomplete.
286    /// If a resource is complete, its status is changed to Seeding.
287    /// Returns the list of verified and uncorrupted/complete seeding resources.
288    async fn get_seeding_resources(&self) -> Result<Vec<Resource>> {
289        let mut resources_write = self.resources.write().await;
290
291        let update_resource =
292            async |resource: &mut Resource,
293                   status: ResourceStatus,
294                   chunked_file: Option<&ChunkedFile>| {
295                resource.status = status;
296                resource.chunks_total = match chunked_file {
297                    Some(chunked_file) => chunked_file.len() as u64,
298                    None => 0,
299                };
300                resource.chunks_downloaded = match chunked_file {
301                    Some(chunked_file) => chunked_file.local_chunks() as u64,
302                    None => 0,
303                };
304            };
305
306        let mut seeding_resources: Vec<Resource> = vec![];
307        for (_, mut resource) in resources_write.iter_mut() {
308            match resource.status {
309                ResourceStatus::Seeding => {}
310                ResourceStatus::Incomplete => {}
311                _ => continue,
312            };
313
314            // Make sure the resource is not corrupted or incomplete
315            let chunked_file = match self.geode.get(&resource.hash).await {
316                Ok(v) => v,
317                Err(_) => {
318                    update_resource(&mut resource, ResourceStatus::Incomplete, None).await;
319                    continue;
320                }
321            };
322            if !chunked_file.is_complete() {
323                update_resource(&mut resource, ResourceStatus::Incomplete, Some(&chunked_file))
324                    .await;
325                continue;
326            }
327
328            update_resource(&mut resource, ResourceStatus::Seeding, Some(&chunked_file)).await;
329            seeding_resources.push(resource.clone());
330        }
331
332        Ok(seeding_resources)
333    }

335    /// Query `nodes` to find the seeders for `key`
336    async fn fetch_seeders(
337        &self,
338        nodes: &Vec<DhtNode>,
339        key: &blake3::Hash,
340    ) -> HashSet<DhtRouterItem> {
341        let mut seeders: HashSet<DhtRouterItem> = HashSet::new();
342
343        for node in nodes {
344            let channel = match self.get_channel(node).await {
345                Ok(channel) => channel,
346                Err(e) => {
347                    warn!(target: "fud::fetch_seeders()", "Could not get a channel for node {}: {}", hash_to_string(&node.id), e);
348                    continue;
349                }
350            };
351            let msg_subsystem = channel.message_subsystem();
352            msg_subsystem.add_dispatch::<FudFindSeedersReply>().await;
353
354            let msg_subscriber = match channel.subscribe_msg::<FudFindSeedersReply>().await {
355                Ok(msg_subscriber) => msg_subscriber,
356                Err(e) => {
357                    warn!(target: "fud::fetch_seeders()", "Error subscribing to msg: {}", e);
358                    continue;
359                }
360            };
361
362            let send_res = channel.send(&FudFindSeedersRequest { key: *key }).await;
363            if let Err(e) = send_res {
364                warn!(target: "fud::fetch_seeders()", "Error while sending FudFindSeedersRequest: {}", e);
365                msg_subscriber.unsubscribe().await;
366                continue;
367            }
368
369            let reply = match msg_subscriber.receive_with_timeout(self.dht().settings.timeout).await
370            {
371                Ok(reply) => reply,
372                Err(e) => {
373                    warn!(target: "fud::fetch_seeders()", "Error waiting for reply: {}", e);
374                    msg_subscriber.unsubscribe().await;
375                    continue;
376                }
377            };
378
379            msg_subscriber.unsubscribe().await;
380
381            seeders.extend(reply.seeders.clone());
382        }
383
384        info!(target: "fud::fetch_seeders()", "Found {} seeders for {}", seeders.len(), hash_to_string(key));
385        seeders
386    }

388    /// Fetch chunks for a file from `seeders`
389    async fn fetch_chunks(
390        &self,
391        file_hash: &blake3::Hash,
392        chunk_hashes: &HashSet<blake3::Hash>,
393        seeders: &HashSet<DhtRouterItem>,
394    ) -> Result<()> {
395        let mut remaining_chunks = chunk_hashes.clone();
396        let mut shuffled_seeders = {
397            let mut vec: Vec<_> = seeders.iter().cloned().collect();
398            vec.shuffle(&mut OsRng);
399            vec
400        };
401
402        while let Some(seeder) = shuffled_seeders.pop() {
403            let channel = match self.get_channel(&seeder.node).await {
404                Ok(channel) => channel,
405                Err(e) => {
406                    warn!(target: "fud::fetch_chunks()", "Could not get a channel for node {}: {}", hash_to_string(&seeder.node.id), e);
407                    continue;
408                }
409            };
410            info!("Requesting chunks from seeder {}", hash_to_string(&seeder.node.id));
411            loop {
412                let msg_subsystem = channel.message_subsystem();
413                msg_subsystem.add_dispatch::<FudChunkReply>().await;
414                msg_subsystem.add_dispatch::<FudNotFound>().await;
415                let msg_subscriber_chunk = channel.subscribe_msg::<FudChunkReply>().await.unwrap();
416                let msg_subscriber_notfound = channel.subscribe_msg::<FudNotFound>().await.unwrap();
417
418                let mut chunks_to_query = remaining_chunks.clone();
419
420                // Select a chunk to request
421                let mut chunk_hash: Option<blake3::Hash> = None;
422                if let Some(&random_chunk) = chunks_to_query.iter().choose(&mut OsRng) {
423                    chunk_hash = Some(random_chunk);
424                }
425
426                if chunk_hash.is_none() {
427                    // No more chunks to request from this seeder
428                    break; // Switch to another seeder
429                }
430                let chunk_hash = chunk_hash.unwrap();
431
432                let send_res = channel.send(&FudFindRequest { key: chunk_hash }).await;
433                if let Err(e) = send_res {
434                    warn!(target: "fud::fetch_chunks()", "Error while sending FudFindRequest: {}", e);
435                    break; // Switch to another seeder
436                }
437
438                let chunk_recv =
439                    msg_subscriber_chunk.receive_with_timeout(self.chunk_timeout).fuse();
440                let notfound_recv =
441                    msg_subscriber_notfound.receive_with_timeout(self.chunk_timeout).fuse();
442
443                pin_mut!(chunk_recv, notfound_recv);
444
445                // Wait for a FudChunkReply or FudNotFound
446                select! {
447                    chunk_reply = chunk_recv => {
448                        if let Err(e) = chunk_reply {
449                            warn!(target: "fud::fetch_chunks()", "Error waiting for chunk reply: {}", e);
450                            break; // Switch to another seeder
451                        }
452                        chunks_to_query.remove(&chunk_hash);
453                        let reply = chunk_reply.unwrap();
454
455                        match self.geode.insert_chunk(&reply.chunk).await {
456                            Ok(inserted_hash) => {
457                                if inserted_hash != chunk_hash {
458                                    warn!("Received chunk does not match requested chunk");
459                                    msg_subscriber_chunk.unsubscribe().await;
460                                    msg_subscriber_notfound.unsubscribe().await;
461                                    continue; // Skip to next chunk, will retry this chunk later
462                                }
463
464                                // Upade resource `chunks_downloaded`
465                                let mut resources_write = self.resources.write().await;
466                                let resource = match resources_write.get_mut(file_hash) {
467                                    Some(resource) => {
468                                        resource.status = ResourceStatus::Downloading;
469                                        resource.chunks_downloaded += 1;
470                                        resource.clone()
471                                    }
472                                    None => return Ok(()) // Resource was removed, abort
473                                };
474                                drop(resources_write);
475
476                                info!(target: "fud::fetch_chunks()", "Received chunk {} from seeder {}", hash_to_string(&chunk_hash), hash_to_string(&seeder.node.id));
477                                self.event_publisher
478                                    .notify(FudEvent::ChunkDownloadCompleted(ChunkDownloadCompleted {
479                                        file_hash: *file_hash,
480                                        chunk_hash,
481                                        resource,
482                                    }))
483                                    .await;
484                                remaining_chunks.remove(&chunk_hash);
485                            }
486                            Err(e) => {
487                                error!("Failed inserting chunk {} to Geode: {}", hash_to_string(&chunk_hash), e);
488                            }
489                        };
490                    }
491                    notfound_reply = notfound_recv => {
492                        if let Err(e) = notfound_reply {
493                            warn!(target: "fud::fetch_chunks()", "Error waiting for NOTFOUND reply: {}", e);
494                            msg_subscriber_chunk.unsubscribe().await;
495                            msg_subscriber_notfound.unsubscribe().await;
496                            break; // Switch to another seeder
497                        }
498                        info!(target: "fud::fetch_chunks()", "Received NOTFOUND {} from seeder {}", hash_to_string(&chunk_hash), hash_to_string(&seeder.node.id));
499                        self.event_publisher
500                            .notify(FudEvent::ChunkNotFound(ChunkNotFound {
501                                file_hash: *file_hash,
502                                chunk_hash,
503                            }))
504                        .await;
505                        chunks_to_query.remove(&chunk_hash);
506                    }
507                };
508
509                msg_subscriber_chunk.unsubscribe().await;
510                msg_subscriber_notfound.unsubscribe().await;
511            }
512
513            // Stop when there are no missing chunks
514            if remaining_chunks.is_empty() {
515                break;
516            }
517        }
518
519        Ok(())
520    }

522    /// Fetch a single file metadata from `nodes`.
523    /// If the file is smaller than a single chunk then the chunk is returned.
524    /// 1. Request seeders for the file from those nodes
525    /// 2. Request the file from the seeders
526    async fn fetch_file_metadata(
527        &self,
528        nodes: Vec<DhtNode>,
529        file_hash: blake3::Hash,
530    ) -> Option<FetchReply> {
531        let mut queried_seeders: HashSet<blake3::Hash> = HashSet::new();
532        let mut result: Option<FetchReply> = None;
533
534        for node in nodes {
535            // 1. Request list of seeders
536            let channel = match self.get_channel(&node).await {
537                Ok(channel) => channel,
538                Err(e) => {
539                    warn!(target: "fud::fetch_file_metadata()", "Could not get a channel for node {}: {}", hash_to_string(&node.id), e);
540                    continue;
541                }
542            };
543            let msg_subsystem = channel.message_subsystem();
544            msg_subsystem.add_dispatch::<FudFindSeedersReply>().await;
545
546            let msg_subscriber = match channel.subscribe_msg::<FudFindSeedersReply>().await {
547                Ok(msg_subscriber) => msg_subscriber,
548                Err(e) => {
549                    warn!(target: "fud::fetch_file_metadata()", "Error subscribing to msg: {}", e);
550                    continue;
551                }
552            };
553
554            let send_res = channel.send(&FudFindSeedersRequest { key: file_hash }).await;
555            if let Err(e) = send_res {
556                warn!(target: "fud::fetch_file_metadata()", "Error while sending FudFindSeedersRequest: {}", e);
557                msg_subscriber.unsubscribe().await;
558                continue;
559            }
560
561            let reply = match msg_subscriber.receive_with_timeout(self.dht().settings.timeout).await
562            {
563                Ok(reply) => reply,
564                Err(e) => {
565                    warn!(target: "fud::fetch_file_metadata()", "Error waiting for reply: {}", e);
566                    msg_subscriber.unsubscribe().await;
567                    continue;
568                }
569            };
570
571            let mut seeders = reply.seeders.clone();
572            info!(target: "fud::fetch_file_metadata()", "Found {} seeders for {} (from {})", seeders.len(), hash_to_string(&file_hash), hash_to_string(&node.id));
573
574            msg_subscriber.unsubscribe().await;
575
576            // 2. Request the file/chunk from the seeders
577            while let Some(seeder) = seeders.pop() {
578                // Only query a seeder once
579                if queried_seeders.iter().any(|s| *s == seeder.node.id) {
580                    continue;
581                }
582                queried_seeders.insert(seeder.node.id);
583
584                if let Ok(channel) = self.get_channel(&seeder.node).await {
585                    let msg_subsystem = channel.message_subsystem();
586                    msg_subsystem.add_dispatch::<FudChunkReply>().await;
587                    msg_subsystem.add_dispatch::<FudFileReply>().await;
588                    msg_subsystem.add_dispatch::<FudNotFound>().await;
589                    let msg_subscriber_chunk =
590                        channel.subscribe_msg::<FudChunkReply>().await.unwrap();
591                    let msg_subscriber_file =
592                        channel.subscribe_msg::<FudFileReply>().await.unwrap();
593                    let msg_subscriber_notfound =
594                        channel.subscribe_msg::<FudNotFound>().await.unwrap();
595
596                    let send_res = channel.send(&FudFindRequest { key: file_hash }).await;
597                    if let Err(e) = send_res {
598                        warn!(target: "fud::fetch_file_metadata()", "Error while sending FudFindRequest: {}", e);
599                        msg_subscriber_chunk.unsubscribe().await;
600                        msg_subscriber_file.unsubscribe().await;
601                        msg_subscriber_notfound.unsubscribe().await;
602                        continue;
603                    }
604
605                    let chunk_recv =
606                        msg_subscriber_chunk.receive_with_timeout(self.chunk_timeout).fuse();
607                    let file_recv =
608                        msg_subscriber_file.receive_with_timeout(self.chunk_timeout).fuse();
609                    let notfound_recv =
610                        msg_subscriber_notfound.receive_with_timeout(self.chunk_timeout).fuse();
611
612                    pin_mut!(chunk_recv, file_recv, notfound_recv);
613
614                    let cleanup = async || {
615                        msg_subscriber_chunk.unsubscribe().await;
616                        msg_subscriber_file.unsubscribe().await;
617                        msg_subscriber_notfound.unsubscribe().await;
618                    };
619
620                    // Wait for a FudChunkReply, FudFileReply, or FudNotFound
621                    select! {
622                        // Received a chunk while requesting a file, this is allowed to
623                        // optimize fetching files smaller than a single chunk
624                        chunk_reply = chunk_recv => {
625                            cleanup().await;
626                            if let Err(e) = chunk_reply {
627                                warn!(target: "fud::fetch_file_metadata()", "Error waiting for chunk reply: {}", e);
628                                continue;
629                            }
630                            let reply = chunk_reply.unwrap();
631                            let chunk_hash = blake3::hash(&reply.chunk);
632                            // Check that this is the only chunk in the file
633                            if !self.geode.verify_file(&file_hash, &[chunk_hash]) {
634                                warn!(target: "fud::fetch_file_metadata()", "Received a chunk while fetching a file, the chunk did not match the file hash");
635                                continue;
636                            }
637                            info!(target: "fud::fetch_file_metadata()", "Received chunk {} (for file {}) from seeder {}", hash_to_string(&chunk_hash), hash_to_string(&file_hash), hash_to_string(&seeder.node.id));
638                            result = Some(FetchReply::Chunk((*reply).clone()));
639                            break;
640                        }
641                        file_reply = file_recv => {
642                            cleanup().await;
643                            if let Err(e) = file_reply {
644                                warn!(target: "fud::fetch_file_metadata()", "Error waiting for file reply: {}", e);
645                                continue;
646                            }
647                            let reply = file_reply.unwrap();
648                            if !self.geode.verify_file(&file_hash, &reply.chunk_hashes) {
649                                warn!(target: "fud::fetch_file_metadata()", "Received invalid file metadata");
650                                continue;
651                            }
652                            info!(target: "fud::fetch_file_metadata()", "Received file {} from seeder {}", hash_to_string(&file_hash), hash_to_string(&seeder.node.id));
653                            result = Some(FetchReply::File((*reply).clone()));
654                            break;
655                        }
656                        notfound_reply = notfound_recv => {
657                            cleanup().await;
658                            if let Err(e) = notfound_reply {
659                                warn!(target: "fud::fetch_file_metadata()", "Error waiting for NOTFOUND reply: {}", e);
660                                continue;
661                            }
662                            info!(target: "fud::fetch_file_metadata()", "Received NOTFOUND {} from seeder {}", hash_to_string(&file_hash), hash_to_string(&seeder.node.id));
663                        }
664                    };
665                }
666            }
667
668            if result.is_some() {
669                break;
670            }
671        }
672
673        result
674    }
675}

678fn generate_node_id() -> Result<blake3::Hash> {
679    let mut rng = OsRng;
680    let mut random_data = [0u8; 32];
681    rng.fill_bytes(&mut random_data);
682    let node_id = blake3::Hash::from_bytes(random_data);
683    Ok(node_id)
684}

686async_daemonize!(realmain);
687async fn realmain(args: Args, ex: Arc<Executor<'static>>) -> Result<()> {
688    // The working directory for this daemon and geode.
689    let basedir = expand_path(&args.base_dir)?;
690
691    // The directory to store the downloaded files
692    let downloads_path = match args.downloads_path {
693        Some(downloads_path) => expand_path(&downloads_path)?,
694        None => basedir.join("downloads"),
695    };
696
697    // Hashmap used for routing
698    let seeders_router = Arc::new(RwLock::new(HashMap::new()));
699
700    info!("Instantiating Geode instance");
701    let geode = Geode::new(&basedir).await?;
702
703    info!("Instantiating P2P network");
704    let net_settings: NetSettings = args.net.into();
705    let p2p = P2p::new(net_settings.clone(), ex.clone()).await?;
706
707    info!("Starting dnet subs task");
708    let dnet_sub = JsonSubscriber::new("dnet.subscribe_events");
709    let dnet_sub_ = dnet_sub.clone();
710    let p2p_ = p2p.clone();
711    let dnet_task = StoppableTask::new();
712    dnet_task.clone().start(
713        async move {
714            let dnet_sub = p2p_.dnet_subscribe().await;
715            loop {
716                let event = dnet_sub.receive().await;
717                debug!("Got dnet event: {:?}", event);
718                dnet_sub_.notify(vec![event.into()].into()).await;
719            }
720        },
721        |res| async {
722            match res {
723                Ok(()) | Err(Error::DetachedTaskStopped) => { /* Do nothing */ }
724                Err(e) => panic!("{}", e),
725            }
726        },
727        Error::DetachedTaskStopped,
728        ex.clone(),
729    );
730
731    // Get or generate the node id
732    let node_id: Result<blake3::Hash> = {
733        let mut node_id_path: PathBuf = basedir.clone();
734        node_id_path.push(NODE_ID_PATH);
735        match File::open(node_id_path.clone()).await {
736            Ok(mut file) => {
737                let mut buffer = Vec::new();
738                file.read_to_end(&mut buffer).await?;
739                let mut out_buf = [0u8; 32];
740                bs58::decode(buffer).onto(&mut out_buf)?;
741                let node_id = blake3::Hash::from_bytes(out_buf);
742                Ok(node_id)
743            }
744            Err(e) if e.kind() == ErrorKind::NotFound => {
745                let node_id = generate_node_id()?;
746                let mut file =
747                    OpenOptions::new().write(true).create(true).open(node_id_path).await?;
748                file.write_all(&bs58::encode(node_id.as_bytes()).into_vec()).await?;
749                file.flush().await?;
750                Ok(node_id)
751            }
752            Err(e) => Err(e.into()),
753        }
754    };
755
756    let node_id_ = node_id?;
757
758    info!(target: "fud", "Your node ID: {}", hash_to_string(&node_id_));
759
760    // Daemon instantiation
761    let event_sub = JsonSubscriber::new("event");
762    let (get_tx, get_rx) = smol::channel::unbounded();
763    let (file_fetch_tx, file_fetch_rx) = smol::channel::unbounded();
764    let (file_fetch_end_tx, file_fetch_end_rx) = smol::channel::unbounded();
765    let dht_settings: DhtSettings = args.dht.into();
766    let dht: Arc<Dht> = Arc::new(Dht::new(&node_id_, &dht_settings, p2p.clone(), ex.clone()).await);
767    let fud = Arc::new(Fud {
768        seeders_router,
769        p2p: p2p.clone(),
770        geode,
771        downloads_path,
772        chunk_timeout: args.chunk_timeout,
773        dht: dht.clone(),
774        resources: Arc::new(RwLock::new(HashMap::new())),
775        get_tx,
776        get_rx,
777        file_fetch_tx,
778        file_fetch_rx,
779        file_fetch_end_tx,
780        file_fetch_end_rx,
781        rpc_connections: Mutex::new(HashSet::new()),
782        dnet_sub,
783        event_sub: event_sub.clone(),
784        event_publisher: Publisher::new(),
785    });
786    fud.init().await?;
787
788    info!(target: "fud", "Starting download subs task");
789    let event_sub_ = event_sub.clone();
790    let fud_ = fud.clone();
791    let event_task = StoppableTask::new();
792    event_task.clone().start(
793        async move {
794            let event_sub = fud_.event_publisher.clone().subscribe().await;
795            loop {
796                let event = event_sub.receive().await;
797                debug!(target: "fud", "Got event: {:?}", event);
798                event_sub_.notify(event.into()).await;
799            }
800        },
801        |res| async {
802            match res {
803                Ok(()) | Err(Error::DetachedTaskStopped) => { /* Do nothing */ }
804                Err(e) => panic!("{}", e),
805            }
806        },
807        Error::DetachedTaskStopped,
808        ex.clone(),
809    );
810
811    info!(target: "fud", "Starting fetch file task");
812    let file_task = StoppableTask::new();
813    file_task.clone().start(
814        tasks::fetch_file_task(fud.clone()),
815        |res| async {
816            match res {
817                Ok(()) | Err(Error::DetachedTaskStopped) => { /* Do nothing */ }
818                Err(e) => error!(target: "fud", "Failed starting fetch file task: {}", e),
819            }
820        },
821        Error::DetachedTaskStopped,
822        ex.clone(),
823    );
824
825    info!(target: "fud", "Starting get task");
826    let get_task_ = StoppableTask::new();
827    get_task_.clone().start(
828        tasks::get_task(fud.clone()),
829        |res| async {
830            match res {
831                Ok(()) | Err(Error::DetachedTaskStopped) => { /* Do nothing */ }
832                Err(e) => error!(target: "fud", "Failed starting get task: {}", e),
833            }
834        },
835        Error::DetachedTaskStopped,
836        ex.clone(),
837    );
838
839    let rpc_settings: RpcSettings = args.rpc.into();
840    info!(target: "fud", "Starting JSON-RPC server on {}", rpc_settings.listen);
841    let rpc_task = StoppableTask::new();
842    let fud_ = fud.clone();
843    rpc_task.clone().start(
844        listen_and_serve(rpc_settings, fud.clone(), None, ex.clone()),
845        |res| async move {
846            match res {
847                Ok(()) | Err(Error::RpcServerStopped) => fud_.stop_connections().await,
848                Err(e) => error!(target: "fud", "Failed starting sync JSON-RPC server: {}", e),
849            }
850        },
851        Error::RpcServerStopped,
852        ex.clone(),
853    );
854
855    info!("Starting P2P protocols");
856    let registry = p2p.protocol_registry();
857    let fud_ = fud.clone();
858    registry
859        .register(SESSION_DEFAULT, move |channel, p2p| {
860            let fud_ = fud_.clone();
861            async move { ProtocolFud::init(fud_, channel, p2p).await.unwrap() }
862        })
863        .await;
864    p2p.clone().start().await?;
865
866    let p2p_settings_lock = p2p.settings();
867    let p2p_settings = p2p_settings_lock.read().await;
868    if p2p_settings.external_addrs.is_empty() {
869        warn!(target: "fud::realmain", "No external addresses, you won't be able to seed")
870    }
871    drop(p2p_settings);
872
873    info!(target: "fud", "Starting DHT tasks");
874    let dht_channel_task = StoppableTask::new();
875    let fud_ = fud.clone();
876    dht_channel_task.clone().start(
877        async move { fud_.channel_task::<FudFindNodesReply>().await },
878        |res| async {
879            match res {
880                Ok(()) | Err(Error::DetachedTaskStopped) => { /* Do nothing */ }
881                Err(e) => error!(target: "fud", "Failed starting dht channel task: {}", e),
882            }
883        },
884        Error::DetachedTaskStopped,
885        ex.clone(),
886    );
887    let dht_disconnect_task = StoppableTask::new();
888    let fud_ = fud.clone();
889    dht_disconnect_task.clone().start(
890        async move { fud_.disconnect_task().await },
891        |res| async {
892            match res {
893                Ok(()) | Err(Error::DetachedTaskStopped) => { /* Do nothing */ }
894                Err(e) => error!(target: "fud", "Failed starting dht disconnect task: {}", e),
895            }
896        },
897        Error::DetachedTaskStopped,
898        ex.clone(),
899    );
900    let announce_task = StoppableTask::new();
901    let fud_ = fud.clone();
902    announce_task.clone().start(
903        async move { tasks::announce_seed_task(fud_.clone()).await },
904        |res| async {
905            match res {
906                Ok(()) | Err(Error::DetachedTaskStopped) => { /* Do nothing */ }
907                Err(e) => error!(target: "fud", "Failed starting announce task: {}", e),
908            }
909        },
910        Error::DetachedTaskStopped,
911        ex.clone(),
912    );
913
914    // Signal handling for graceful termination.
915    let (signals_handler, signals_task) = SignalHandler::new(ex)?;
916    signals_handler.wait_termination(signals_task).await?;
917    info!(target: "fud", "Caught termination signal, cleaning up and exiting...");
918
919    info!(target: "fud", "Stopping fetch file task...");
920    file_task.stop().await;
921
922    info!(target: "fud", "Stopping get task...");
923    get_task_.stop().await;
924
925    info!(target: "fud", "Stopping JSON-RPC server...");
926    rpc_task.stop().await;
927
928    info!(target: "fud", "Stopping P2P network...");
929    p2p.stop().await;
930
931    info!(target: "fud", "Stopping DHT tasks");
932    dht_channel_task.stop().await;
933    dht_disconnect_task.stop().await;
934    announce_task.stop().await;
935
936    info!("Bye!");
937    Ok(())
938}
