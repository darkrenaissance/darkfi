49/// Create a new database instance for the calling contract.
50///
51/// This function expects to receive a pointer from which a `ContractId`
52/// and the `db_name` will be read.
53///
54/// This function should **only** be allowed in `ContractSection::Deploy`, as that
55/// is called when a contract is being (re)deployed and databases have to be created.
56///
57/// Permissions: deploy
58pub(crate) fn db_init(mut ctx: FunctionEnvMut<Env>, ptr: WasmPtr<u8>, ptr_len: u32) -> i64 {
59    let (env, mut store) = ctx.data_and_store_mut();
60    let cid = env.contract_id;
61
62    // Enforce function ACL
63    if let Err(e) = acl_allow(env, &[ContractSection::Deploy]) {
64        error!(
65            target: "runtime::db::db_init",
66            "[WASM] [{cid}] db_init(): Called in unauthorized section: {e}"
67        );
68        return darkfi_sdk::error::CALLER_ACCESS_DENIED
69    }
70
71    // Subtract used gas.
72    // TODO: There should probably be an additional fee to open a new sled tree.
73    env.subtract_gas(&mut store, 1);
74
75    // This takes lock of the blockchain overlay reference in the wasm env
76    let contracts = &env.blockchain.lock().unwrap().contracts;
77
78    // Create a mem slice of the wasm VM memory
79    let memory_view = env.memory_view(&store);
80    let Ok(mem_slice) = ptr.slice(&memory_view, ptr_len) else {
81        error!(
82            target: "runtime::db::db_init",
83            "[WASM] [{cid}] db_init(): Failed to make slice from ptr"
84        );
85        return darkfi_sdk::error::DB_INIT_FAILED
86    };
87
88    // Allocate a buffer and copy all the data from the pointer into the buffer
89    let mut buf = vec![0_u8; ptr_len as usize];
90    if let Err(e) = mem_slice.read_slice(&mut buf) {
91        error!(
92            target: "runtime::db::db_init",
93            "[WASM] [{cid}] db_init(): Failed to read memory slice: {e}"
94        );
95        return darkfi_sdk::error::DB_INIT_FAILED
96    };
97
98    // Once the data is copied, we'll attempt to deserialize it into the objects
99    // we're expecting.
100    let mut buf_reader = Cursor::new(buf);
101    let read_cid: ContractId = match Decodable::decode(&mut buf_reader) {
102        Ok(v) => v,
103        Err(e) => {
104            error!(
105                target: "runtime::db::db_init",
106                "[WASM] [{cid}] db_init(): Failed decoding ContractId: {e}"
107            );
108            return darkfi_sdk::error::DB_INIT_FAILED
109        }
110    };
111
112    let read_db_name: String = match Decodable::decode(&mut buf_reader) {
113        Ok(v) => v,
114        Err(e) => {
115            error!(
116                target: "runtime::db::db_init",
117                "[WASM] [{cid}] db_init(): Failed decoding db_name: {e}"
118            );
119            return darkfi_sdk::error::DB_INIT_FAILED
120        }
121    };
122
123    // Make sure we've read the entire buffer
124    if buf_reader.position() != ptr_len as u64 {
125        error!(
126            target: "runtime::db::db_init",
127            "[WASM] [{cid}] db_init(): Trailing bytes in argument stream"
128        );
129        return darkfi_sdk::error::DB_INIT_FAILED
130    }
131
132    // We cannot allow initializing the special zkas db:
133    if read_db_name == SMART_CONTRACT_ZKAS_DB_NAME {
134        error!(
135            target: "runtime::db::db_init",
136            "[WASM] [{cid}] db_init(): Attempted to init zkas db"
137        );
138        return darkfi_sdk::error::CALLER_ACCESS_DENIED
139    }
140
141    // Nor can we allow another contract to initialize a db for someone else:
142    if cid != read_cid {
143        error!(
144            target: "runtime::db::db_init",
145            "[WASM] [{cid}] db_init(): Unauthorized ContractId"
146        );
147        return darkfi_sdk::error::CALLER_ACCESS_DENIED
148    }
149
150    // Now try to initialize the tree. If this returns an error,
151    // it usually means that this DB was already initialized.
152    // An alternative error might happen if something in sled fails,
153    // for this we should take care to stop the node or do something to
154    // be able to gracefully recover.
155    // (src/blockchain/contract_store.rs holds this init() function)
156    let tree_handle = match contracts.init(&read_cid, &read_db_name) {
157        Ok(v) => v,
158        Err(e) => {
159            error!(
160                target: "runtime::db::db_init",
161                "[WASM] [{cid}] db_init(): Failed to init db: {e}"
162            );
163            return darkfi_sdk::error::DB_INIT_FAILED
164        }
165    };
166
167    // Create the DbHandle
168    let db_handle = DbHandle::new(read_cid, tree_handle);
169    let mut db_handles = env.db_handles.borrow_mut();
170
171    // Make sure we don't duplicate the DbHandle in the vec.
172    // It's not really an issue, but it's better to be pedantic.
173    if db_handles.contains(&db_handle) {
174        error!(
175            target: "runtime::db::db_init",
176            "[WASM] [{cid}] db_init(): DbHandle initialized twice during execution"
177        );
178        return darkfi_sdk::error::DB_INIT_FAILED
179    }
180
181    // This tries to cast into u32
182    match db_handles.len().try_into() {
183        Ok(db_handle_idx) => {
184            db_handles.push(db_handle);
185            // Return the db handle index
186            db_handle_idx
187        }
188        Err(_) => {
189            error!(
190                target: "runtime::db::db_init",
191                "[WASM] [{cid}] db_init(): Too many open DbHandles"
192            );
193            darkfi_sdk::error::DB_INIT_FAILED
194        }
195    }
196}

198/// Lookup a database handle from its name.
199/// If it exists, push it to the Vector of db_handles.
200///
201/// Returns the index of the DbHandle in the db_handles Vector on success.
202/// Otherwise, returns an error value.
203///
204/// This function can be called from any [`ContractSection`].
205///
206/// Permissions: deploy, metadata, exec, update
207pub(crate) fn db_lookup(mut ctx: FunctionEnvMut<Env>, ptr: WasmPtr<u8>, ptr_len: u32) -> i64 {
208    let (env, mut store) = ctx.data_and_store_mut();
209    let cid = env.contract_id;
210
211    // Enforce function ACL
212    if let Err(e) = acl_allow(
213        env,
214        &[
215            ContractSection::Deploy,
216            ContractSection::Metadata,
217            ContractSection::Exec,
218            ContractSection::Update,
219        ],
220    ) {
221        error!(
222            target: "runtime::db::db_lookup",
223            "[WASM] [{cid}] db_lookup() called in unauthorized section: {e}"
224        );
225        return darkfi_sdk::error::CALLER_ACCESS_DENIED
226    }
227
228    // Subtract used gas. Opening an existing db should be free (i.e. 1 gas unit).
229    env.subtract_gas(&mut store, 1);
230
231    // Read memory location that contains the ContractId and DB name
232    let memory_view = env.memory_view(&store);
233    let contracts = &env.blockchain.lock().unwrap().contracts;
234
235    let Ok(mem_slice) = ptr.slice(&memory_view, ptr_len) else {
236        error!(
237            target: "runtime::db::db_lookup",
238            "[WASM] [{cid}] db_lookup(): Failed to make slice from ptr."
239        );
240        return darkfi_sdk::error::DB_LOOKUP_FAILED
241    };
242
243    let mut buf = vec![0_u8; ptr_len as usize];
244    if let Err(e) = mem_slice.read_slice(&mut buf) {
245        error!(
246            target: "runtime::db::db_lookup",
247            "[WASM] [{cid}] db_lookup(): Failed to read from memory slice: {e}"
248        );
249        return darkfi_sdk::error::DB_LOOKUP_FAILED
250    };
251
252    // Wrap the buffer into a Cursor for stream reading
253    let mut buf_reader = Cursor::new(buf);
254
255    // Decode ContractId from memory
256    let cid: ContractId = match Decodable::decode(&mut buf_reader) {
257        Ok(v) => v,
258        Err(e) => {
259            error!(
260                target: "runtime::db::db_lookup",
261                "[WASM] [{cid}] db_lookup(): Failed to decode ContractId: {e}"
262            );
263            return darkfi_sdk::error::DB_LOOKUP_FAILED
264        }
265    };
266
267    // Decode DB name from memory
268    let db_name: String = match Decodable::decode(&mut buf_reader) {
269        Ok(v) => v,
270        Err(e) => {
271            error!(
272                target: "runtime::db::db_lookup",
273                "[WASM] [{cid}] db_lookup(): Failed to decode db_name: {e}"
274            );
275            return darkfi_sdk::error::DB_LOOKUP_FAILED
276        }
277    };
278
279    // Make sure we've read the entire buffer
280    if buf_reader.position() != ptr_len as u64 {
281        error!(
282            target: "runtime::db::db_lookup",
283            "[WASM] [{cid}] db_lookup(), Trailing bytes in argument stream"
284        );
285        return darkfi_sdk::error::DB_LOOKUP_FAILED
286    }
287
288    if db_name == SMART_CONTRACT_ZKAS_DB_NAME {
289        error!(
290            target: "runtime::db::db_lookup",
291            "[WASM] [{cid}] db_lookup(): Attempted to lookup zkas db"
292        );
293        return darkfi_sdk::error::CALLER_ACCESS_DENIED
294    }
295
296    // Lookup contract state
297    let tree_handle = match contracts.lookup(&cid, &db_name) {
298        Ok(v) => v,
299        Err(_) => return darkfi_sdk::error::DB_LOOKUP_FAILED,
300    };
301
302    // Create the DbHandle
303    let db_handle = DbHandle::new(cid, tree_handle);
304    let mut db_handles = env.db_handles.borrow_mut();
305
306    // Make sure we don't duplicate the DbHandle in the vec
307    if let Some(index) = db_handles.iter().position(|x| x == &db_handle) {
308        return index as i64
309    }
310
311    // Push the new DbHandle to the Vec of opened DbHandles
312    match db_handles.len().try_into() {
313        Ok(db_handle_idx) => {
314            db_handles.push(db_handle);
315            db_handle_idx
316        }
317        Err(_) => {
318            error!(
319                target: "runtime::db::db_lookup",
320                "[WASM] [{cid}] db_lookup(): Too many open DbHandles"
321            );
322            darkfi_sdk::error::DB_LOOKUP_FAILED
323        }
324    }
325}

327/// Set a value within the transaction.
328///
329/// * `ptr` must contain the DbHandle index and the key-value pair.
330/// * The DbHandle must match the ContractId.
331///
332/// This function can be called only from the Deploy or Update [`ContractSection`].
333/// Returns `SUCCESS` on success, otherwise returns an error value.
334///
335/// Permissions: deploy, update
336pub(crate) fn db_set(mut ctx: FunctionEnvMut<Env>, ptr: WasmPtr<u8>, ptr_len: u32) -> i64 {
337    let (env, mut store) = ctx.data_and_store_mut();
338    let cid = env.contract_id;
339
340    if let Err(e) = acl_allow(env, &[ContractSection::Deploy, ContractSection::Update]) {
341        error!(
342            target: "runtime::db::db_set",
343            "[WASM] [{cid}] db_set(): Called in unauthorized section: {e}"
344        );
345        return darkfi_sdk::error::CALLER_ACCESS_DENIED
346    }
347
348    // Subtract used gas. Here we count the bytes written into the database.
349    // TODO: We might want to count only the difference in size if we're replacing
350    // data and the new data is larger.
351    env.subtract_gas(&mut store, ptr_len as u64);
352
353    // Ensure that it is possible to read from the memory that this function needs
354    let memory_view = env.memory_view(&store);
355    let Ok(mem_slice) = ptr.slice(&memory_view, ptr_len) else {
356        error!(
357            target: "runtime::db::db_set",
358            "[WASM] [{cid}] db_set(): Failed to make slice from ptr"
359        );
360        return darkfi_sdk::error::DB_SET_FAILED
361    };
362
363    let mut buf = vec![0_u8; ptr_len as usize];
364    if let Err(e) = mem_slice.read_slice(&mut buf) {
365        error!(
366            target: "runtime::db::db_set",
367            "[WASM] [{cid}] db_set(): Failed to read from memory slice: {e}"
368        );
369        return darkfi_sdk::error::DB_SET_FAILED
370    };
371
372    let mut buf_reader = Cursor::new(buf);
373
374    // Decode DbHandle index
375    let db_handle_index: u32 = match Decodable::decode(&mut buf_reader) {
376        Ok(v) => v,
377        Err(e) => {
378            error!(
379                target: "runtime::db::db_set",
380                "[WASM] [{cid}] db_set(): Failed to decode DbHandle: {e}"
381            );
382            return darkfi_sdk::error::DB_SET_FAILED
383        }
384    };
385
386    let db_handle_index = db_handle_index as usize;
387
388    // Decode key and value
389    let key: Vec<u8> = match Decodable::decode(&mut buf_reader) {
390        Ok(v) => v,
391        Err(e) => {
392            error!(
393                target: "runtime::db::db_set",
394                "[WASM] [{cid}] db_set(): Failed to decode key vec: {e}"
395            );
396            return darkfi_sdk::error::DB_SET_FAILED
397        }
398    };
399
400    let value: Vec<u8> = match Decodable::decode(&mut buf_reader) {
401        Ok(v) => v,
402        Err(e) => {
403            error!(
404                target: "runtime::db::db_set",
405                "[WASM] [{cid}] db_set(): Failed to decode value vec: {e}"
406            );
407            return darkfi_sdk::error::DB_SET_FAILED
408        }
409    };
410
411    // Make sure we've read the entire buffer
412    if buf_reader.position() != ptr_len as u64 {
413        error!(
414            target: "runtime::db::db_set",
415            "[WASM] [{cid}] db_set(): Trailing bytes in argument stream"
416        );
417        return darkfi_sdk::error::DB_SET_FAILED
418    }
419
420    let db_handles = env.db_handles.borrow();
421
422    // Check DbHandle index is within bounds
423    if db_handles.len() <= db_handle_index {
424        error!(
425            target: "runtime::db::db_set",
426            "[WASM] [{cid}] db_set(): Requested DbHandle that is out of bounds"
427        );
428        return darkfi_sdk::error::DB_SET_FAILED
429    }
430
431    // Retrive DbHandle using the index
432    let db_handle = &db_handles[db_handle_index];
433
434    // Validate that the DbHandle matches the contract ID
435    if db_handle.contract_id != env.contract_id {
436        error!(
437            target: "runtime::db::db_set",
438            "[WASM] [{cid}] db_set(): Unauthorized to write to DbHandle"
439        );
440        return darkfi_sdk::error::CALLER_ACCESS_DENIED
441    }
442
443    // Insert key-value pair into the database corresponding to this contract
444    if env
445        .blockchain
446        .lock()
447        .unwrap()
448        .overlay
449        .lock()
450        .unwrap()
451        .insert(&db_handle.tree, &key, &value)
452        .is_err()
453    {
454        error!(
455            target: "runtime::db::db_set",
456            "[WASM] [{cid}] db_set(): Couldn't insert to db_handle tree"
457        );
458        return darkfi_sdk::error::DB_SET_FAILED
459    }
460
461    wasm::entrypoint::SUCCESS
462}

464/// Remove a key from the database.
465///

480    }
481
482    // Subtract used gas. We make deletion free.
483    env.subtract_gas(&mut store, 1);

494    };
495

497    if let Err(e) = mem_slice.read_slice(&mut buf) {

499            target: "runtime::db::db_del",

503    };
504
505    let mut buf_reader = Cursor::new(buf);
506
507    // Decode DbHandle index
508    let db_handle_index: u32 = match Decodable::decode(&mut buf_reader) {
509        Ok(v) => v,

512                target: "runtime::db::db_del",

515            return darkfi_sdk::error::DB_DEL_FAILED
516        }
517    };
518    let db_handle_index = db_handle_index as usize;
519
520    // Decode key corresponding to the value that will be deleted
521    let key: Vec<u8> = match Decodable::decode(&mut buf_reader) {
522        Ok(v) => v,
523        Err(e) => {
524            error!(
525                target: "runtime::db::db_del",
526                "[WASM] [{cid}] db_del(): Failed to decode key vec: {e}"
527            );
528            return darkfi_sdk::error::DB_DEL_FAILED
529        }
530    };
531

533    if buf_reader.position() != ptr_len as u64 {
534        error!(
535            target: "runtime::db::db_del",
536            "[WASM] [{cid}] db_del(): Trailing bytes in argument stream"
537        );
538        return darkfi_sdk::error::DB_DEL_FAILED
539    }
540
541    let db_handles = env.db_handles.borrow();
542
543    if db_handles.len() <= db_handle_index {
544        error!(
545            target: "runtime::db::db_del",
546            "[WASM] [{cid}] db_del(): Requested DbHandle that is out of bounds"
547        );
548        return darkfi_sdk::error::DB_DEL_FAILED
549    }
550
551    // Retrive DbHandle using the index
552    let db_handle = &db_handles[db_handle_index];
553
554    // Validate that the DbHandle matches the contract ID
555    if db_handle.contract_id != cid {
556        error!(
557            target: "runtime::db::db_del",
558            "[WASM] [{cid}] db_del(): Unauthorized to write to DbHandle"
559        );
560        return darkfi_sdk::error::CALLER_ACCESS_DENIED
561    }
562
563    // Remove key-value pair from the database corresponding to this contract
564    if env.blockchain.lock().unwrap().overlay.lock().unwrap().remove(&db_handle.tree, &key).is_err()
565    {
566        error!(
567            target: "runtime::db::db_del",
568            "[WASM] [{cid}] db_del(): Couldn't remove key from db_handle tree"
569        );
570        return darkfi_sdk::error::DB_DEL_FAILED
571    }
572
573    wasm::entrypoint::SUCCESS
574}

576/// Reads a value by key from the key-value store.
577///
578/// This function can be called from the Deploy, Exec, or Metadata [`ContractSection`].
579///
580/// On success, returns the length of the `objects` Vector in the environment.
581/// Otherwise, returns an error code.
582///
583/// Permissions: deploy, metadata, exec
584pub(crate) fn db_get(mut ctx: FunctionEnvMut<Env>, ptr: WasmPtr<u8>, ptr_len: u32) -> i64 {
585    let (env, mut store) = ctx.data_and_store_mut();
586    let cid = env.contract_id;
587
588    if let Err(e) =
589        acl_allow(env, &[ContractSection::Deploy, ContractSection::Metadata, ContractSection::Exec])
590    {
591        error!(
592            target: "runtime::db::db_get",
593            "[WASM] [{cid}] db_get(): Called in unauthorized section: {e}"
594        );
595        return darkfi_sdk::error::CALLER_ACCESS_DENIED
596    }
597
598    // Subtract used gas. Reading is free.
599    env.subtract_gas(&mut store, 1);
600
601    // Ensure that it is possible to read memory
602    let memory_view = env.memory_view(&store);
603    let Ok(mem_slice) = ptr.slice(&memory_view, ptr_len) else {
604        error!(
605            target: "runtime::db::db_get",
606            "[WASM] [{cid}] db_get(): Failed to make slice from ptr"
607        );
608        return darkfi_sdk::error::DB_GET_FAILED
609    };
610
611    let mut buf = vec![0_u8; ptr_len as usize];
612    if let Err(e) = mem_slice.read_slice(&mut buf) {
613        error!(
614            target: "runtime::db::db_get",
615            "[WASM] [{cid}] db_get(): Failed to read from memory slice: {e}"
616        );
617        return darkfi_sdk::error::DB_GET_FAILED
618    };
619
620    let mut buf_reader = Cursor::new(buf);
621
622    // Decode DbHandle index
623    let db_handle_index: u32 = match Decodable::decode(&mut buf_reader) {
624        Ok(v) => v,
625        Err(e) => {
626            error!(
627                target: "runtime::db::db_get",
628                "[WASM] [{cid}] db_get(): Failed to decode DbHandle: {e}"
629            );
630            return darkfi_sdk::error::DB_GET_FAILED
631        }
632    };
633
634    let db_handle_index = db_handle_index as usize;
635
636    // Decode key for key-value pair that we wish to retrieve
637    let key: Vec<u8> = match Decodable::decode(&mut buf_reader) {
638        Ok(v) => v,
639        Err(e) => {
640            error!(
641                target: "runtime::db::db_get",
642                "[WASM] [{cid}] db_get(): Failed to decode key from vec: {e}"
643            );
644            return darkfi_sdk::error::DB_GET_FAILED
645        }
646    };
647
648    // Make sure there are no trailing bytes in the buffer. This means we've used all data that was
649    // supplied.
650    if buf_reader.position() != ptr_len as u64 {
651        error!(
652            target: "runtime::db::db_get",
653            "[WASM] [{cid}] db_get(): Trailing bytes in argument stream"
654        );
655        return darkfi_sdk::error::DB_GET_FAILED
656    }
657
658    let db_handles = env.db_handles.borrow();
659
660    // Ensure that the index is within bounds
661    if db_handles.len() <= db_handle_index {
662        error!(
663            target: "runtime::db::db_get",
664            "[WASM] [{cid}] db_get(): Requested DbHandle that is out of bounds"
665        );
666        return darkfi_sdk::error::DB_GET_FAILED
667    }
668
669    // Get DbHandle using db_handle_index
670    let db_handle = &db_handles[db_handle_index];
671
672    // Retrieve data using the `key`
673    let ret =
674        match env.blockchain.lock().unwrap().overlay.lock().unwrap().get(&db_handle.tree, &key) {
675            Ok(v) => v,
676            Err(e) => {
677                error!(
678                    target: "runtime::db::db_get",
679                    "[WASM] [{cid}] db_get(): Internal error getting from tree: {e}"
680                );
681                return darkfi_sdk::error::DB_GET_FAILED
682            }
683        };
684    drop(db_handles);
685
686    // Return special error if the data is empty
687    let Some(return_data) = ret else {
688        debug!(
689            target: "runtime::db::db_get",
690            "[WASM] [{cid}] db_get(): Return data is empty"
691        );
692        return darkfi_sdk::error::DB_GET_EMPTY
693    };
694
695    if return_data.len() > u32::MAX as usize {
696        return darkfi_sdk::error::DATA_TOO_LARGE
697    }
698
699    // Subtract used gas. Here we count the length of the data read from db.
700    env.subtract_gas(&mut store, return_data.len() as u64);
701
702    // Copy the data (Vec<u8>) to the VM by pushing it to the objects Vector.
703    let mut objects = env.objects.borrow_mut();
704    if objects.len() == u32::MAX as usize {
705        return darkfi_sdk::error::DATA_TOO_LARGE
706    }
707
708    // Return the length of the objects Vector.
709    // This is the location of the data that was retrieved and pushed
710    objects.push(return_data.to_vec());
711    (objects.len() - 1) as i64
712}

714/// Check if a database contains a given key.
715///
716/// Returns `1` if the key is found.
717/// Returns `0` if the key is not found and there are no errors.
718/// Otherwise, returns an error code.
719///
720/// Permissions: deploy, metadata, exec
721pub(crate) fn db_contains_key(mut ctx: FunctionEnvMut<Env>, ptr: WasmPtr<u8>, ptr_len: u32) -> i64 {
722    let (env, mut store) = ctx.data_and_store_mut();
723    let cid = env.contract_id;
724
725    if let Err(e) =
726        acl_allow(env, &[ContractSection::Deploy, ContractSection::Metadata, ContractSection::Exec])
727    {
728        error!(
729            target: "runtime::db::db_contains_key",
730            "[WASM] [{cid}] db_contains_key(): Called in unauthorized section: {e}"
731        );
732        return darkfi_sdk::error::CALLER_ACCESS_DENIED
733    }
734
735    // Subtract used gas. Reading is free.
736    env.subtract_gas(&mut store, 1);
737
738    // Ensure memory is readable
739    let memory_view = env.memory_view(&store);
740    let Ok(mem_slice) = ptr.slice(&memory_view, ptr_len) else {
741        error!(
742            target: "runtime::db::db_contains_key",
743            "[WASM] [{cid}] db_contains_key(): Failed to make slice from ptr"
744        );
745        return darkfi_sdk::error::DB_CONTAINS_KEY_FAILED
746    };
747
748    let mut buf = vec![0_u8; ptr_len as usize];
749    if let Err(e) = mem_slice.read_slice(&mut buf) {
750        error!(
751            target: "runtime::db::db_contains_key",
752            "[WASM] [{cid}] db_contains_key(): Failed to read from memory slice: {e}"
753        );
754        return darkfi_sdk::error::DB_CONTAINS_KEY_FAILED
755    };
756
757    let mut buf_reader = Cursor::new(buf);
758
759    // Decode DbHandle index
760    let db_handle_index: u32 = match Decodable::decode(&mut buf_reader) {
761        Ok(v) => v,
762        Err(e) => {
763            error!(
764                target: "runtime::db::db_contains_key",
765                "[WASM] [{cid}] db_contains_key(): Failed to decode DbHandle: {e}"
766            );
767            return darkfi_sdk::error::DB_CONTAINS_KEY_FAILED
768        }
769    };
770
771    let db_handle_index = db_handle_index as usize;
772
773    // Decode key
774    let key: Vec<u8> = match Decodable::decode(&mut buf_reader) {
775        Ok(v) => v,
776        Err(e) => {
777            error!(
778                target: "runtime::db::db_contains_key",
779                "[WASM] [{cid}] db_contains_key(): Failed to decode key vec: {e}"
780            );
781            return darkfi_sdk::error::DB_CONTAINS_KEY_FAILED
782        }
783    };
784
785    // Make sure there are no trailing bytes in the buffer.
786    // This means we've used all data that was supplied.
787    if buf_reader.position() != ptr_len as u64 {
788        error!(
789            target: "runtime::db::db_contains_key",
790            "[WASM] [{cid}] db_contains_key(): Trailing bytes in argument stream"
791        );
792        return darkfi_sdk::error::DB_CONTAINS_KEY_FAILED
793    }
794
795    let db_handles = env.db_handles.borrow();
796
797    // Ensure DbHandle index is within bounds
798    if db_handles.len() <= db_handle_index {
799        error!(
800            target: "runtime::db::db_contains_key",
801            "[WASM] [{cid}] db_contains_key(): Requested DbHandle that is out of bounds"
802        );
803        return darkfi_sdk::error::DB_CONTAINS_KEY_FAILED
804    }
805
806    // Retrieve DbHandle using the index
807    let db_handle = &db_handles[db_handle_index];
808
809    // Lookup key parameter in the database
810    match env.blockchain.lock().unwrap().overlay.lock().unwrap().contains_key(&db_handle.tree, &key)
811    {
812        Ok(v) => i64::from(v), // <- 0=false, 1=true. Convert bool to i64.
813        Err(e) => {
814            error!(
815                target: "runtime::db::db_contains_key",
816                "[WASM] [{cid}] db_contains_key(): sled.tree.contains_key failed: {e}"
817            );
818            darkfi_sdk::error::DB_CONTAINS_KEY_FAILED
819        }
820    }
821}

823/// Given a zkas circuit, create a VerifyingKey and insert them both into the db.
824///
825/// This function can only be called from the Deploy [`ContractSection`].
826/// Returns `SUCCESS` on success, otherwise returns an error code.
827///
828/// Permissions: deploy
829pub(crate) fn zkas_db_set(mut ctx: FunctionEnvMut<Env>, ptr: WasmPtr<u8>, ptr_len: u32) -> i64 {
830    let (env, mut store) = ctx.data_and_store_mut();
831    let cid = env.contract_id;
832
833    if let Err(e) = acl_allow(env, &[ContractSection::Deploy]) {
834        error!(
835            target: "runtime::db::zkas_db_set",
836            "[WASM] [{cid}] zkas_db_set(): Called in unauthorized section: {e}"
837        );
838        return darkfi_sdk::error::CALLER_ACCESS_DENIED
839    }
840
841    let memory_view = env.memory_view(&store);
842
843    // Ensure that the memory is readable
844    let Ok(mem_slice) = ptr.slice(&memory_view, ptr_len) else {
845        error!(
846            target: "runtime::db::zkas_db_set",
847            "[WASM] [{cid}] zkas_db_set(): Failed to make slice from ptr"
848        );
849        return darkfi_sdk::error::DB_SET_FAILED
850    };
851
852    let mut buf = vec![0u8; ptr_len as usize];
853    if let Err(e) = mem_slice.read_slice(&mut buf) {
854        error!(
855            target: "runtime::db::zkas_db_set",
856            "[WASM] [{cid}] zkas_db_set(): Failed to read from memory slice: {e}"
857        );
858        return darkfi_sdk::error::DB_SET_FAILED
859    };
860
861    // Deserialize the ZkBinary bytes from the buffer
862    let zkbin_bytes: Vec<u8> = match deserialize(&buf) {
863        Ok(zkbin) => zkbin,
864        Err(e) => {
865            error!(
866                target: "runtime::db::zkas_db_set",
867                "[WASM] [{cid}] zkas_db_set(): Could not deserialize bytes from buffer: {e}"
868            );
869            return darkfi_sdk::error::DB_SET_FAILED
870        }
871    };
872
873    // Validate the bytes by decoding them into the ZkBinary format
874    let zkbin = match ZkBinary::decode(&zkbin_bytes) {
875        Ok(zkbin) => zkbin,
876        Err(e) => {
877            error!(
878                target: "runtime::db::zkas_db_set",
879                "[WASM] [{cid}] zkas_db_set(): Invalid zkas bincode passed to function: {e}"
880            );
881            return darkfi_sdk::error::DB_SET_FAILED
882        }
883    };
884
885    // Subtract used gas. We count 100 gas per opcode, witness, and literal.
886    // This is likely bad.
887    // TODO: This should be better-priced.
888    let gas_cost =
889        (zkbin.literals.len() + zkbin.witnesses.len() + zkbin.opcodes.len()) as u64 * 100;
890    env.subtract_gas(&mut store, gas_cost);
891
892    // Because of `Runtime::Deploy`, we should be sure that the zkas db is index zero.
893    let db_handles = env.db_handles.borrow();
894    let db_handle = &db_handles[0];
895    // Redundant check
896    if db_handle.contract_id != cid {
897        error!(
898            target: "runtime::db::zkas_db_set",
899            "[WASM] [{cid}] zkas_db_set(): Internal error, zkas db at index 0 incorrect"
900        );
901        return darkfi_sdk::error::DB_SET_FAILED
902    }
903
904    // Check if there is existing bincode and compare it. Return DB_SUCCESS if
905    // they're the same. The assumption should be that VerifyingKey was generated
906    // already so we can skip things after this guard.
907    match env
908        .blockchain
909        .lock()
910        .unwrap()
911        .overlay
912        .lock()
913        .unwrap()
914        .get(&db_handle.tree, &serialize(&zkbin.namespace))
915    {
916        Ok(v) => {
917            if let Some(bytes) = v {
918                // We allow a panic here because this db should never be corrupted in this way.
919                let (existing_zkbin, _): (Vec<u8>, Vec<u8>) =
920                    deserialize(&bytes).expect("deserialize tuple");
921
922                if existing_zkbin == zkbin_bytes {
923                    debug!(
924                        target: "runtime::db::zkas_db_set",
925                        "[WASM] [{cid}] zkas_db_set(): Existing zkas bincode is the same. Skipping."
926                    );
927                    return wasm::entrypoint::SUCCESS
928                }
929            }
930        }
931        Err(e) => {
932            error!(
933                target: "runtime::db::zkas_db_set",
934                "[WASM] [{cid}] zkas_db_set(): Internal error getting from tree: {e}"
935            );
936            return darkfi_sdk::error::DB_SET_FAILED
937        }
938    };
939
940    // We didn't find any existing bincode, so let's create a new VerifyingKey and write it all.
941    info!(
942        target: "runtime::db::zkas_db_set",
943        "[WASM] [{cid}] zkas_db_set(): Creating VerifyingKey for {} zkas circuit",
944        zkbin.namespace,
945    );
946
947    let witnesses = match empty_witnesses(&zkbin) {
948        Ok(w) => w,
949        Err(e) => {
950            error!(
951                target: "runtime::db::zkas_db_set",
952                "[WASM] [{cid}] zkas_db_set(): Failed to create empty witnesses: {e}"
953            );
954            return darkfi_sdk::error::DB_SET_FAILED
955        }
956    };
957
958    // Construct the circuit and build the VerifyingKey
959    let circuit = ZkCircuit::new(witnesses, &zkbin);
960    let vk = VerifyingKey::build(zkbin.k, &circuit);
961    let mut vk_buf = vec![];
962    if let Err(e) = vk.write(&mut vk_buf) {
963        error!(
964            target: "runtime::db::zkas_db_set",
965            "[WASM] [{cid}] zkas_db_set(): Failed to serialize VerifyingKey: {e}"
966        );
967        return darkfi_sdk::error::DB_SET_FAILED
968    }
969
970    // Insert the key-value pair into the database.
971    let key = serialize(&zkbin.namespace);
972    let value = serialize(&(zkbin_bytes, vk_buf));
973    if env
974        .blockchain
975        .lock()
976        .unwrap()
977        .overlay
978        .lock()
979        .unwrap()
980        .insert(&db_handle.tree, &key, &value)
981        .is_err()
982    {
983        error!(
984            target: "runtime::db::zkas_db_set",
985            "[WASM] [{cid}] zkas_db_set(): Couldn't insert to db_handle tree"
986        );
987        return darkfi_sdk::error::DB_SET_FAILED
988    }
989    drop(db_handles);
990
991    // Subtract used gas. Here we count the bytes written into the db.
992    env.subtract_gas(&mut store, (key.len() + value.len()) as u64);
993
994    wasm::entrypoint::SUCCESS
995}
