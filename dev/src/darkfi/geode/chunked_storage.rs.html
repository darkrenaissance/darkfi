19use std::path::{Path, PathBuf};
20
21use crate::geode::{file_sequence::FileSequence, MAX_CHUNK_SIZE};
22
23/// `ChunkedStorage` is a representation of a file or directory we're trying to
24/// retrieve from `Geode`.
25#[derive(Debug)]
26pub struct ChunkedStorage {
27    /// Vector of chunk hashes and a bool which is `true` if the chunk is
28    /// available locally.
29    chunks: Vec<(blake3::Hash, bool)>,
30    /// FileSequence containing the list of file paths and file sizes, it has
31    /// a single item if this is not a directory but a single file.
32    fileseq: FileSequence,
33    /// Set to `true` if this ChunkedStorage is the representation of a
34    /// directory.
35    is_dir: bool,
36}
37
38impl ChunkedStorage {
39    pub fn new(hashes: &[blake3::Hash], files: &[(PathBuf, u64)], is_dir: bool) -> Self {
40        Self {
41            chunks: hashes.iter().map(|x| (*x, false)).collect(),
42            fileseq: FileSequence::new(files, is_dir),
43            is_dir,
44        }
45    }
46
47    /// Check whether we have all the chunks available locally.
48    pub fn is_complete(&self) -> bool {
49        !self.chunks.iter().any(|(_, available)| !available)
50    }
51
52    /// Return an iterator over the chunks and their availability.
53    pub fn iter(&self) -> core::slice::Iter<'_, (blake3::Hash, bool)> {
54        self.chunks.iter()
55    }
56
57    /// Return an mutable iterator over the chunks and their availability.
58    pub fn iter_mut(&mut self) -> core::slice::IterMut<'_, (blake3::Hash, bool)> {
59        self.chunks.iter_mut()
60    }
61
62    /// Return the number of chunks.
63    pub fn len(&self) -> usize {
64        self.chunks.len()
65    }
66
67    /// Return `true` if the chunked file contains no chunk.
68    pub fn is_empty(&self) -> bool {
69        self.chunks.is_empty()
70    }
71
72    /// Return the number of chunks available locally.
73    pub fn local_chunks(&self) -> usize {
74        self.chunks.iter().filter(|(_, p)| *p).count()
75    }
76
77    /// Return `chunks`.
78    pub fn get_chunks(&self) -> &Vec<(blake3::Hash, bool)> {
79        &self.chunks
80    }
81
82    /// Return a mutable chunk from `chunks`.
83    pub fn get_chunk_mut(&mut self, index: usize) -> &mut (blake3::Hash, bool) {
84        &mut self.chunks[index]
85    }
86
87    /// Return the list of files from the `reader`.
88    pub fn get_files(&self) -> &Vec<(PathBuf, u64)> {
89        self.fileseq.get_files()
90    }
91
92    /// Return `fileseq`.
93    pub fn get_fileseq(&self) -> &FileSequence {
94        &self.fileseq
95    }
96
97    /// Return a mutable `fileseq`.
98    pub fn get_fileseq_mut(&mut self) -> &mut FileSequence {
99        &mut self.fileseq
100    }
101
102    /// Return `is_dir`.
103    pub fn is_dir(&self) -> bool {
104        self.is_dir
105    }
106
107    /// Return all chunks that contain parts of `file`.
108    pub fn get_chunks_of_file(&self, file: &Path) -> Vec<blake3::Hash> {
109        let files = self.fileseq.get_files();
110        let file_index = files.iter().position(|(f, _)| f == file);
111        if file_index.is_none() {
112            return vec![];
113        }
114        let file_index = file_index.unwrap();
115
116        let start_pos = self.fileseq.get_file_position(file_index);
117
118        let end_pos = start_pos + files[file_index].1;
119
120        let start_index = (start_pos as f64 / MAX_CHUNK_SIZE as f64).floor();
121        let end_index = (end_pos as f64 / MAX_CHUNK_SIZE as f64).floor();
122
123        let chunk_indexes: Vec<usize> = (start_index as usize..=end_index as usize).collect();
124
125        chunk_indexes
126            .iter()
127            .filter_map(|&index| self.chunks.get(index))
128            .map(|(hash, _)| hash)
129            .cloned()
130            .collect()
131    }
132}
