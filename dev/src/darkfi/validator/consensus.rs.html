87    /// Generate a new empty fork.
88    pub async fn generate_empty_fork(&self) -> Result<()> {
89        debug!(target: "validator::consensus::generate_empty_fork", "Generating new empty fork...");
90        let mut forks = self.forks.write().await;
91        // Check if we already have an empty fork
92        for fork in forks.iter() {
93            if fork.proposals.is_empty() {
94                debug!(target: "validator::consensus::generate_empty_fork", "An empty fork already exists.");
95                drop(forks);
96                return Ok(())
97            }
98        }
99        let fork = Fork::new(self.blockchain.clone(), self.module.read().await.clone()).await?;
100        forks.push(fork);
101        drop(forks);
102        debug!(target: "validator::consensus::generate_empty_fork", "Fork generated!");
103        Ok(())
104    }

106    /// Given a proposal, the node verifys it and finds which fork it extends.
107    /// If the proposal extends the canonical blockchain, a new fork chain is created.
108    pub async fn append_proposal(&self, proposal: &Proposal, verify_fees: bool) -> Result<()> {
109        debug!(target: "validator::consensus::append_proposal", "Appending proposal {}", proposal.hash);
110
111        // Check if proposal already exists
112        let lock = self.forks.read().await;
113        for fork in lock.iter() {
114            for p in fork.proposals.iter().rev() {
115                if p == &proposal.hash {
116                    drop(lock);
117                    debug!(target: "validator::consensus::append_proposal", "Proposal {} already exists", proposal.hash);
118                    return Err(Error::ProposalAlreadyExists)
119                }
120            }
121        }
122        // Check if proposal is canonical
123        if let Ok(canonical_headers) =
124            self.blockchain.blocks.get_order(&[proposal.block.header.height], true)
125        {
126            if canonical_headers[0].unwrap() == proposal.hash {
127                drop(lock);
128                debug!(target: "validator::consensus::append_proposal", "Proposal {} already exists", proposal.hash);
129                return Err(Error::ProposalAlreadyExists)
130            }
131        }
132        drop(lock);
133
134        // Verify proposal and grab corresponding fork
135        let (mut fork, index) = verify_proposal(self, proposal, verify_fees).await?;
136
137        // Append proposal to the fork
138        fork.append_proposal(proposal).await?;
139
140        // TODO: to keep memory usage low, we should only append forks that
141        // are higher ranking than our current best one
142
143        // If a fork index was found, replace forks with the mutated one,
144        // otherwise push the new fork.
145        let mut lock = self.forks.write().await;
146        match index {
147            Some(i) => {
148                if i < lock.len() && lock[i].proposals == fork.proposals[..fork.proposals.len() - 1]
149                {
150                    lock[i] = fork;
151                } else {
152                    lock.push(fork);
153                }
154            }
155            None => {
156                lock.push(fork);
157            }
158        }
159        drop(lock);
160
161        info!(target: "validator::consensus::append_proposal", "Appended proposal {}", proposal.hash);
162
163        Ok(())
164    }

166    /// Given a proposal, find the fork chain it extends, and return its full clone.
167    /// If the proposal extends the fork not on its tail, a new fork is created and
168    /// we re-apply the proposals up to the extending one. If proposal extends canonical,
169    /// a new fork is created. Additionally, we return the fork index if a new fork
170    /// was not created, so caller can replace the fork.
171    pub async fn find_extended_fork(&self, proposal: &Proposal) -> Result<(Fork, Option<usize>)> {
172        // Grab a lock over current forks
173        let forks = self.forks.read().await;
174
175        // Check if proposal extends any fork
176        let found = find_extended_fork_index(&forks, proposal);
177        if found.is_err() {
178            if let Err(Error::ProposalAlreadyExists) = found {
179                return Err(Error::ProposalAlreadyExists)
180            }
181
182            // Check if proposal extends canonical
183            let (last_height, last_block) = self.blockchain.last()?;
184            if proposal.block.header.previous != last_block ||
185                proposal.block.header.height <= last_height
186            {
187                return Err(Error::ExtendedChainIndexNotFound)
188            }
189
190            // Check if we have an empty fork to use
191            for (f_index, fork) in forks.iter().enumerate() {
192                if fork.proposals.is_empty() {
193                    return Ok((forks[f_index].full_clone()?, Some(f_index)))
194                }
195            }
196
197            // Generate a new fork extending canonical
198            let fork = Fork::new(self.blockchain.clone(), self.module.read().await.clone()).await?;
199            return Ok((fork, None))
200        }
201
202        let (f_index, p_index) = found.unwrap();
203        let original_fork = &forks[f_index];
204        // Check if proposal extends fork at last proposal
205        if p_index == (original_fork.proposals.len() - 1) {
206            return Ok((original_fork.full_clone()?, Some(f_index)))
207        }
208
209        // Rebuild fork
210        let mut fork = Fork::new(self.blockchain.clone(), self.module.read().await.clone()).await?;
211        fork.proposals = original_fork.proposals[..p_index + 1].to_vec();
212        fork.diffs = original_fork.diffs[..p_index + 1].to_vec();
213
214        // Retrieve proposals blocks from original fork
215        let blocks = &original_fork.overlay.lock().unwrap().get_blocks_by_hash(&fork.proposals)?;
216        for (index, block) in blocks.iter().enumerate() {
217            // Apply block diffs
218            fork.overlay.lock().unwrap().overlay.lock().unwrap().add_diff(&fork.diffs[index])?;
219
220            // Grab next mine target and difficulty
221            let (next_target, next_difficulty) = fork.module.next_mine_target_and_difficulty()?;
222
223            // Calculate block rank
224            let (target_distance_sq, hash_distance_sq) = block_rank(block, &next_target)?;
225
226            // Update PoW module
227            fork.module.append(&block.header, &next_difficulty)?;
228
229            // Update fork ranks
230            fork.targets_rank += target_distance_sq;
231            fork.hashes_rank += hash_distance_sq;
232        }
233
234        // Rebuild fork contracts states monotree
235        fork.compute_monotree()?;
236
237        // Drop forks lock
238        drop(forks);
239
240        Ok((fork, None))
241    }

243    /// Check if best fork proposals can be confirmed.
244    /// Consensus confirmation logic:
245    /// - If the current best fork has reached greater length than the security threshold,
246    ///   and no other fork exist with same rank, first proposal(s) in that fork can be
247    ///   appended to canonical blockchain (confirme).
248    ///
249    /// When best fork can be confirmed, first block(s) should be appended to canonical,
250    /// and forks should be rebuilt.
251    pub async fn confirmation(&self) -> Result<Option<usize>> {
252        debug!(target: "validator::consensus::confirmation", "Started confirmation check");
253
254        // Grab best fork
255        let forks = self.forks.read().await;
256        let index = best_fork_index(&forks)?;
257        let fork = &forks[index];
258
259        // Check its length
260        let length = fork.proposals.len();
261        if length < self.confirmation_threshold {
262            debug!(target: "validator::consensus::confirmation", "Nothing to confirme yet, best fork size: {length}");
263            drop(forks);
264            return Ok(None)
265        }
266
267        // Drop forks lock
268        drop(forks);
269
270        Ok(Some(index))
271    }

273    /// Auxiliary function to retrieve the fork header hash of provided height.
274    /// The fork is identified by the provided header hash.
275    pub async fn get_fork_header_hash(
276        &self,
277        height: u32,
278        fork_header: &HeaderHash,
279    ) -> Result<Option<HeaderHash>> {
280        // Grab a lock over current forks
281        let forks = self.forks.read().await;
282
283        // Find the fork containing the provided header
284        let mut found = None;
285        'outer: for (index, fork) in forks.iter().enumerate() {
286            for p in fork.proposals.iter().rev() {
287                if p == fork_header {
288                    found = Some(index);
289                    break 'outer
290                }
291            }
292        }
293        if found.is_none() {
294            drop(forks);
295            return Ok(None)
296        }
297        let index = found.unwrap();
298
299        // Grab header if it exists
300        let header = forks[index].overlay.lock().unwrap().blocks.get_order(&[height], false)?[0];
301
302        // Drop forks lock
303        drop(forks);
304
305        Ok(header)
306    }

382    /// Auxiliary function to retrieve a fork proposals, starting from provided tip.
383    /// If provided tip is too far behind, unknown, or fork doesn't exists, an empty
384    /// vector is returned. The fork is identified by the optional provided header hash.
385    /// If its `None`, we use our best fork.
386    pub async fn get_fork_proposals_after(
387        &self,
388        tip: HeaderHash,
389        fork_tip: Option<HeaderHash>,
390        limit: u32,
391    ) -> Result<Vec<Proposal>> {
392        // Grab a lock over current forks
393        let forks = self.forks.read().await;
394
395        // Create return vector
396        let mut proposals = vec![];
397
398        // Grab fork index to use
399        let index = match fork_tip {
400            Some(fork_tip) => {
401                let mut found = None;
402                'outer: for (index, fork) in forks.iter().enumerate() {
403                    for p in fork.proposals.iter().rev() {
404                        if p == &fork_tip {
405                            found = Some(index);
406                            break 'outer
407                        }
408                    }
409                }
410                if found.is_none() {
411                    drop(forks);
412                    return Ok(proposals)
413                }
414                found.unwrap()
415            }
416            None => best_fork_index(&forks)?,
417        };
418
419        // Check tip exists
420        let Ok(existing_tips) = forks[index].overlay.lock().unwrap().get_blocks_by_hash(&[tip])
421        else {
422            drop(forks);
423            return Ok(proposals)
424        };
425
426        // Check tip is not far behind
427        let last_block_height = forks[index].overlay.lock().unwrap().last()?.0;
428        if last_block_height - existing_tips[0].header.height >= limit {
429            drop(forks);
430            return Ok(proposals)
431        }
432
433        // Retrieve all proposals after requested one
434        let headers = self.blockchain.blocks.get_all_after(existing_tips[0].header.height)?;
435        let blocks = self.blockchain.get_blocks_by_hash(&headers)?;
436        for block in blocks {
437            proposals.push(Proposal::new(block));
438        }
439        let blocks =
440            forks[index].overlay.lock().unwrap().get_blocks_by_hash(&forks[index].proposals)?;
441        for block in blocks {
442            proposals.push(Proposal::new(block));
443        }
444
445        // Drop forks lock
446        drop(forks);
447
448        Ok(proposals)
449    }

451    /// Auxiliary function to retrieve current best fork last header.
452    /// If no forks exist, grab the last header from canonical.
453    pub async fn best_fork_last_header(&self) -> Result<(u32, HeaderHash)> {
454        // Grab a lock over current forks
455        let forks = self.forks.read().await;
456
457        // Check if node has any forks
458        if forks.is_empty() {
459            drop(forks);
460            return self.blockchain.last()
461        }
462
463        // Grab best fork
464        let fork = &forks[best_fork_index(&forks)?];
465
466        // Grab its last header
467        let last = fork.last_proposal()?;
468        drop(forks);
469        Ok((last.block.header.height, last.hash))
470    }

472    /// Auxiliary function to purge current forks and reset the ones starting
473    /// with the provided prefix, excluding provided confirmed fork.
474    /// Additionally, remove confirmed transactions from the forks mempools,
475    /// along with the unporposed transactions sled trees.
476    /// This function assumes that the prefix blocks have already been appended
477    /// to canonical chain from the confirmed fork.
478    pub async fn reset_forks(
479        &self,
480        prefix: &[HeaderHash],
481        confirmed_fork_index: &usize,
482        confirmed_txs: &[Transaction],
483    ) -> Result<()> {
484        // Grab a lock over current forks
485        let mut forks = self.forks.write().await;
486
487        // Find all the forks that start with the provided prefix,
488        // excluding confirmed fork index, and remove their prefixed
489        // proposals, and their corresponding diffs.
490        // If the fork is not starting with the provided prefix,
491        // drop it. Additionally, keep track of all the referenced
492        // trees in overlays that are valid.
493        let excess = prefix.len();
494        let prefix_last_index = excess - 1;
495        let prefix_last = prefix.last().unwrap();
496        let mut keep = vec![true; forks.len()];
497        let mut referenced_trees = HashSet::new();
498        let mut referenced_txs = HashSet::new();
499        let confirmed_txs_hashes: Vec<TransactionHash> =
500            confirmed_txs.iter().map(|tx| tx.hash()).collect();
501        for (index, fork) in forks.iter_mut().enumerate() {
502            if &index == confirmed_fork_index {
503                // Store its tree references
504                let fork_overlay = fork.overlay.lock().unwrap();
505                let overlay = fork_overlay.overlay.lock().unwrap();
506                for tree in &overlay.state.initial_tree_names {
507                    referenced_trees.insert(tree.clone());
508                }
509                for tree in &overlay.state.new_tree_names {
510                    referenced_trees.insert(tree.clone());
511                }
512                for tree in overlay.state.dropped_trees.keys() {
513                    referenced_trees.insert(tree.clone());
514                }
515                // Remove confirmed proposals txs from fork's mempool
516                fork.mempool.retain(|tx| !confirmed_txs_hashes.contains(tx));
517                // Store its txs references
518                for tx in &fork.mempool {
519                    referenced_txs.insert(*tx);
520                }
521                drop(overlay);
522                drop(fork_overlay);
523                continue
524            }
525
526            if fork.proposals.is_empty() ||
527                prefix_last_index >= fork.proposals.len() ||
528                &fork.proposals[prefix_last_index] != prefix_last
529            {
530                keep[index] = false;
531                continue
532            }
533
534            // Remove confirmed proposals txs from fork's mempool
535            fork.mempool.retain(|tx| !confirmed_txs_hashes.contains(tx));
536            // Store its txs references
537            for tx in &fork.mempool {
538                referenced_txs.insert(*tx);
539            }
540
541            // Remove the commited differences
542            let rest_proposals = fork.proposals.split_off(excess);
543            let rest_diffs = fork.diffs.split_off(excess);
544            let mut diffs = fork.diffs.clone();
545            fork.proposals = rest_proposals;
546            fork.diffs = rest_diffs;
547            for diff in diffs.iter_mut() {
548                fork.overlay.lock().unwrap().overlay.lock().unwrap().remove_diff(diff);
549            }
550
551            // Store its tree references
552            let fork_overlay = fork.overlay.lock().unwrap();
553            let overlay = fork_overlay.overlay.lock().unwrap();
554            for tree in &overlay.state.initial_tree_names {
555                referenced_trees.insert(tree.clone());
556            }
557            for tree in &overlay.state.new_tree_names {
558                referenced_trees.insert(tree.clone());
559            }
560            for tree in overlay.state.dropped_trees.keys() {
561                referenced_trees.insert(tree.clone());
562            }
563            drop(overlay);
564            drop(fork_overlay);
565        }
566
567        // Find the trees and pending txs that are no longer referenced by valid forks
568        let mut dropped_trees = HashSet::new();
569        let mut dropped_txs = HashSet::new();
570        for (index, fork) in forks.iter_mut().enumerate() {
571            if keep[index] {
572                continue
573            }
574            for tx in &fork.mempool {
575                if !referenced_txs.contains(tx) {
576                    dropped_txs.insert(*tx);
577                }
578            }
579            let fork_overlay = fork.overlay.lock().unwrap();
580            let overlay = fork_overlay.overlay.lock().unwrap();
581            for tree in &overlay.state.initial_tree_names {
582                if !referenced_trees.contains(tree) {
583                    dropped_trees.insert(tree.clone());
584                }
585            }
586            for tree in &overlay.state.new_tree_names {
587                if !referenced_trees.contains(tree) {
588                    dropped_trees.insert(tree.clone());
589                }
590            }
591            for tree in overlay.state.dropped_trees.keys() {
592                if !referenced_trees.contains(tree) {
593                    dropped_trees.insert(tree.clone());
594                }
595            }
596            drop(overlay);
597            drop(fork_overlay);
598        }
599
600        // Drop unreferenced trees from the database
601        for tree in dropped_trees {
602            self.blockchain.sled_db.drop_tree(tree)?;
603        }
604
605        // Drop invalid forks
606        let mut iter = keep.iter();
607        forks.retain(|_| *iter.next().unwrap());
608
609        // Remove confirmed proposals txs from the unporposed txs sled tree
610        self.blockchain.remove_pending_txs_hashes(&confirmed_txs_hashes)?;
611
612        // Remove unreferenced txs from the unporposed txs sled tree
613        self.blockchain.remove_pending_txs_hashes(&Vec::from_iter(dropped_txs))?;
614
615        // Drop forks lock
616        drop(forks);
617
618        Ok(())
619    }

647    /// Auxiliary function to check current contracts states
648    /// Monotree(SMT) validity in all active forks and canonical.
649    pub async fn healthcheck(&self) -> Result<()> {
650        // Grab a lock over current forks
651        let lock = self.forks.read().await;
652
653        // Rebuild current canonical contract states monotree
654        let state_monotree = self.blockchain.get_state_monotree()?;
655
656        // Check that the root matches last block header state root
657        let Some(state_root) = state_monotree.get_headroot()? else {
658            return Err(Error::ContractsStatesRootNotFoundError);
659        };
660        let last_block_state_root = self.blockchain.last_header()?.state_root;
661        if state_root != last_block_state_root {
662            return Err(Error::ContractsStatesRootError(
663                blake3::Hash::from_bytes(state_root).to_string(),
664                blake3::Hash::from_bytes(last_block_state_root).to_string(),
665            ));
666        }
667
668        // Check each fork health
669        for fork in lock.iter() {
670            fork.healthcheck()?;
671        }
672
673        Ok(())
674    }
675}

699/// Struct representing a forked blockchain state.
700///
701/// An overlay over the original blockchain is used, containing all pending to-write
702/// records. Additionally, each fork keeps a vector of valid pending transactions hashes,
703/// in order of receival, and the proposals hashes sequence, for validations.
704#[derive(Clone)]
705pub struct Fork {
706    /// Canonical (confirmed) blockchain
707    pub blockchain: Blockchain,
708    /// Overlay cache over canonical Blockchain
709    pub overlay: BlockchainOverlayPtr,
710    /// Current PoW module state
711    pub module: PoWModule,
712    /// Current contracts states Monotree(SMT)
713    pub state_monotree: Monotree<monotree::MemoryDb>,
714    /// Fork proposal hashes sequence
715    pub proposals: Vec<HeaderHash>,
716    /// Fork proposal overlay diffs sequence
717    pub diffs: Vec<SledDbOverlayStateDiff>,
718    /// Valid pending transaction hashes
719    pub mempool: Vec<TransactionHash>,
720    /// Current fork mining targets rank, cached for better performance
721    pub targets_rank: BigUint,
722    /// Current fork hashes rank, cached for better performance
723    pub hashes_rank: BigUint,
724}
725
726impl Fork {
727    pub async fn new(blockchain: Blockchain, module: PoWModule) -> Result<Self> {
728        let mempool = blockchain.get_pending_txs()?.iter().map(|tx| tx.hash()).collect();
729        let overlay = BlockchainOverlay::new(&blockchain)?;
730        // Build current contract states monotree
731        let state_monotree = overlay.lock().unwrap().get_state_monotree()?;
732        // Retrieve last block difficulty to access current ranks
733        let last_difficulty = blockchain.last_block_difficulty()?;
734        let targets_rank = last_difficulty.ranks.targets_rank;
735        let hashes_rank = last_difficulty.ranks.hashes_rank;
736        Ok(Self {
737            blockchain,
738            overlay,
739            module,
740            state_monotree,
741            proposals: vec![],
742            diffs: vec![],
743            mempool,
744            targets_rank,
745            hashes_rank,
746        })
747    }
748
749    /// Auxiliary function to append a proposal and update current fork rank.
750    pub async fn append_proposal(&mut self, proposal: &Proposal) -> Result<()> {
751        // Grab next mine target and difficulty
752        let (next_target, next_difficulty) = self.module.next_mine_target_and_difficulty()?;
753
754        // Calculate block rank
755        let (target_distance_sq, hash_distance_sq) = block_rank(&proposal.block, &next_target)?;
756
757        // Update fork ranks
758        self.targets_rank += target_distance_sq.clone();
759        self.hashes_rank += hash_distance_sq.clone();
760
761        // Generate block difficulty and update PoW module
762        let cumulative_difficulty =
763            self.module.cumulative_difficulty.clone() + next_difficulty.clone();
764        let ranks = BlockRanks::new(
765            target_distance_sq,
766            self.targets_rank.clone(),
767            hash_distance_sq,
768            self.hashes_rank.clone(),
769        );
770        let block_difficulty = BlockDifficulty::new(
771            proposal.block.header.height,
772            proposal.block.header.timestamp,
773            next_difficulty,
774            cumulative_difficulty,
775            ranks,
776        );
777        self.module.append_difficulty(&self.overlay, &proposal.block.header, block_difficulty)?;
778
779        // Push proposal's hash
780        self.proposals.push(proposal.hash);
781
782        // Push proposal overlay diff
783        self.diffs.push(self.overlay.lock().unwrap().overlay.lock().unwrap().diff(&self.diffs)?);
784
785        Ok(())
786    }
787
788    /// Auxiliary function to retrieve last proposal.
789    pub fn last_proposal(&self) -> Result<Proposal> {
790        let block = if let Some(last) = self.proposals.last() {
791            self.overlay.lock().unwrap().get_blocks_by_hash(&[*last])?[0].clone()
792        } else {
793            self.overlay.lock().unwrap().last_block()?
794        };
795
796        Ok(Proposal::new(block))
797    }
798
799    /// Auxiliary function to compute forks' next block height.
800    pub fn get_next_block_height(&self) -> Result<u32> {
801        let proposal = self.last_proposal()?;
802        Ok(proposal.block.header.height + 1)
803    }
804
805    /// Auxiliary function to retrieve unproposed valid transactions,
806    /// along with their total gas used, total paid fees and the overlay
807    /// used to verify the transactions for further processing.
808    ///
809    /// Note: Always remember to purge new trees from the overlay if not needed.
810    pub async fn unproposed_txs(
811        &self,
812        blockchain: &Blockchain,
813        verifying_block_height: u32,
814        block_target: u32,
815        verify_fees: bool,
816    ) -> Result<(Vec<Transaction>, u64, u64, BlockchainOverlayPtr)> {
817        // Clone forks' overlay
818        let overlay = self.overlay.lock().unwrap().full_clone()?;
819
820        // Check if our mempool is not empty
821        if self.mempool.is_empty() {
822            return Ok((vec![], 0, 0, overlay))
823        }
824
825        // Transactions Merkle tree
826        let mut tree = MerkleTree::new(1);
827
828        // Total gas accumulators
829        let mut total_gas_used = 0;
830        let mut total_gas_paid = 0;
831
832        // Map of ZK proof verifying keys for the current transaction batch
833        let mut vks: HashMap<[u8; 32], HashMap<String, VerifyingKey>> = HashMap::new();
834
835        // Grab all current proposals transactions hashes
836        let proposals_txs = overlay.lock().unwrap().get_blocks_txs_hashes(&self.proposals)?;
837
838        // Iterate through all pending transactions in the forks' mempool
839        let mut unproposed_txs = vec![];
840        for tx in &self.mempool {
841            // If the hash is contained in the proposals transactions vec, skip it
842            if proposals_txs.contains(tx) {
843                continue
844            }
845
846            // Retrieve the actual unproposed transaction
847            let unproposed_tx =
848                blockchain.transactions.get_pending(&[*tx], true)?[0].clone().unwrap();
849
850            // Update the verifying keys map
851            for call in &unproposed_tx.calls {
852                vks.entry(call.data.contract_id.to_bytes()).or_default();
853            }
854
855            // Verify the transaction against current state
856            overlay.lock().unwrap().checkpoint();
857            let gas_data = match verify_transaction(
858                &overlay,
859                verifying_block_height,
860                block_target,
861                &unproposed_tx,
862                &mut tree,
863                &mut vks,
864                verify_fees,
865            )
866            .await
867            {
868                Ok(gas_values) => gas_values,
869                Err(e) => {
870                    debug!(target: "validator::consensus::unproposed_txs", "Transaction verification failed: {e}");
871                    overlay.lock().unwrap().revert_to_checkpoint()?;
872                    continue
873                }
874            };
875
876            // Store the gas used by the verified transaction
877            let tx_gas_used = gas_data.total_gas_used();
878
879            // Calculate current accumulated gas usage
880            let accumulated_gas_usage = total_gas_used + tx_gas_used;
881
882            // Check gas limit - if accumulated gas used exceeds it, break out of loop
883            if accumulated_gas_usage > BLOCK_GAS_LIMIT {
884                warn!(
885                    target: "validator::consensus::unproposed_txs",
886                    "Retrieving transaction {tx} would exceed configured unproposed transaction gas limit: {accumulated_gas_usage} - {BLOCK_GAS_LIMIT}"
887                );
888                overlay.lock().unwrap().revert_to_checkpoint()?;
889                break
890            }
891
892            // Update accumulated total gas
893            total_gas_used += tx_gas_used;
894            total_gas_paid += gas_data.paid;
895
896            // Push the tx hash into the unproposed transactions vector
897            unproposed_txs.push(unproposed_tx);
898        }
899
900        Ok((unproposed_txs, total_gas_used, total_gas_paid, overlay))
901    }

903    /// Auxiliary function to create a full clone using BlockchainOverlay::full_clone.
904    /// Changes to this copy don't affect original fork overlay records, since underlying
905    /// overlay pointer have been updated to the cloned one.
906    pub fn full_clone(&self) -> Result<Self> {
907        let blockchain = self.blockchain.clone();
908        let overlay = self.overlay.lock().unwrap().full_clone()?;
909        let module = self.module.clone();
910        let state_monotree = self.state_monotree.clone();
911        let proposals = self.proposals.clone();
912        let diffs = self.diffs.clone();
913        let mempool = self.mempool.clone();
914        let targets_rank = self.targets_rank.clone();
915        let hashes_rank = self.hashes_rank.clone();
916
917        Ok(Self {
918            blockchain,
919            overlay,
920            module,
921            state_monotree,
922            proposals,
923            diffs,
924            mempool,
925            targets_rank,
926            hashes_rank,
927        })
928    }

930    /// Build current contract states monotree.
931    pub fn compute_monotree(&mut self) -> Result<()> {
932        self.state_monotree = self.overlay.lock().unwrap().get_state_monotree()?;
933        Ok(())
934    }

936    /// Auxiliary function to check current contracts states
937    /// Monotree(SMT) validity.
938    ///
939    /// Note: This should be executed on fresh forks and/or when
940    ///       a fork doesn't contain changes over the last appended
941    //        proposal.
942    pub fn healthcheck(&self) -> Result<()> {
943        // Rebuild current contract states monotree
944        let state_monotree = self.overlay.lock().unwrap().get_state_monotree()?;
945
946        // Check that it matches forks' tree
947        let Some(state_root) = state_monotree.get_headroot()? else {
948            return Err(Error::ContractsStatesRootNotFoundError);
949        };
950        let Some(fork_state_root) = self.state_monotree.get_headroot()? else {
951            return Err(Error::ContractsStatesRootNotFoundError);
952        };
953        if state_root != fork_state_root {
954            return Err(Error::ContractsStatesRootError(
955                blake3::Hash::from_bytes(state_root).to_string(),
956                blake3::Hash::from_bytes(fork_state_root).to_string(),
957            ));
958        }
959
960        // Check that the root matches last block header state root
961        let last_block_state_root = self.last_proposal()?.block.header.state_root;
962        if state_root != last_block_state_root {
963            return Err(Error::ContractsStatesRootError(
964                blake3::Hash::from_bytes(state_root).to_string(),
965                blake3::Hash::from_bytes(last_block_state_root).to_string(),
966            ));
967        }
968
969        Ok(())
970    }
