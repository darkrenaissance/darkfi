44/// Malicious behaviour threshold. If the threshold is reached, we will
45/// drop the peer from our P2P connection.
46const MALICIOUS_THRESHOLD: usize = 5;
47
48/// Global limit of messages per window
49const WINDOW_MAXSIZE: usize = 200;
50/// Rolling length of the window
51const WINDOW_EXPIRY_TIME: NanoTimestamp = NanoTimestamp::from_secs(60);
52
53/// Rolling length of the window
54const RATELIMIT_EXPIRY_TIME: NanoTimestamp = NanoTimestamp::from_secs(10);
55/// Ratelimit kicks in above this count
56const RATELIMIT_MIN_COUNT: usize = 6;
57/// Sample point used to calculate sleep time when ratelimit is active
58const RATELIMIT_SAMPLE_IDX: usize = 10;
59/// Sleep for this amount of time when `count == RATE_LIMIT_SAMPLE_IDX`.
60const RATELIMIT_SAMPLE_SLEEP: usize = 1000;
61
62struct MovingWindow {
63    times: VecDeque<NanoTimestamp>,
64    expiry_time: NanoTimestamp,
65}
66
67impl MovingWindow {
68    fn new(expiry_time: NanoTimestamp) -> Self {
69        Self { times: VecDeque::new(), expiry_time }
70    }
71
72    /// Clean out expired timestamps from the window.
73    fn clean(&mut self) {
74        while let Some(ts) = self.times.front() {
75            let Ok(elapsed) = ts.elapsed() else {
76                debug!(target: "event_graph::protocol::MovingWindow::clean()", "Timestamp [{ts}] is in future. Removing...");
77                let _ = self.times.pop_front();
78                continue
79            };
80            if elapsed < self.expiry_time {
81                break
82            }
83            let _ = self.times.pop_front();
84        }
85    }
86
87    /// Add new timestamp
88    fn ticktock(&mut self) {
89        self.clean();
90        self.times.push_back(NanoTimestamp::current_time());
91    }
92
93    #[inline]
94    fn count(&self) -> usize {
95        self.times.len()
96    }
97}

152impl ProtocolBase for ProtocolEventGraph {
153    async fn start(self: Arc<Self>, ex: Arc<Executor<'_>>) -> Result<()> {
154        self.jobsman.clone().start(ex.clone());
155        self.jobsman.clone().spawn(self.clone().handle_event_put(), ex.clone()).await;
156        self.jobsman.clone().spawn(self.clone().handle_event_req(), ex.clone()).await;
157        self.jobsman.clone().spawn(self.clone().handle_tip_req(), ex.clone()).await;
158        self.jobsman.clone().spawn(self.clone().broadcast_rate_limiter(), ex.clone()).await;
159        Ok(())
160    }

163        "ProtocolEventGraph"
164    }

168    pub async fn init(event_graph: EventGraphPtr, channel: ChannelPtr) -> Result<ProtocolBasePtr> {
169        let msg_subsystem = channel.message_subsystem();
170        msg_subsystem.add_dispatch::<EventPut>().await;
171        msg_subsystem.add_dispatch::<EventReq>().await;
172        msg_subsystem.add_dispatch::<EventRep>().await;
173        msg_subsystem.add_dispatch::<TipReq>().await;
174        msg_subsystem.add_dispatch::<TipRep>().await;
175
176        let ev_put_sub = channel.subscribe_msg::<EventPut>().await?;
177        let ev_req_sub = channel.subscribe_msg::<EventReq>().await?;
178        let ev_rep_sub = channel.subscribe_msg::<EventRep>().await?;
179        let tip_req_sub = channel.subscribe_msg::<TipReq>().await?;
180        let _tip_rep_sub = channel.subscribe_msg::<TipRep>().await?;
181
182        let (broadcaster_push, broadcaster_pull) = smol::channel::unbounded();
183
184        Ok(Arc::new(Self {
185            channel: channel.clone(),
186            event_graph,
187            ev_put_sub,
188            ev_req_sub,
189            ev_rep_sub,
190            tip_req_sub,
191            _tip_rep_sub,
192            malicious_count: AtomicUsize::new(0),
193            jobsman: ProtocolJobsManager::new("ProtocolEventGraph", channel.clone()),
194            broadcaster_push,
195            broadcaster_pull,
196        }))
197    }

199    async fn increase_malicious_count(self: Arc<Self>) -> Result<()> {
200        let malicious_count = self.malicious_count.fetch_add(1, SeqCst);
201        if malicious_count + 1 == MALICIOUS_THRESHOLD {
202            error!(
203                target: "event_graph::protocol::handle_event_put()",
204                "[EVENTGRAPH] Peer {} reached malicious threshold. Dropping connection.",
205                self.channel.display_address(),
206            );
207            self.channel.stop().await;
208            return Err(Error::ChannelStopped)
209        }
210
211        warn!(
212            target: "event_graph::protocol::handle_event_put()",
213            "[EVENTGRAPH] Peer {} sent us a malicious event", self.channel.display_address(),
214        );
215
216        Ok(())
217    }

219    /// Protocol function handling `EventPut`.
220    /// This is triggered whenever someone broadcasts (or relays) a new
221    /// event on the network.
222    async fn handle_event_put(self: Arc<Self>) -> Result<()> {
223        // Rolling window of event timestamps on this channel
224        let mut bantimes = MovingWindow::new(WINDOW_EXPIRY_TIME);
225
226        loop {
227            let event = match self.ev_put_sub.receive().await {
228                Ok(v) => v.0.clone(),
229                Err(_) => continue,
230            };
231            trace!(
232                 target: "event_graph::protocol::handle_event_put()",
233                 "Got EventPut: {} [{}]", event.id(), self.channel.display_address(),
234            );
235
236            // Check if node has finished syncing its DAG
237            if !*self.event_graph.synced.read().await {
238                debug!(
239                    target: "event_graph::protocol::handle_event_put",
240                    "DAG is still syncing, skipping..."
241                );
242                continue
243            }
244
245            // If we have already seen the event, we'll stay quiet.
246            let event_id = event.id();
247            if self.event_graph.dag.contains_key(event_id.as_bytes()).unwrap() {
248                debug!(
249                    target: "event_graph::protocol::handle_event_put()",
250                    "Event {event_id} is already known"
251                );
252                continue
253            }
254
255            // There's a new unique event.
256            // Apply ban logic to stop network floods.
257            bantimes.ticktock();
258            if bantimes.count() > WINDOW_MAXSIZE {
259                self.channel.ban().await;
260                // This error is actually unused. We could return Ok here too.
261                return Err(Error::MaliciousFlood)
262            }
263
264            // We received an event. Check if we already have it in our DAG.
265            // Check event is not older that current genesis event timestamp.
266            // Also check if we have the event's parents. In the case we do
267            // not have the parents, we'll request them from the peer that has
268            // sent this event to us. In case they do not reply in time, we drop
269            // the event.
270
271            // Check if the event is older than the genesis event. If so, we should
272            // not include it in our Dag.
273            // The genesis event marks the last time the Dag has been pruned of old
274            // events. The pruning interval is defined by the days_rotation field
275            // of [`EventGraph`].
276            let genesis_timestamp = self.event_graph.current_genesis.read().await.timestamp;
277            if event.timestamp < genesis_timestamp {
278                debug!(
279                    target: "event_graph::protocol::handle_event_put()",
280                    "Event {} is older than genesis. Event timestamp: `{}`. Genesis timestamp: `{genesis_timestamp}`",
281                event.id(), event.timestamp
282                );
283            }
284
285            // Validate the new event first. If we do not consider it valid, we
286            // will just drop it and stay quiet. If the malicious threshold
287            // is reached, we will stop the connection.
288            if !event.validate_new() {
289                self.clone().increase_malicious_count().await?;
290                continue
291            }
292
293            // At this point, this is a new event to us. Let's see if we
294            // have all of its parents.
295            debug!(
296                target: "event_graph::protocol::handle_event_put()",
297                "Event {event_id} is new"
298            );
299
300            let mut missing_parents = HashSet::new();
301            for parent_id in event.parents.iter() {
302                // `event.validate_new()` should have already made sure that
303                // not all parents are NULL, and that there are no duplicates.
304                if parent_id == &NULL_ID {
305                    continue
306                }
307
308                if !self.event_graph.dag.contains_key(parent_id.as_bytes()).unwrap() {
309                    missing_parents.insert(*parent_id);
310                }
311            }
312
313            // If we have missing parents, then we have to attempt to
314            // fetch them from this peer. Do this recursively until we
315            // find all of them.
316            if !missing_parents.is_empty() {
317                // We track the received events mapped by their layer.
318                // If/when we get all of them, we need to insert them in order so
319                // the DAG state stays correct and unreferenced tips represent the
320                // actual thing they should. If we insert them out of order, then
321                // we might have wrong unreferenced tips.
322                let mut received_events: BTreeMap<u64, Vec<Event>> = BTreeMap::new();
323                let mut received_events_hashes = HashSet::new();
324
325                debug!(
326                    target: "event_graph::protocol::handle_event_put()",
327                    "Event has {} missing parents. Requesting...", missing_parents.len(),
328                );
329
330                while !missing_parents.is_empty() {
331                    // for parent_id in missing_parents.clone().iter() {
332                    debug!(
333                        target: "event_graph::protocol::handle_event_put()",
334                        "Requesting {missing_parents:?}..."
335                    );
336
337                    self.channel
338                        .send(&EventReq(missing_parents.clone().into_iter().collect()))
339                        .await?;
340
341                    // Node waits for response
342                    let Ok(parents) = self
343                        .ev_rep_sub
344                        .receive_with_timeout(
345                            self.event_graph.p2p.settings().read().await.outbound_connect_timeout,
346                        )
347                        .await
348                    else {
349                        error!(
350                            target: "event_graph::protocol::handle_event_put()",
351                            "[EVENTGRAPH] Timeout while waiting for parents {missing_parents:?} from {}",
352                            self.channel.display_address(),
353                        );
354                        self.channel.stop().await;
355                        return Err(Error::ChannelStopped)
356                    };
357
358                    let parents = parents.0.clone();
359
360                    for parent in parents {
361                        let parent_id = parent.id();
362                        if !missing_parents.contains(&parent_id) {
363                            error!(
364                                target: "event_graph::protocol::handle_event_put()",
365                                "[EVENTGRAPH] Peer {} replied with a wrong event: {}",
366                                self.channel.display_address(), parent.id(),
367                            );
368                            self.channel.stop().await;
369                            return Err(Error::ChannelStopped)
370                        }
371
372                        debug!(
373                            target: "event_graph::protocol::handle_event_put()",
374                            "Got correct parent event {}", parent.id(),
375                        );
376
377                        if let Some(layer_events) = received_events.get_mut(&parent.layer) {
378                            layer_events.push(parent.clone());
379                        } else {
380                            let layer_events = vec![parent.clone()];
381                            received_events.insert(parent.layer, layer_events);
382                        }
383                        received_events_hashes.insert(parent_id);
384
385                        missing_parents.remove(&parent_id);
386
387                        // See if we have the upper parents
388                        for upper_parent in parent.parents.iter() {
389                            if upper_parent == &NULL_ID {
390                                continue
391                            }
392
393                            if !missing_parents.contains(upper_parent) &&
394                                !received_events_hashes.contains(upper_parent) &&
395                                !self
396                                    .event_graph
397                                    .dag
398                                    .contains_key(upper_parent.as_bytes())
399                                    .unwrap()
400                            {
401                                debug!(
402                                    target: "event_graph::protocol::handle_event_put()",
403                                    "Found upper missing parent event {upper_parent}"
404                                );
405                                missing_parents.insert(*upper_parent);
406                            }
407                        }
408                    }
409                } // <-- while !missing_parents.is_empty()
410
411                // At this point we should've got all the events.
412                // We should add them to the DAG.
413                let mut events = vec![];
414                for (_, tips) in received_events {
415                    for tip in tips {
416                        events.push(tip);
417                    }
418                }
419                if self.event_graph.dag_insert(&events).await.is_err() {
420                    self.clone().increase_malicious_count().await?;
421                    continue
422                }
423            } // <-- !missing_parents.is_empty()
424
425            // If we're here, we have all the parents, and we can now
426            // perform a full validation and add the actual event to
427            // the DAG.
428            debug!(
429                target: "event_graph::protocol::handle_event_put()",
430                "Got all parents necessary for insertion",
431            );
432            if self.event_graph.dag_insert(&[event.clone()]).await.is_err() {
433                self.clone().increase_malicious_count().await?;
434                continue
435            }
436
437            self.broadcaster_push.send(EventPut(event)).await.expect("push broadcaster closed");
438        }
439    }

441    /// Protocol function handling `EventReq`.
442    /// This is triggered whenever someone requests an event from us.
443    async fn handle_event_req(self: Arc<Self>) -> Result<()> {
444        loop {
445            let event_ids = match self.ev_req_sub.receive().await {
446                Ok(v) => v.0.clone(),
447                Err(_) => continue,
448            };
449            trace!(
450                target: "event_graph::protocol::handle_event_req()",
451                "Got EventReq: {event_ids:?} [{}]", self.channel.display_address(),
452            );
453
454            // Check if node has finished syncing its DAG
455            if !*self.event_graph.synced.read().await {
456                debug!(
457                    target: "event_graph::protocol::handle_event_req()",
458                    "DAG is still syncing, skipping..."
459                );
460                continue
461            }
462
463            // We received an event request from somebody.
464            // If we do have it, we will send it back to them as `EventRep`.
465            // Otherwise, we'll stay quiet. An honest node should always have
466            // something to reply with provided that the request is legitimate,
467            // i.e. we've sent something to them and they did not have some of
468            // the parents.
469
470            // Check if we expected this request to come around.
471            // I dunno if this is a good idea, but it seems it will help
472            // against malicious event requests where they want us to keep
473            // reading our db and steal our bandwidth.
474            let mut events = vec![];
475            for event_id in event_ids.iter() {
476                if !self.event_graph.broadcasted_ids.read().await.contains(event_id) {
477                    let malicious_count = self.malicious_count.fetch_add(1, SeqCst);
478                    if malicious_count + 1 == MALICIOUS_THRESHOLD {
479                        error!(
480                            target: "event_graph::protocol::handle_event_req()",
481                            "[EVENTGRAPH] Peer {} reached malicious threshold. Dropping connection.",
482                            self.channel.display_address(),
483                        );
484                        self.channel.stop().await;
485                        return Err(Error::ChannelStopped)
486                    }
487
488                    warn!(
489                        target: "event_graph::protocol::handle_event_req()",
490                        "[EVENTGRAPH] Peer {} requested an unexpected event {event_id:?}",
491                        self.channel.display_address()
492                    );
493                    continue
494                }
495
496                // At this point we should have it in our DAG.
497                // This code panics if this is not the case.
498                debug!(
499                    target: "event_graph::protocol::handle_event_req()",
500                    "Fetching event {event_id:?} from DAG"
501                );
502                events.push(self.event_graph.dag_get(event_id).await.unwrap().unwrap());
503            }
504
505            // Check if the incoming event is older than the genesis event. If so, something
506            // has gone wrong. The event should have been pruned during the last
507            // rotation.
508            let genesis_timestamp = self.event_graph.current_genesis.read().await.timestamp;
509            let mut bcast_ids = self.event_graph.broadcasted_ids.write().await;
510
511            for event in events.iter() {
512                if event.timestamp < genesis_timestamp {
513                    error!(
514                        target: "event_graph::protocol::handle_event_req()",
515                        "Requested event by peer {} is older than previous rotation period. It should have been pruned.
516                    Event timestamp: `{}`. Genesis timestamp: `{genesis_timestamp}`",
517                    event.id(), event.timestamp
518                    );
519                }
520
521                // Now let's get the upper level of event IDs. When we reply, we could
522                // get requests for those IDs as well.
523                for parent_id in event.parents.iter() {
524                    if parent_id != &NULL_ID {
525                        bcast_ids.insert(*parent_id);
526                    }
527                }
528            }
529            // TODO: We should remove the reply from the bcast IDs for this specific channel.
530            //       We can't remove them for everyone.
531            //bcast_ids.remove(&event_id);
532            drop(bcast_ids);
533
534            // Reply with the event
535            self.channel.send(&EventRep(events)).await?;
536        }
537    }

539    /// Protocol function handling `TipReq`.
540    /// This is triggered when someone requests the current unreferenced
541    /// tips of our DAG.
542    async fn handle_tip_req(self: Arc<Self>) -> Result<()> {
543        loop {
544            self.tip_req_sub.receive().await?;
545            trace!(
546                target: "event_graph::protocol::handle_tip_req()",
547                "Got TipReq [{}]", self.channel.display_address(),
548            );
549
550            // Check if node has finished syncing its DAG
551            if !*self.event_graph.synced.read().await {
552                debug!(
553                    target: "event_graph::protocol::handle_tip_req()",
554                    "DAG is still syncing, skipping..."
555                );
556                continue
557            }
558
559            // TODO: Rate limit
560
561            // We received a tip request. Let's find them, add them to
562            // our bcast ids list, and reply with them.
563            let layers = self.event_graph.unreferenced_tips.read().await.clone();
564            let mut bcast_ids = self.event_graph.broadcasted_ids.write().await;
565            for (_, tips) in layers.iter() {
566                for tip in tips {
567                    bcast_ids.insert(*tip);
568                }
569            }
570            drop(bcast_ids);
571
572            self.channel.send(&TipRep(layers)).await?;
573        }
574    }

576    /// We need to rate limit message propagation so malicious nodes don't get us banned
577    /// for flooding. We do that by aggregating messages here into a queue then apply
578    /// rate limit logic before broadcasting.
579    ///
580    /// The rate limit logic is this:
581    ///
582    /// * If the count is less then RATELIMIT_MIN_COUNT then do nothing.
583    /// * Otherwise sleep for `sleep_time` ms.
584    ///
585    /// To calculate the sleep time, we use the RATELIMIT_SAMPLE_* values.
586    /// For example RATELIMIT_SAMPLE_IDX = 10, RATELIMIT_SAMPLE_SLEEP = 1000
587    /// means that when N = 10, then sleep for 1000 ms.
588    ///
589    /// Let RATELIMIT_MIN_COUNT = 6, then here's a table of sleep times:
590    ///
591    /// | Count | Sleep Time / ms |
592    /// |-------|-----------------|
593    /// | 0     | 0               |
594    /// | 4     | 0               |
595    /// | 6     | 0               |
596    /// | 10    | 1000            |
597    /// | 14    | 2000            |
598    /// | 18    | 3000            |
599    ///
600    /// So we use the sample to calculate a straight line from RATELIMIT_MIN_COUNT.
601    async fn broadcast_rate_limiter(self: Arc<Self>) -> Result<()> {
602        let mut ratelimit = MovingWindow::new(RATELIMIT_EXPIRY_TIME);
603
604        loop {
605            let event_put = self.broadcaster_pull.recv().await.expect("pull broadcaster closed");
606
607            ratelimit.ticktock();
608            if ratelimit.count() > RATELIMIT_MIN_COUNT {
609                let sleep_time =
610                    ((ratelimit.count() - RATELIMIT_MIN_COUNT) * RATELIMIT_SAMPLE_SLEEP /
611                        (RATELIMIT_SAMPLE_IDX - RATELIMIT_MIN_COUNT)) as u64;
612                debug!(
613                    target: "event_graph::protocol::broadcast_rate_limiter",
614                    "Activated rate limit: sleeping {sleep_time} ms [count={}]",
615                    ratelimit.count()
616                );
617                // Apply the ratelimit
618                msleep(sleep_time).await;
619            }
620
621            // Relay the event to other peers.
622            self.event_graph
623                .p2p
624                .broadcast_with_exclude(&event_put, &[self.channel.address().clone()])
625                .await;
626        }
627    }
