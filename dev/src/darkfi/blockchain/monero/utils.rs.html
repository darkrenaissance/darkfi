24/// Returns the Keccak 256 hash of the byte input
25pub fn cn_fast_hash(data: &[u8]) -> Hash {
26    Hash::new(data)
27}
28
29/// Returns the Keccak 256 hash of 2 hashes
30pub fn cn_fast_hash2(hash1: &Hash, hash2: &Hash) -> Hash {
31    let mut tmp = [0u8; 64];
32    tmp[..32].copy_from_slice(hash1.as_bytes());
33    tmp[32..].copy_from_slice(hash2.as_bytes());
34    cn_fast_hash(&tmp)
35}

37/// Round down to power of two.
38/// Should error for count < 3 or if the count is unreasonably large
39/// for tree hash calculations.
40#[allow(unused)]
41fn tree_hash_count(count: usize) -> Result<usize> {
42    if count < 3 {
43        return Err(Error::MoneroHashingError(format!(
44            "Cannot calculate tree hash root. Expected count to be >3 but got {count}"
45        )));
46    }
47
48    if count > 0x10000000 {
49        return Err(Error::MoneroHashingError(format!(
50            "Cannot calculate tree hash root. Expected count to be less than 0x10000000 but got {count}"
51        )));
52    }
53
54    // Essentially we are doing 1 << floor(log2(count))
55    let mut pow: usize = 2;
56    while pow < count {
57        pow <<= 1;
58    }
59
60    Ok(pow >> 1)
61}

63/// Tree hash algorithm in Monero
64#[allow(unused)]
65pub fn tree_hash(hashes: &[Hash]) -> Result<Hash> {
66    if hashes.is_empty() {
67        return Err(Error::MoneroHashingError(
68            "Cannot calculate Merkle root, no hashes".to_string(),
69        ));
70    }
71
72    match hashes.len() {
73        1 => Ok(hashes[0]),
74        2 => Ok(cn_fast_hash2(&hashes[0], &hashes[1])),
75        n => {
76            let mut cnt = tree_hash_count(n)?;
77            let mut buf = vec![Hash::null(); cnt];
78
79            // c is the number of elements between the number of hashes
80            // and the next power of 2.
81            let c = 2 * cnt - hashes.len();
82
83            buf[..c].copy_from_slice(&hashes[..c]);
84
85            // hash the rest of the hashes together
86            let mut i: usize = c;
87            for b in &mut buf[c..cnt] {
88                *b = cn_fast_hash2(&hashes[i], &hashes[i + 1]);
89                i += 2;
90            }
91
92            if i != hashes.len() {
93                return Err(Error::MoneroHashingError(
94                    "Cannot calculate the Merkle root, hashes not equal to count".to_string(),
95                ));
96            }
97
98            while cnt > 2 {
99                cnt >>= 1;
100                let mut i = 0;
101                for j in 0..cnt {
102                    buf[j] = cn_fast_hash2(&buf[i], &buf[i + 1]);
103                    i += 2;
104                }
105            }
106
107            Ok(cn_fast_hash2(&buf[0], &buf[1]))
108        }
109    }
110}

112/// Creates a Merkle proof for the given hash within the set of hashes.
113/// This function returns None if the hash is not in hashes.
114/// This is a port of Monero's tree_branch function.
115#[allow(clippy::cognitive_complexity)]
116#[allow(unused)]
117pub fn create_merkle_proof(hashes: &[Hash], hash: &Hash) -> Option<MerkleProof> {
118    match hashes.len() {
119        0 => None,
120        1 => {
121            if hashes[0] != *hash {
122                return None;
123            }
124            MerkleProof::try_construct(vec![], 0)
125        }
126        2 => hashes.iter().enumerate().find_map(|(pos, h)| {
127            if h != hash {
128                return None
129            }
130            let i = usize::from(pos == 0);
131            MerkleProof::try_construct(vec![hashes[i]], u32::from(pos != 0))
132        }),
133        len => {
134            let mut idx = hashes.iter().position(|node| node == hash)?;
135            let mut count = tree_hash_count(len).ok()?;
136
137            let mut ints = vec![Hash::null(); count];
138
139            let c = 2 * count - len;
140            ints[..c].copy_from_slice(&hashes[..c]);
141
142            let mut branch = vec![];
143            let mut path = 0u32;
144            let mut i = c;
145            for (j, val) in ints.iter_mut().enumerate().take(count).skip(c) {
146                // Left or right
147                if idx == i || idx == i + 1 {
148                    let ii = if idx == i { i + 1 } else { i };
149                    branch.push(hashes[ii]);
150                    path = (path << 1) | u32::from(idx != i);
151                    idx = j;
152                }
153                *val = cn_fast_hash2(&hashes[i], &hashes[i + 1]);
154                i += 2;
155            }
156
157            debug_assert_eq!(i, len);
158
159            while count > 2 {
160                count >>= 1;
161                let mut i = 0;
162                for j in 0..count {
163                    if idx == i || idx == i + 1 {
164                        let ii = if idx == i { i + 1 } else { i };
165                        branch.push(ints[ii]);
166                        path = (path << 1) | u32::from(idx != i);
167                        idx = j;
168                    }
169                    ints[j] = cn_fast_hash2(&ints[i], &ints[i + 1]);
170                    i += 2;
171                }
172            }
173
174            if idx == 0 || idx == 1 {
175                let ii = usize::from(idx == 0);
176                branch.push(ints[ii]);
177                path = (path << 1) | u32::from(idx != 0);
178            }
179
180            MerkleProof::try_construct(branch, path)
181        }
182    }
