44/// This struct represents all the Proof of Work information required
45/// for merge mining.
46#[derive(Clone)]
47pub struct MoneroPowData {
48    /// Monero Header fields
49    pub header: BlockHeader,
50    /// RandomX VM key - length varies to a max len of 60.
51    /// TODO: Implement a type, or use randomx_key[0] to define len.
52    pub randomx_key: [u8; 64],
53    /// The number of transactions included in this Monero block.
54    /// This is used to produce the blockhashing_blob.
55    pub transaction_count: u16,
56    /// Transaction root
57    pub merkle_root: Hash,
58    /// Coinbase Merkle proof hashes
59    pub coinbase_merkle_proof: MerkleProof,
60    /// Incomplete hashed state of the coinbase transaction
61    pub coinbase_tx_hasher: Keccak,
62    /// Extra field of the coinbase
63    pub coinbase_tx_extra: RawExtraField,
64    /// Aux chain Merkle proof hashes
65    pub aux_chain_merkle_proof: MerkleProof,
66}

68impl fmt::Debug for MoneroPowData {
69    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
70        let mut digest = [0u8; 32];
71        self.coinbase_tx_hasher.clone().finalize(&mut digest);
72        f.debug_struct("MoneroPowData")
73            .field("header", &self.header)
74            .field("randomx_key", &self.randomx_key)
75            .field("transaction_count", &self.transaction_count)
76            .field("merkle_root", &self.merkle_root)
77            .field("coinbase_merkle_proof", &self.coinbase_merkle_proof)
78            .field("coinbase_tx_extra", &self.coinbase_tx_extra)
79            .field("aux_chain_merkle_proof", &self.aux_chain_merkle_proof)
80            .finish()
81    }
82}

84impl Encodable for MoneroPowData {
85    fn encode<S: Write>(&self, s: &mut S) -> io::Result<usize> {
86        let mut n = 0;
87
88        n += self.header.consensus_encode(s)?;
89        n += self.randomx_key.encode(s)?;
90        n += self.transaction_count.encode(s)?;
91        n += self.merkle_root.consensus_encode(s)?;
92        n += self.coinbase_merkle_proof.encode(s)?;
93
94        // This is an incomplete hasher. Dump it from memory
95        // and write it down. We can restore it the same way.
96        let buf = keccak_to_bytes(&self.coinbase_tx_hasher);
97        n += buf.encode(s)?;
98
99        n += self.coinbase_tx_extra.0.encode(s)?;
100        n += self.aux_chain_merkle_proof.encode(s)?;
101
102        Ok(n)
103    }
104}

137impl Decodable for MoneroPowData {
138    fn decode<D: Read>(d: &mut D) -> io::Result<Self> {
139        let header =
140            BlockHeader::consensus_decode(d).map_err(|_| Error::other("Invalid XMR header"))?;
141
142        let randomx_key: [u8; 64] = Decodable::decode(d)?;
143        let transaction_count: u16 = Decodable::decode(d)?;
144
145        let merkle_root =
146            Hash::consensus_decode(d).map_err(|_| Error::other("Invamid XMR hash"))?;
147
148        let coinbase_merkle_proof: MerkleProof = Decodable::decode(d)?;
149
150        let buf: Vec<u8> = Decodable::decode(d)?;
151        let coinbase_tx_hasher = keccak_from_bytes(&buf);
152
153        let coinbase_tx_extra: Vec<u8> = Decodable::decode(d)?;
154        let coinbase_tx_extra = RawExtraField(coinbase_tx_extra);
155        let aux_chain_merkle_proof: MerkleProof = Decodable::decode(d)?;
156
157        Ok(Self {
158            header,
159            randomx_key,
160            transaction_count,
161            merkle_root,
162            coinbase_merkle_proof,
163            coinbase_tx_hasher,
164            coinbase_tx_extra,
165            aux_chain_merkle_proof,
166        })
167    }
168}

209impl MoneroPowData {
210    /// Returns true if the coinbase Merkle proof produces the `merkle_root` hash.
211    pub fn is_coinbase_valid_merkle_root(&self) -> bool {
212        let mut finalised_prefix_keccak = self.coinbase_tx_hasher.clone();
213        let mut encoder_extra_field = vec![];
214        self.coinbase_tx_extra.consensus_encode(&mut encoder_extra_field).unwrap();
215        finalised_prefix_keccak.update(&encoder_extra_field);
216        let mut prefix_hash: [u8; 32] = [0; 32];
217        finalised_prefix_keccak.finalize(&mut prefix_hash);
218
219        let final_prefix_hash = Hash::from_slice(&prefix_hash);
220
221        // let mut finalised_keccak = Keccak::v256();
222        let rct_sig_base = RctSigBase {
223            rct_type: RctType::Null,
224            txn_fee: Default::default(),
225            pseudo_outs: vec![],
226            ecdh_info: vec![],
227            out_pk: vec![],
228        };
229
230        let hashes = vec![final_prefix_hash, rct_sig_base.hash(), Hash::null()];
231        let encoder_final: Vec<u8> =
232            hashes.into_iter().flat_map(|h| Vec::from(&h.to_bytes()[..])).collect();
233        let coinbase_hash = Hash::new(encoder_final);
234
235        let merkle_root = self.coinbase_merkle_proof.calculate_root(&coinbase_hash);
236        (self.merkle_root == merkle_root) && self.coinbase_merkle_proof.check_coinbase_path()
237    }
