37pub const SLED_CONTRACTS_TREE: &[u8] = b"_contracts";
38pub const SLED_BINCODE_TREE: &[u8] = b"_wasm_bincode";

40/// The hardcoded db name for the zkas circuits database tree
41pub const SMART_CONTRACT_ZKAS_DB_NAME: &str = "_zkas";

43/// The `ContractStore` is a structure representing all `sled` trees related
44/// to storing the blockchain's contracts information.
45#[derive(Clone)]
46pub struct ContractStore {
47    /// The `sled` tree storing the wasm bincode for deployed contracts.
48    /// The layout looks like this:
49    /// ```plaintext
50    ///  tree: "_wasm_bincode"
51    ///   key: ContractId
52    /// value: Vec<u8>
53    pub wasm: sled::Tree,
54    /// The `sled` tree storing the pointers to contracts' databases.
55    /// See the rustdoc for the impl functions for more info.
56    /// The layout looks like this:
57    /// ```plaintext
58    ///  tree: "_contracts"
59    ///   key: ContractId
60    /// value: Vec<blake3(ContractId || tree_name)>
61    /// ```
62    /// These values get mutated with `init()` and `remove()`.
63    pub state: sled::Tree,
64}
65
66impl ContractStore {
67    /// Opens a new or existing `ContractStore` on the given sled database.
68    pub fn new(db: &sled::Db) -> Result<Self> {
69        let wasm = db.open_tree(SLED_BINCODE_TREE)?;
70        let state = db.open_tree(SLED_CONTRACTS_TREE)?;
71        Ok(Self { wasm, state })
72    }

74    /// Fetches the bincode for a given ContractId from the store's wasm tree.
75    /// Returns an error if the bincode is not found.
76    pub fn get(&self, contract_id: ContractId) -> Result<Vec<u8>> {
77        if let Some(bincode) = self.wasm.get(serialize(&contract_id))? {
78            return Ok(bincode.to_vec())
79        }
80
81        Err(Error::WasmBincodeNotFound)
82    }

84    /// Do a lookup of an existing contract state. In order to succeed, the
85    /// state must have been previously initialized with `init()`. If the
86    /// state has been found, a handle to it will be returned. Otherwise, we
87    /// return an error.
88    pub fn lookup(
89        &self,
90        db: &sled::Db,
91        contract_id: &ContractId,
92        tree_name: &str,
93    ) -> Result<sled::Tree> {
94        debug!(target: "blockchain::contractstore", "Looking up state tree for {}:{}", contract_id, tree_name);
95
96        // A guard to make sure we went through init()
97        let contract_id_bytes = serialize(contract_id);
98        if !self.state.contains_key(&contract_id_bytes)? {
99            return Err(Error::ContractNotFound(contract_id.to_string()))
100        }
101
102        let state_pointers = self.state.get(&contract_id_bytes)?.unwrap();
103        let state_pointers: Vec<[u8; 32]> = deserialize(&state_pointers)?;
104
105        // We assume the tree has been created already, so it should be listed
106        // in this array. If not, that's an error.
107        let ptr = contract_id.hash_state_id(tree_name);
108        if !state_pointers.contains(&ptr) {
109            return Err(Error::ContractStateNotFound)
110        }
111
112        // We open the tree and return its handle
113        let tree = db.open_tree(ptr)?;
114        Ok(tree)
115    }
116
117    /// Attempt to remove an existing contract state. In order to succeed, the
118    /// state must have been previously initialized with `init()`. If the state
119    /// has been found, its contents in the tree will be cleared, and the pointer
120    /// will be removed from the main `ContractStateStore`. If anything is not
121    /// found as initialized, an error is returned.
122    /// NOTE: this function is not used right now, we keep it for future proofing,
123    ///       and its obviously untested.
124    pub fn remove(&self, db: &sled::Db, contract_id: &ContractId, tree_name: &str) -> Result<()> {
125        debug!(target: "blockchain::contractstore", "Removing state tree for {}:{}", contract_id, tree_name);
126
127        // A guard to make sure we went through init()
128        let contract_id_bytes = serialize(contract_id);
129        if !self.state.contains_key(&contract_id_bytes)? {
130            return Err(Error::ContractNotFound(contract_id.to_string()))
131        }
132
133        let state_pointers = self.state.get(&contract_id_bytes)?.unwrap();
134        let mut state_pointers: Vec<[u8; 32]> = deserialize(&state_pointers)?;
135
136        // We assume the tree has been created already, so it should be listed
137        // in this array. If not, that's an error.
138        let ptr = contract_id.hash_state_id(tree_name);
139        if !state_pointers.contains(&ptr) {
140            return Err(Error::ContractStateNotFound)
141        }
142
143        // Remove the deleted tree from the state pointer set.
144        state_pointers.retain(|x| *x != ptr);
145        self.state.insert(contract_id_bytes, serialize(&state_pointers))?;
146
147        // Drop the deleted tree from the database
148        db.drop_tree(ptr)?;
149
150        Ok(())
151    }

153    /// Abstraction function for fetching a `ZkBinary` and its respective `VerifyingKey`
154    /// from a contract's zkas sled tree.
155    pub fn get_zkas(
156        &self,
157        db: &sled::Db,
158        contract_id: &ContractId,
159        zkas_ns: &str,
160    ) -> Result<(ZkBinary, VerifyingKey)> {
161        debug!(target: "blockchain::contractstore", "Looking up \"{}:{}\" zkas circuit & vk", contract_id, zkas_ns);
162
163        let zkas_tree = self.lookup(db, contract_id, SMART_CONTRACT_ZKAS_DB_NAME)?;
164
165        let Some(zkas_bytes) = zkas_tree.get(serialize(&zkas_ns))? else {
166            return Err(Error::ZkasBincodeNotFound)
167        };
168
169        // If anything in this function panics, that means corrupted data managed
170        // to get into this sled tree. This should not be possible.
171        let (zkbin, vkbin): (Vec<u8>, Vec<u8>) = deserialize(&zkas_bytes).unwrap();
172
173        // The first vec is the compiled zkas binary
174        let zkbin = ZkBinary::decode(&zkbin).unwrap();
175
176        // Construct the circuit to be able to read the VerifyingKey
177        let circuit = ZkCircuit::new(empty_witnesses(&zkbin).unwrap(), &zkbin);
178
179        // The second one is the serialized VerifyingKey for it
180        let mut vk_buf = Cursor::new(vkbin);
181        let vk = VerifyingKey::read::<Cursor<Vec<u8>>, ZkCircuit>(&mut vk_buf, circuit).unwrap();
182
183        Ok((zkbin, vk))
184    }

186    /// Retrieve all wasm bincodes from the store's wasm tree in the form
187    /// of a tuple (`contract_id`, `bincode`).
188    /// Be careful as this will try to load everything in memory.
189    pub fn get_all_wasm(&self) -> Result<Vec<(ContractId, Vec<u8>)>> {
190        let mut bincodes = vec![];
191
192        for bincode in self.wasm.iter() {
193            let bincode = bincode.unwrap();
194            let contract_id = deserialize(&bincode.0)?;
195            bincodes.push((contract_id, bincode.1.to_vec()));
196        }
197
198        Ok(bincodes)
199    }

204    pub fn get_all_states(&self) -> Result<Vec<(ContractId, Vec<blake3::Hash>)>> {

206
207        for contract in self.state.iter() {
208            contracts.push(parse_record(contract.unwrap())?);

214    /// Retrieve provided key value bytes from a contract's zkas sled tree.
215    pub fn get_state_tree_value(
216        &self,
217        db: &sled::Db,
218        contract_id: &ContractId,
219        tree_name: &str,
220        key: &[u8],
221    ) -> Result<Vec<u8>> {
222        debug!(target: "blockchain::contractstore", "Looking up state tree value for {}:{}", contract_id, tree_name);
223
224        // Grab the state tree
225        let state_tree = self.lookup(db, contract_id, tree_name)?;
226
227        // Grab the key value
228        match state_tree.get(key)? {
229            Some(value) => Ok(value.to_vec()),
230            None => Err(Error::DatabaseError(format!(
231                "State tree {}:{} doesn't contain key: {:?}",
232                contract_id, tree_name, key
233            ))),
234        }
235    }

237    /// Retrieve all records from a contract's zkas sled tree, as a `BTreeMap`.
238    /// Be careful as this will try to load everything in memory.
239    pub fn get_state_tree_records(
240        &self,
241        db: &sled::Db,
242        contract_id: &ContractId,
243        tree_name: &str,
244    ) -> Result<BTreeMap<Vec<u8>, Vec<u8>>> {
245        debug!(target: "blockchain::contractstore", "Looking up state tree records for {}:{}", contract_id, tree_name);
246
247        // Grab the state tree
248        let state_tree = self.lookup(db, contract_id, tree_name)?;
249
250        // Retrieve its records
251        let mut ret = BTreeMap::new();
252        for record in state_tree.iter() {
253            let (key, value) = record.unwrap();
254            ret.insert(key.to_vec(), value.to_vec());
255        }
256
257        Ok(ret)
258    }

260    /// Generate a Monotree(SMT) containing all contracts states
261    /// checksums, along with the wasm bincodes checksum.
262    ///
263    /// Note: native contracts wasm bincodes are excluded.
264    pub fn get_state_monotree(&self, db: &sled::Db) -> Result<Monotree> {
265        // Initialize the monotree
266        let mut root = None;
267        let mut tree = Monotree::new();
268
269        // Iterate over current contracts states records
270        // TODO: parallelize this with a threadpool
271        for state_record in self.state.iter().values() {
272            // Iterate over contract states pointers
273            let state_pointers: Vec<[u8; 32]> = deserialize(&state_record?)?;
274            for state_ptr in state_pointers {
275                // Grab the state tree
276                let state_tree = db.open_tree(state_ptr)?;
277
278                // Compute its checksum
279                let checksum = sled_tree_checksum(&state_tree)?;
280
281                // Insert record to monotree
282                root = tree.insert(root.as_ref(), &state_ptr, &checksum)?;
283                tree.set_headroot(root.as_ref());
284            }
285        }
286
287        // Iterate over current contracts wasm bincodes to compute its checksum
288        let mut hasher = blake3::Hasher::new();
289        for record in self.wasm.iter() {
290            let (key, value) = record?;
291
292            // Skip native ones
293            if key == MONEY_CONTRACT_ID.to_bytes() ||
294                key == DAO_CONTRACT_ID.to_bytes() ||
295                key == DEPLOYOOOR_CONTRACT_ID.to_bytes()
296            {
297                continue
298            }
299
300            // Hash record
301            hasher.update(&key);
302            hasher.update(&value);
303        }
304
305        // Insert wasm bincodes record to monotree
306        root = tree.insert(
307            root.as_ref(),
308            blake3::hash(SLED_BINCODE_TREE).as_bytes(),
309            hasher.finalize().as_bytes(),
310        )?;
311        tree.set_headroot(root.as_ref());
312
313        Ok(tree)
314    }

317/// Overlay structure over a [`ContractStore`] instance.

320impl ContractStoreOverlay {
321    pub fn new(overlay: &SledDbOverlayPtr) -> Result<Self> {
322        overlay.lock().unwrap().open_tree(SLED_BINCODE_TREE, true)?;
323        overlay.lock().unwrap().open_tree(SLED_CONTRACTS_TREE, true)?;
324        Ok(Self(overlay.clone()))
325    }

327    /// Fetches the bincode for a given ContractId from the overlay's wasm tree.
328    /// Returns an error if the bincode is not found.
329    pub fn get(&self, contract_id: ContractId) -> Result<Vec<u8>> {
330        if let Some(bincode) =
331            self.0.lock().unwrap().get(SLED_BINCODE_TREE, &serialize(&contract_id))?
332        {
333            return Ok(bincode.to_vec())
334        }
335
336        Err(Error::WasmBincodeNotFound)
337    }

339    /// Inserts or replaces the bincode for a given ContractId into the overlay's
340    /// wasm tree.
341    pub fn insert(&self, contract_id: ContractId, bincode: &[u8]) -> Result<()> {
342        if let Err(e) =
343            self.0.lock().unwrap().insert(SLED_BINCODE_TREE, &serialize(&contract_id), bincode)
344        {
345            error!(target: "blockchain::contractstoreoverlay", "Failed to insert bincode to Wasm tree: {}", e);
346            return Err(e.into())
347        }
348
349        Ok(())
350    }

352    /// Try to initialize a new contract state. Contracts can create a number
353    /// of trees, separated by `tree_name`, which they can then use from the
354    /// smart contract API. `init()` will look into the main `ContractStateStoreOverlay`
355    /// tree to check if the smart contract was already deployed, and if so
356    /// it will fetch a vector of these states that were initialized. If the
357    /// state was already found, this function will return an error, because
358    /// in this case the handle should be fetched using `lookup()`.
359    /// If the tree was not initialized previously, it will be appended to
360    /// the main `ContractStateStoreOverlay` tree and a handle to it will be
361    /// returned.
362    pub fn init(&self, contract_id: &ContractId, tree_name: &str) -> Result<[u8; 32]> {
363        debug!(target: "blockchain::contractstoreoverlay", "Initializing state overlay tree for {}:{}", contract_id, tree_name);
364        let mut lock = self.0.lock().unwrap();
365
366        // See if there are existing state trees.
367        // If not, just start with an empty vector.
368        let contract_id_bytes = serialize(contract_id);
369        let mut state_pointers: Vec<[u8; 32]> =
370            if lock.contains_key(SLED_CONTRACTS_TREE, &contract_id_bytes)? {
371                let bytes = lock.get(SLED_CONTRACTS_TREE, &contract_id_bytes)?.unwrap();
372                deserialize(&bytes)?
373            } else {
374                vec![]
375            };
376
377        // If the db was never initialized, it should not be in here.
378        let ptr = contract_id.hash_state_id(tree_name);
379        if state_pointers.contains(&ptr) {
380            return Err(Error::ContractAlreadyInitialized)
381        }
382
383        // Now we add it so it's marked as initialized and create its tree.
384        state_pointers.push(ptr);
385        lock.insert(SLED_CONTRACTS_TREE, &contract_id_bytes, &serialize(&state_pointers))?;
386        lock.open_tree(&ptr, false)?;
387
388        Ok(ptr)
389    }

452    /// Generate a Monotree(SMT) containing all contracts states
453    /// checksums, along with the wasm bincodes checksum.
454    /// Be carefull as this will open all states trees in the overlay.
455    ///
456    /// Note: native contracts wasm bincodes are excluded.
457    pub fn get_state_monotree(&self) -> Result<Monotree> {
458        let mut lock = self.0.lock().unwrap();
459
460        // Grab all states pointers
461        let mut states_pointers = vec![];
462        for state_record in lock.iter(SLED_CONTRACTS_TREE)? {
463            let state_pointers: Vec<[u8; 32]> = deserialize(&state_record?.1)?;
464            for state_ptr in state_pointers {
465                states_pointers.push(state_ptr);
466            }
467        }
468
469        // Initialize the monotree
470        let mut root = None;
471        let mut tree = Monotree::new();
472
473        // Iterate over contract states pointers
474        // TODO: parallelize this with a threadpool
475        for state_ptr in states_pointers {
476            // Open the state tree in the overlay
477            lock.open_tree(&state_ptr, false)?;
478
479            // Compute its checksum
480            let checksum = sled_overlay_tree_checksum(&lock, &state_ptr)?;
481
482            // Insert record to monotree
483            root = tree.insert(root.as_ref(), &state_ptr, &checksum)?;
484            tree.set_headroot(root.as_ref());
485        }
486
487        // Iterate over current contracts wasm bincodes to compute its checksum
488        let mut hasher = blake3::Hasher::new();
489        for record in lock.iter(SLED_BINCODE_TREE)? {
490            let (key, value) = record?;
491
492            // Skip native ones
493            if key == MONEY_CONTRACT_ID.to_bytes() ||
494                key == DAO_CONTRACT_ID.to_bytes() ||
495                key == DEPLOYOOOR_CONTRACT_ID.to_bytes()
496            {
497                continue
498            }
499
500            // Hash record
501            hasher.update(&key);
502            hasher.update(&value);
503        }
504
505        // Insert wasm bincodes record to monotree
506        root = tree.insert(
507            root.as_ref(),
508            blake3::hash(SLED_BINCODE_TREE).as_bytes(),
509            hasher.finalize().as_bytes(),
510        )?;
511        tree.set_headroot(root.as_ref());
512
513        Ok(tree)
514    }

516    /// Compute all updated contracts states and wasm bincodes
517    /// checksums and update their records in the provided
518    /// Monotree(SMT).
519    pub fn update_state_monotree(&self, tree: &mut Monotree) -> Result<()> {
520        let lock = self.0.lock().unwrap();
521
522        // Iterate over overlay's caches
523        // TODO: parallelize this with a threadpool
524        let mut root = tree.get_headroot()?;
525        for (state_key, state_cache) in &lock.state.caches {
526            // Check if that cache is a contract state one.
527            // Overlay protected trees are all the native/non-contract ones.
528            if !lock.state.protected_tree_names.contains(state_key) {
529                // Compute its checksum
530                let checksum = sled_overlay_tree_checksum(&lock, state_key)?;
531
532                // Insert record to monotree
533                root = tree.insert(root.as_ref(), &deserialize(state_key)?, &checksum)?;
534                tree.set_headroot(root.as_ref());
535
536                continue
537            }
538
539            // Skip if its not the wasm bincodes cache
540            if state_key != SLED_BINCODE_TREE {
541                continue
542            }
543
544            // Check if wasm bincodes cache is updated
545            if state_cache.state.cache.is_empty() && state_cache.state.removed.is_empty() {
546                continue
547            }
548
549            // Iterate over current contracts wasm bincodes to compute
550            // its checksum.
551            let mut hasher = blake3::Hasher::new();
552            for record in lock.iter(SLED_BINCODE_TREE)? {
553                let (key, value) = record?;
554
555                // Skip native ones
556                if key == MONEY_CONTRACT_ID.to_bytes() ||
557                    key == DAO_CONTRACT_ID.to_bytes() ||
558                    key == DEPLOYOOOR_CONTRACT_ID.to_bytes()
559                {
560                    continue
561                }
562
563                // Hash record
564                hasher.update(&key);
565                hasher.update(&value);
566            }
567
568            // Insert wasm bincodes record to monotree
569            root = tree.insert(
570                root.as_ref(),
571                blake3::hash(SLED_BINCODE_TREE).as_bytes(),
572                hasher.finalize().as_bytes(),
573            )?;
574            tree.set_headroot(root.as_ref());
575        }
576
577        Ok(())
578    }

581/// Auxiliary function to compute a blake3 checksum for provided sled
582/// tree.
583fn sled_tree_checksum(tree: &sled::Tree) -> Result<[u8; 32]> {
584    // Generate a new blake3 hashed
585    let mut hasher = blake3::Hasher::new();
586
587    // Iterate over tree records to compute its checksum
588    for record in tree.iter() {
589        let (key, value) = record?;
590        hasher.update(&key);
591        hasher.update(&value);
592    }
593
594    // Return the finalized hasher bytes
595    Ok(*hasher.finalize().as_bytes())
596}

598/// Auxiliary function to compute a blake3 checksum for provided sled
599/// overlay tree.
600fn sled_overlay_tree_checksum(overlay: &SledDbOverlay, tree_key: &[u8]) -> Result<[u8; 32]> {
601    // Generate a new blake3 hashed
602    let mut hasher = blake3::Hasher::new();
603
604    // Iterate over tree records to compute its checksum
605    for record in overlay.iter(tree_key)? {
606        let (key, value) = record?;
607        hasher.update(&key);
608        hasher.update(&value);
609    }
