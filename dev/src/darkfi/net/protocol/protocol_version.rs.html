38/// Implements the protocol version handshake sent out by nodes at
39/// the beginning of a connection.
40pub struct ProtocolVersion {
41    channel: ChannelPtr,
42    version_sub: MessageSubscription<VersionMessage>,
43    verack_sub: MessageSubscription<VerackMessage>,
44    settings: Arc<AsyncRwLock<Settings>>,
45}
46
47impl ProtocolVersion {
48    /// Create a new version protocol. Makes a version and version ack
49    /// subscription, then adds them to a version protocol instance.
50    // TODO: This function takes settings as a param, however, it is also reachable through Channel.
51    //       Maybe we want to navigate towards Settings through channel->session->p2p->settings
52    pub async fn new(channel: ChannelPtr, settings: Arc<AsyncRwLock<Settings>>) -> Arc<Self> {
53        // Creates a version subscription
54        let version_sub =
55            channel.subscribe_msg::<VersionMessage>().await.expect("Missing version dispatcher!");
56
57        // Creates a version acknowledgement subscription
58        let verack_sub =
59            channel.subscribe_msg::<VerackMessage>().await.expect("Missing verack dispatcher!");
60
61        Arc::new(Self { channel, version_sub, verack_sub, settings })
62    }
63
64    /// Start version information exchange. Start the timer. Send version
65    /// info and wait for version ack. Wait for version info and send
66    /// version ack.
67    pub async fn run(self: Arc<Self>, executor: Arc<Executor<'_>>) -> Result<()> {
68        debug!(target: "net::protocol_version::run()", "START => address={}", self.channel.display_address());
69        let timeout =
70            Timer::after(Duration::from_secs(self.settings.read().await.channel_handshake_timeout));
71        let version = self.clone().exchange_versions(executor);
72
73        pin_mut!(timeout);
74        pin_mut!(version);
75
76        // Run timer and version exchange at the same time. Either deal
77        // with the success or failure of the version exchange or
78        // time out.
79        match select(version, timeout).await {
80            Either::Left((Ok(_), _)) => {
81                debug!(target: "net::protocol_version::run()", "END => address={}",
82                self.channel.display_address());
83
84                Ok(())
85            }
86            Either::Left((Err(e), _)) => {
87                error!(
88                    target: "net::protocol_version::run()",
89                    "[P2P] Version Exchange failed [{}]: {e}",
90                    self.channel.display_address()
91                );
92
93                self.channel.stop().await;
94                Err(e)
95            }
96
97            Either::Right((_, _)) => {
98                error!(
99                    target: "net::protocol_version::run()",
100                    "[P2P] Version Exchange timed out [{}]",
101                    self.channel.display_address(),
102                );
103
104                self.channel.stop().await;
105                Err(Error::ChannelTimeout)
106            }
107        }
108    }
109
110    /// Send and receive version information
111    async fn exchange_versions(self: Arc<Self>, executor: Arc<Executor<'_>>) -> Result<()> {
112        debug!(
113            target: "net::protocol_version::exchange_versions()",
114            "START => address={}", self.channel.display_address(),
115        );
116
117        let send = executor.spawn(self.clone().send_version());
118        let recv = executor.spawn(self.clone().recv_version());
119
120        let rets = join_all(vec![send, recv]).await;
121        if let Err(e) = &rets[0] {
122            error!(
123                target: "net::protocol_version::exchange_versions()",
124                "send_version() failed: {e}"
125            );
126            return Err(e.clone())
127        }
128
129        if let Err(e) = &rets[1] {
130            error!(
131                target: "net::protocol_version::exchange_versions()",
132                "recv_version() failed: {e}"
133            );
134            return Err(e.clone())
135        }
136
137        debug!(
138            target: "net::protocol_version::exchange_versions()",
139            "END => address={}", self.channel.display_address(),
140        );
141        Ok(())
142    }
143
144    /// Send version info and wait for version acknowledgement.
145    /// Ensures that the app version is the same.
146    async fn send_version(self: Arc<Self>) -> Result<()> {
147        debug!(
148            target: "net::protocol_version::send_version()",
149            "START => address={}", self.channel.display_address(),
150        );
151
152        let settings = self.settings.read().await;
153        let node_id = settings.node_id.clone();
154        let app_version = settings.app_version.clone();
155        drop(settings);
156
157        let external_addrs = self.channel.hosts().external_addrs().await;
158
159        let version = VersionMessage {
160            node_id,
161            version: app_version.clone(),
162            timestamp: UNIX_EPOCH.elapsed().unwrap().as_secs(),
163            connect_recv_addr: self.channel.connect_addr().clone(),
164            resolve_recv_addr: self.channel.resolve_addr(),
165            ext_send_addr: external_addrs,
166            /* NOTE: `features` is a list of enabled features in the
167            format Vec<(service, version)>. In the future, Protocols will
168            add their own data to this field when they are attached.*/
169            features: vec![],
170        };
171        self.channel.send(&version).await?;
172
173        // Wait for verack
174        let verack_msg = self.verack_sub.receive().await?;
175
176        // Validate peer received version against our version.
177        debug!(
178            target: "net::protocol_version::send_version()",
179            "App version: {app_version}, Recv version: {}",
180            verack_msg.app_version,
181        );
182
183        // MAJOR and MINOR should be the same.
184        if app_version.major != verack_msg.app_version.major ||
185            app_version.minor != verack_msg.app_version.minor
186        {
187            error!(
188                target: "net::protocol_version::send_version()",
189                "[P2P] Version mismatch from {}. Disconnecting...",
190                self.channel.display_address(),
191            );
192
193            self.channel.stop().await;
194            return Err(Error::ChannelStopped)
195        }
196
197        // Versions are compatible
198        debug!(
199            target: "net::protocol_version::send_version()",
200            "END => address={}", self.channel.display_address(),
201        );
202        Ok(())
203    }
204
205    /// Receive version info, check the message is okay and send verack
206    /// with app version attached.
207    async fn recv_version(self: Arc<Self>) -> Result<()> {
208        debug!(
209            target: "net::protocol_version::recv_version()",
210            "START => address={}", self.channel.display_address(),
211        );
212
213        // Receive version message
214        let version = self.version_sub.receive().await?;
215        if let Some(ipv6_addr) = version.get_ipv6_addr() {
216            let hosts = self.channel.p2p().hosts();
217            hosts.add_auto_addr(ipv6_addr);
218        }
219        self.channel.set_version(version).await;
220
221        // Send verack
222        let verack = VerackMessage { app_version: self.settings.read().await.app_version.clone() };
223        self.channel.send(&verack).await?;
224
225        debug!(
226            target: "net::protocol_version::recv_version()",
227            "END => address={}", self.channel.display_address(),
228        );
229        Ok(())
230    }
