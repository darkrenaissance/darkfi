22type BlacklistEntry = (String, Vec<String>, Vec<u16>);

24/// Ban policies definitions.
25///
26/// If the ban policy is set to `Relaxed` will not ban peers in case
27/// they send a message without a corresponding MessageDispatcher.
28/// This is useful for nodes that may not be subscribed to protocols,
29/// such as Lilith. For most uses this should be set to `Strict`.
30///
31/// TODO: this will be deprecated when we introduce the p2p resource
32/// mananger.
33#[derive(Clone, Debug, Default, PartialEq, serde::Deserialize, serde::Serialize)]
34#[serde(rename_all = "lowercase")]
35pub enum BanPolicy {
36    #[default]
37    Strict,
38
39    Relaxed,
40}

42/// P2P network settings. The scope of this is a P2P network instance
43/// configured by the library user.
44#[derive(Debug, Clone)]
45pub struct Settings {
46    /// Only used for debugging, compromises privacy when set
47    pub node_id: String,
48    /// P2P accept addresses the instance listens on for inbound connections
49    pub inbound_addrs: Vec<Url>,
50    /// P2P external addresses the instance advertises so other peers can
51    /// reach us and connect to us, as long as inbound addrs are configured
52    pub external_addrs: Vec<Url>,
53    /// Peer nodes to manually connect to
54    pub peers: Vec<Url>,
55    /// Seed nodes to connect to for peer discovery and/or advertising our
56    /// own external addresses
57    pub seeds: Vec<Url>,
58    /// Magic bytes should be unique per P2P network.
59    /// Avoid bleeding of networks.
60    pub magic_bytes: MagicBytes,
61    /// Application version, used for convenient protocol matching
62    pub app_version: semver::Version,
63    /// Whitelisted network transports for outbound connections
64    pub allowed_transports: Vec<String>,
65    /// Transports allowed to be mixed (tcp, tcp+tls, tor, tor+tls)
66    /// When transport is added to this list the corresponding transport
67    /// in allowed_transports is used to connect to the node.
68    /// Supported mixing scenarios include
69    /// allowed_transport | mixed_transport
70    ///        tor        |     tcp
71    ///       tor+tls     |    tcp+tls
72    ///       socks5      |      tor
73    ///       socks5      |      tcp
74    ///      socks5+tls   |    tor+tls
75    ///      socks5+tls   |    tcp+tls
76    pub mixed_transports: Vec<String>,
77    /// Tor socks5 proxy to connect to when socks5 or socks5+tls are added to allowed transports
78    /// and transport mixing is enabled
79    pub tor_socks5_proxy: Option<Url>,
80    /// Nym socks5 proxy to connect to when socks5 or socks5+tls are added to allowed transports
81    /// and transport mixing is enabled
82    pub nym_socks5_proxy: Option<Url>,
83    /// I2p Socks5 proxy to connect to i2p eepsite (hidden services)
84    pub i2p_socks5_proxy: Url,
85    /// Outbound connection slots number, this many connections will be
86    /// attempted. (This does not include manual connections)
87    pub outbound_connections: usize,
88    /// Inbound connection slots number, this many active listening connections
89    /// will be allowed. (This does not include manual connections)
90    pub inbound_connections: usize,
91    /// Outbound connection timeout (in seconds)
92    pub outbound_connect_timeout: u64,
93    /// Exchange versions (handshake) timeout (in seconds)
94    pub channel_handshake_timeout: u64,
95    /// Ping-pong exchange execution interval (in seconds)
96    pub channel_heartbeat_interval: u64,
97    /// Allow localnet hosts
98    pub localnet: bool,
99    /// Cooling off time for peer discovery when unsuccessful
100    pub outbound_peer_discovery_cooloff_time: u64,
101    /// Time between peer discovery attempts
102    pub outbound_peer_discovery_attempt_time: u64,
103    /// P2P datastore path
104    pub p2p_datastore: Option<String>,
105    /// Hostlist storage path
106    pub hostlist: Option<String>,
107    /// Pause interval within greylist refinery process
108    pub greylist_refinery_interval: u64,
109    /// Percent of connections to come from the whitelist
110    pub white_connect_percent: usize,
111    /// Number of goldlist connections
112    pub gold_connect_count: usize,
113    /// If this is true, strictly follow the gold_connect_count and
114    /// white_connect_percent settings. Otherwise, connect to greylist
115    /// entries if we have no white or gold connections.
116    pub slot_preference_strict: bool,
117    /// Number of seconds with no connections after which refinery
118    /// process is paused.
119    pub time_with_no_connections: u64,
120    /// Nodes to avoid interacting with for the duration of the program,
121    /// in the format ["host", ["scheme", "scheme"], [port, port]]
122    /// If scheme is left empty it will default to "tcp+tls".
123    /// If ports are left empty all ports from this peer will be blocked.
124    pub blacklist: Vec<BlacklistEntry>,
125    /// Do not ban nodes that send messages without dispatchers if set
126    /// to `Relaxed`. For most uses, should be set to `Strict`.
127    pub ban_policy: BanPolicy,
128}

161            slot_preference_strict: false,
162            time_with_no_connections: 30,
163            blacklist: vec![],
164            ban_policy: BanPolicy::Strict,
165        }
166    }

172/// Distinguishes distinct P2P networks
173#[derive(serde::Deserialize, Debug, Clone)]
174pub struct MagicBytes(pub [u8; 4]);

176impl Default for MagicBytes {
177    fn default() -> Self {
178        Self([0xd9, 0xef, 0xb6, 0x7d])
179    }

336impl From<SettingsOpt> for Settings {
337    fn from(opt: SettingsOpt) -> Self {
338        let def = Settings::default();
339
340        Self {
341            node_id: opt.node_id,
342            inbound_addrs: opt.inbound,
343            external_addrs: opt.external_addrs,
344            magic_bytes: opt.magic_bytes,
345            peers: opt.peers,
346            seeds: opt.seeds,
347            app_version: def.app_version,
348            allowed_transports: opt.allowed_transports.unwrap_or(def.allowed_transports),
349            mixed_transports: opt.mixed_transports.unwrap_or(def.mixed_transports),
350            tor_socks5_proxy: opt.tor_socks5_proxy,
351            nym_socks5_proxy: opt.nym_socks5_proxy,
352            i2p_socks5_proxy: opt.i2p_socks5_proxy.unwrap_or(def.i2p_socks5_proxy),
353            outbound_connections: opt.outbound_connections.unwrap_or(def.outbound_connections),
354            inbound_connections: opt.inbound_connections.unwrap_or(def.inbound_connections),
355            outbound_connect_timeout: opt
356                .outbound_connect_timeout
357                .unwrap_or(def.outbound_connect_timeout),
358            channel_handshake_timeout: opt
359                .channel_handshake_timeout
360                .unwrap_or(def.channel_handshake_timeout),
361            channel_heartbeat_interval: opt
362                .channel_heartbeat_interval
363                .unwrap_or(def.channel_heartbeat_interval),
364            localnet: opt.localnet,
365            outbound_peer_discovery_cooloff_time: opt
366                .outbound_peer_discovery_cooloff_time
367                .unwrap_or(def.outbound_peer_discovery_cooloff_time),
368            outbound_peer_discovery_attempt_time: opt
369                .outbound_peer_discovery_attempt_time
370                .unwrap_or(def.outbound_peer_discovery_attempt_time),
371            p2p_datastore: opt.p2p_datastore,
372            hostlist: opt.hostlist,
373            greylist_refinery_interval: opt
374                .greylist_refinery_interval
375                .unwrap_or(def.greylist_refinery_interval),
376            white_connect_percent: opt.white_connect_percent.unwrap_or(def.white_connect_percent),
377            gold_connect_count: opt.gold_connect_count.unwrap_or(def.gold_connect_count),
378            slot_preference_strict: opt.slot_preference_strict,
379            time_with_no_connections: opt
380                .time_with_no_connections
381                .unwrap_or(def.time_with_no_connections),
382            blacklist: opt.blacklist,
383            ban_policy: opt.ban_policy,
384        }
385    }
