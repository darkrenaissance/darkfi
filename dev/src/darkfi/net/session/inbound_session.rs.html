19//! Inbound connections session. Manages the creation of inbound sessions.
20//! Used to create an inbound session and start and stop the session.
21//!
22//! Class consists of 3 pointers: a weak pointer to the p2p parent class,
23//! an acceptor pointer, and a stoppable task pointer. Using a weak pointer
24//! to P2P allows us to avoid circular dependencies.
25
26use std::sync::{Arc, Weak};
27
28use async_trait::async_trait;
29use log::{debug, error, info, warn};
30use smol::{lock::Mutex, Executor};
31use url::Url;
32
33use super::{
34    super::{
35        acceptor::{Acceptor, AcceptorPtr},
36        channel::ChannelPtr,
37        dnet::{self, dnetev, DnetEvent},
38        p2p::{P2p, P2pPtr},
39    },
40    Session, SessionBitFlag, SESSION_INBOUND,
41};
42use crate::{
43    system::{StoppableTask, StoppableTaskPtr, Subscription},
44    Error, Result,
45};
46
47pub type InboundSessionPtr = Arc<InboundSession>;
48
49/// Defines inbound connections session
50pub struct InboundSession {
51    pub(in crate::net) p2p: Weak<P2p>,
52    acceptors: Mutex<Vec<AcceptorPtr>>,
53    accept_tasks: Mutex<Vec<StoppableTaskPtr>>,
54}
55
56impl InboundSession {
57    /// Create a new inbound session
58    pub fn new(p2p: Weak<P2p>) -> InboundSessionPtr {
59        Arc::new(Self {
60            p2p,
61            acceptors: Mutex::new(Vec::new()),
62            accept_tasks: Mutex::new(Vec::new()),
63        })
64    }
65
66    /// Starts the inbound session. Begins by accepting connections and fails
67    /// if the addresses are not configured. Then runs the channel subscription
68    /// loop.
69    pub async fn start(self: Arc<Self>) -> Result<()> {
70        let inbound_addrs = self.p2p().settings().read().await.inbound_addrs.clone();
71
72        if inbound_addrs.is_empty() {
73            info!(target: "net::inbound_session", "[P2P] Not configured for inbound connections.");
74            return Ok(())
75        }
76
77        let ex = self.p2p().executor();
78
79        // Activate mutex lock on accept tasks.
80        let mut accept_tasks = self.accept_tasks.lock().await;
81
82        for (index, accept_addr) in inbound_addrs.iter().enumerate() {
83            // First initialize an Acceptor and its Subscriber.
84            let parent = Arc::downgrade(&self);
85            let acceptor = Acceptor::new(parent);
86
87            // Now start the Subscriber. The Subscriber will return a Channel once it has been
88            // prepared by the Acceptor.
89            let channel_sub = acceptor.clone().subscribe().await;
90
91            // Then start listening for a Channel returned by the Subscriber. Call setup_channel()
92            // to register the Channel when it has been received.
93            let task = StoppableTask::new();
94            task.clone().start(
95                self.clone().channel_sub_loop(channel_sub, index, ex.clone()),
96                // Ignore stop handler
97                |_| async {},
98                Error::NetworkServiceStopped,
99                ex.clone(),
100            );
101
102            accept_tasks.push(task);
103
104            // Finally, run the Acceptor to start accepting inbound connections. Only when
105            // the Subscriber has been set up can we safely do this.
106            self.clone()
107                .start_accept_session(index, accept_addr.clone(), acceptor, ex.clone())
108                .await?;
109        }
110
111        Ok(())
112    }
113
114    /// Stops the inbound session.
115    pub async fn stop(&self) {
116        if self.p2p().settings().read().await.inbound_addrs.is_empty() {
117            info!(target: "net::inbound_session", "[P2P] Stopping inbound session.");
118            return
119        }
120
121        let acceptors = &*self.acceptors.lock().await;
122        for acceptor in acceptors {
123            acceptor.stop().await;
124        }
125
126        let accept_tasks = &*self.accept_tasks.lock().await;
127        for accept_task in accept_tasks {
128            accept_task.stop().await;
129        }
130    }
131
132    /// Start accepting connections for inbound session.
133    async fn start_accept_session(
134        self: Arc<Self>,
135        index: usize,
136        accept_addr: Url,
137        acceptor: AcceptorPtr,
138        ex: Arc<Executor<'_>>,
139    ) -> Result<()> {
140        info!(target: "net::inbound_session", "[P2P] Starting Inbound session #{index} on {accept_addr}");
141        // Start listener
142        let result = acceptor.clone().start(accept_addr, ex).await;
143        if let Err(e) = &result {
144            error!(target: "net::inbound_session", "[P2P] Error starting listener #{index}: {e}");
145            acceptor.stop().await;
146        } else {
147            self.acceptors.lock().await.push(acceptor);
148        }
149
150        result
151    }
152
153    /// Wait for all new channels created by the acceptor and call setup_channel() on them.
154    async fn channel_sub_loop(
155        self: Arc<Self>,
156        channel_sub: Subscription<Result<ChannelPtr>>,
157        index: usize,
158        ex: Arc<Executor<'_>>,
159    ) -> Result<()> {
160        loop {
161            let channel = channel_sub.receive().await?;
162
163            // Spawn a detached task to process the channel.
164            // This will just perform the channel setup then exit.
165            ex.spawn(self.clone().setup_channel(index, channel, ex.clone())).detach();
166        }
167    }
168
169    /// Registers the channel. First performs a network handshake and starts the channel.
170    /// Then starts sending keep-alive and address messages across the channel.
171    async fn setup_channel(
172        self: Arc<Self>,
173        index: usize,
174        channel: ChannelPtr,
175        ex: Arc<Executor<'_>>,
176    ) {
177        info!(
178             target: "net::inbound_session::setup_channel",
179             "[P2P] Connected Inbound #{index} [{}]", channel.display_address()
180        );
181
182        dnetev!(self, InboundConnected, {
183            addr: channel.info.connect_addr.clone(),
184            channel_id: channel.info.id,
185        });
186
187        let stop_sub = channel.subscribe_stop().await;
188
189        match self.register_channel(channel.clone(), ex.clone()).await {
190            Ok(()) => {
191                if let Ok(stop_sub) = stop_sub {
192                    // Wait for a stop signal, then cleanup.
193                    stop_sub.receive().await;
194
195                    debug!(
196                        target: "net::inbound_session::setup_channel()",
197                        "Received stop_sub, channel removed from P2P",
198                    );
199                }
200            }
201            Err(e) => {
202                warn!(
203                    target: "net::inbound_session::setup_channel()",
204                    "Channel setup failed! Err={e}"
205                );
206            }
207        }
208
209        dnetev!(self, InboundDisconnected, {
210            addr: channel.info.connect_addr.clone(),
211            channel_id: channel.info.id,
212        });
213    }
214}
215
216#[async_trait]
217impl Session for InboundSession {
218    fn p2p(&self) -> P2pPtr {
219        self.p2p.upgrade().unwrap()
220    }
221
222    fn type_id(&self) -> SessionBitFlag {
223        SESSION_INBOUND
224    }
225}
