19//! Outbound connections session. Manages the creation of outbound sessions.
20//! Used to create an outbound session and to stop and start the session.
21//!
22//! Class consists of a weak pointer to the p2p interface and a vector of
23//! outbound connection slots. Using a weak pointer to p2p allows us to
24//! avoid circular dependencies. The vector of slots is wrapped in a mutex
25//! lock. This is switched on every time we instantiate a connection slot
26//! and insures that no other part of the program uses the slots at the
27//! same time.
28

61/// Defines outbound connections session.
62pub struct OutboundSession {
63    /// Weak pointer to parent p2p object
64    pub(in crate::net) p2p: Weak<P2p>,
65    /// Outbound connection slots
66    slots: Mutex<Vec<Arc<Slot>>>,
67    /// Peer discovery task
68    peer_discovery: Arc<PeerDiscovery>,
69}
70
71impl OutboundSession {
72    /// Create a new outbound session.
73    pub(crate) fn new(p2p: Weak<P2p>) -> OutboundSessionPtr {
74        Arc::new_cyclic(|session| Self {
75            p2p,
76            slots: Mutex::new(Vec::new()),
77            peer_discovery: PeerDiscovery::new(session.clone()),
78        })
79    }
80
81    /// Start the outbound session. Runs the channel connect loop.
82    pub(crate) async fn start(self: Arc<Self>) {
83        let n_slots = self.p2p().settings().read().await.outbound_connections;
84        info!(target: "net::outbound_session", "[P2P] Starting {n_slots} outbound connection slots.");
85
86        // Activate mutex lock on connection slots.
87        let mut slots = self.slots.lock().await;
88
89        let mut futures = FuturesUnordered::new();
90
91        let self_ = Arc::downgrade(&self);
92
93        for i in 0..n_slots as u32 {
94            let slot = Slot::new(self_.clone(), i);
95            futures.push(slot.clone().start());
96            slots.push(slot);
97        }
98
99        while (futures.next().await).is_some() {}
100
101        self.peer_discovery.clone().start().await;
102    }
103
104    /// Stops the outbound session.
105    pub(crate) async fn stop(&self) {
106        debug!(target: "net::outbound_session", "Stopping outbound session..");
107        let slots = &*self.slots.lock().await;
108        let mut futures = FuturesUnordered::new();
109
110        for slot in slots {
111            futures.push(slot.clone().stop());
112        }
113
114        while (futures.next().await).is_some() {}
115
116        self.peer_discovery.clone().stop().await;
117        debug!(target: "net::outbound_session", "Outbound session stopped!");
118    }
119
120    pub async fn slot_info(&self) -> Vec<u32> {
121        let mut info = Vec::new();
122        let slots = &*self.slots.lock().await;
123        for slot in slots {
124            info.push(slot.channel_id.load(Ordering::Relaxed));
125        }
126        info
127    }
128
129    fn wakeup_peer_discovery(&self) {
130        self.peer_discovery.notify()
131    }
132
133    async fn wakeup_slots(&self) {
134        let slots = &*self.slots.lock().await;
135        for slot in slots {
136            slot.notify();
137        }
138    }
139}

152struct Slot {
153    slot: u32,
154    process: StoppableTaskPtr,
155    wakeup_self: CondVar,
156    session: Weak<OutboundSession>,
157    connector: Connector,
158    // For debugging
159    channel_id: AtomicU32,
160}
161
162impl Slot {
163    fn new(session: Weak<OutboundSession>, slot: u32) -> Arc<Self> {
164        let settings = session.upgrade().unwrap().p2p().settings();
165
166        Arc::new(Self {
167            slot,
168            process: StoppableTask::new(),
169            wakeup_self: CondVar::new(),
170            session: session.clone(),
171            connector: Connector::new(settings, session),
172            channel_id: AtomicU32::new(0),
173        })
174    }
175
176    async fn start(self: Arc<Self>) {
177        let ex = self.p2p().executor();
178
179        self.process.clone().start(
180            self.run(),
181            |res| async {
182                match res {
183                    Ok(()) | Err(Error::NetworkServiceStopped) => {}
184                    Err(e) => error!("net::outbound_session {e}"),
185                }
186            },
187            Error::NetworkServiceStopped,
188            ex,
189        );
190    }
191
192    async fn stop(self: Arc<Self>) {
193        self.connector.stop();
194        self.process.stop().await;
195    }
196
197    /// Address selection algorithm that works as follows: up to
198    /// gold_count, select from the goldlist. Up to white_count,
199    /// select from the whitelist. For all other slots, select from
200    /// the greylist. If none of these preferences are satisfied, do
201    /// peer discovery.
202    ///
203    /// Selecting from the greylist for some % of the slots is necessary
204    /// and healthy since we require the network retains some unreliable
205    /// connections. A network that purely favors uptime over unreliable
206    /// connections may be vulnerable to sybil by attackers with good uptime.
207    async fn fetch_addrs(&self) -> Option<(Url, u64)> {
208        let hosts = self.p2p().hosts();
209        let slot = self.slot as usize;
210        let container = &self.p2p().hosts().container;
211
212        // Acquire Settings read lock
213        let settings = self.p2p().settings().read_arc().await;
214
215        let white_count = (settings.white_connect_percent * settings.outbound_connections) / 100;
216        let gold_count = settings.gold_connect_count;
217
218        let transports = settings.allowed_transports.clone();
219        let preference_strict = settings.slot_preference_strict;
220
221        // Drop Settings read lock
222        drop(settings);
223
224        let grey_only = hosts.container.is_empty(HostColor::White) &&
225            hosts.container.is_empty(HostColor::Gold) &&
226            !hosts.container.is_empty(HostColor::Grey);
227
228        // If we only have grey entries, select from the greylist. Otherwise,
229        // use the preference defined in settings.
230        let addrs = if grey_only && !preference_strict {
231            container.fetch_with_schemes(HostColor::Grey as usize, &transports, None)
232        } else if slot < gold_count {
233            container.fetch_with_schemes(HostColor::Gold as usize, &transports, None)
234        } else if slot < white_count {
235            container.fetch_with_schemes(HostColor::White as usize, &transports, None)
236        } else {
237            container.fetch_with_schemes(HostColor::Grey as usize, &transports, None)
238        };
239
240        hosts.check_addrs(addrs).await
241    }
242
243    // We first try to make connections to the addresses on our gold list. We then find some
244    // whitelist connections according to the whitelist percent default. Finally, any remaining
245    // connections we make from the greylist.
246    async fn run(self: Arc<Self>) -> Result<()> {
247        let hosts = self.p2p().hosts();
248
249        loop {
250            // Activate the slot
251            debug!(
252                target: "net::outbound_session::try_connect()",
253                "[P2P] Finding a host to connect to for outbound slot #{}",
254                self.slot,
255            );
256
257            // Do peer discovery if we don't have any peers on the Grey, White or Gold list
258            // (first time connecting to the network).
259            if hosts.container.is_empty(HostColor::Grey) &&
260                hosts.container.is_empty(HostColor::White) &&
261                hosts.container.is_empty(HostColor::Gold)
262            {
263                dnetev!(self, OutboundSlotSleeping, {
264                    slot: self.slot,
265                });
266
267                self.wakeup_self.reset();
268                // Peer discovery
269                self.session().wakeup_peer_discovery();
270                // Wait to be woken up by peer discovery
271                self.wakeup_self.wait().await;
272
273                continue
274            }
275
276            let addr = if let Some(addr) = self.fetch_addrs().await {
277                debug!(target: "net::outbound_session::run()", "Fetched addr={}, slot #{}", addr.0,
278                self.slot);
279                addr
280            } else {
281                debug!(target: "net::outbound_session::run()", "No address found! Activating peer discovery...");
282                dnetev!(self, OutboundSlotSleeping, {
283                    slot: self.slot,
284                });
285
286                self.wakeup_self.reset();
287                // Peer discovery
288                self.session().wakeup_peer_discovery();
289                // Wait to be woken up by peer discovery
290                self.wakeup_self.wait().await;
291
292                continue
293            };
294
295            let host = addr.0;
296            let last_seen = addr.1;
297            let slot = self.slot;
298
299            info!(
300                target: "net::outbound_session::try_connect()",
301                "[P2P] Connecting outbound slot #{slot} [{host}]"
302            );
303
304            dnetev!(self, OutboundSlotConnecting, {
305                slot,
306                addr: host.clone(),
307            });
308
309            let (_, channel) = match self.try_connect(host.clone(), last_seen).await {
310                Ok(connect_info) => connect_info,
311                Err(err) => {
312                    debug!(
313                        target: "net::outbound_session::try_connect()",
314                        "[P2P] Outbound slot #{slot} connection failed: {err}"
315                    );
316
317                    dnetev!(self, OutboundSlotDisconnected, {
318                        slot,
319                        err: err.to_string()
320                    });
321
322                    self.channel_id.store(0, Ordering::Relaxed);
323
324                    continue
325                }
326            };
327
328            // At this point we've managed to connect.
329            let stop_sub = channel.subscribe_stop().await?;
330
331            info!(
332                target: "net::outbound_session::try_connect()",
333                "[P2P] Outbound slot #{slot} connected [{}]",
334                channel.display_address()
335            );
336
337            dnetev!(self, OutboundSlotConnected, {
338                slot: self.slot,
339                addr: channel.display_address().clone(),
340                channel_id: channel.info.id
341            });
342
343            // Setup new channel
344            if let Err(err) =
345                self.session().register_channel(channel.clone(), self.p2p().executor()).await
346            {
347                info!(
348                    target: "net::outbound_session",
349                    "[P2P] Outbound slot #{slot} disconnected: {err}"
350                );
351
352                dnetev!(self, OutboundSlotDisconnected, {
353                    slot: self.slot,
354                    err: err.to_string()
355                });
356
357                self.channel_id.store(0, Ordering::Relaxed);
358
359                warn!(
360                    target: "net::outbound_session::try_connect()",
361                    "[P2P] Suspending addr=[{}] slot #{slot}",
362                    channel.display_address()
363                );
364
365                // Peer disconnected during the registry process. We'll downgrade this peer now.
366                if let Err(e) = self
367                    .p2p()
368                    .hosts()
369                    .move_host(channel.address(), last_seen, HostColor::Grey)
370                    .await
371                {
372                    warn!(target: "net::outbound_session", "Error while moving addr={} to greylist: {e}", channel.display_address());
373                    continue
374                }
375
376                // Mark its state as Suspend, which sends this node to the Refinery for processing.
377                if let Err(e) =
378                    self.p2p().hosts().try_register(channel.address().clone(), HostState::Suspend)
379                {
380                    warn!(target: "net::outbound_session", "Error while suspending addr={}: {e}", channel.display_address());
381                }
382
383                continue
384            }
385
386            self.channel_id.store(channel.info.id, Ordering::Relaxed);
387
388            // Wait for channel to close
389            stop_sub.receive().await;
390
391            self.channel_id.store(0, Ordering::Relaxed);
392        }
393    }
394
395    /// Start making an outbound connection, using provided [`Connector`].
396    /// Tries to find a valid address to connect to, otherwise does peer
397    /// discovery. The peer discovery loops until some peer we can connect
398    /// to is found. Once connected, registers the channel, removes it from
399    /// the list of pending channels, and starts sending messages across the
400    /// channel. In case of any failures, a network error is returned and the
401    /// main connect loop (parent of this function) will iterate again.
402    async fn try_connect(&self, addr: Url, last_seen: u64) -> Result<(Url, ChannelPtr)> {
403        match self.connector.connect(&addr).await {
404            Ok((addr_final, channel)) => Ok((addr_final, channel)),
405
406            Err(err) => {
407                info!(
408                    target: "net::outbound_session::try_connect()",
409                    "[P2P] Unable to connect outbound slot #{} {err}",
410                    self.slot
411                );
412
413                // Immediately return if the Connector has stopped.
414                // This indicates a shutdown of the P2P network and
415                // should not result in hostlist modifications.
416                if let Error::ConnectorStopped(message) = err {
417                    return Err(Error::ConnectFailed(message));
418                }
419
420                // At this point we failed to connect. We'll downgrade this peer now.
421                self.p2p().hosts().move_host(&addr, last_seen, HostColor::Grey).await?;
422
423                // Mark its state as Suspend, which sends it to the Refinery for processing.
424                if let Err(e) = self.p2p().hosts().try_register(addr.clone(), HostState::Suspend) {
425                    warn!(target: "net::outbound_session::try_connect()", "Error while suspending addr={addr}: {e}");
426                }
427
428                // Notify that channel processing failed
429                self.p2p().hosts().channel_publisher.notify(Err(err.clone())).await;
430
431                Err(err)
432            }
433        }
434    }
435
436    fn notify(&self) {
437        self.wakeup_self.notify()
438    }

440    fn session(&self) -> OutboundSessionPtr {
441        self.session.upgrade().unwrap()
442    }
443    fn p2p(&self) -> P2pPtr {
444        self.session().p2p()
445    }

464    fn notify(&self);

471/// Main PeerDiscovery process that loops through connected peers
472/// and sends out a `GetAddrs` when it is active. If there are no
473/// connected peers after two attempts, connect to our seed nodes
474/// and perform `SeedSyncSession`.
475struct PeerDiscovery {
476    process: StoppableTaskPtr,
477    wakeup_self: CondVar,
478    session: Weak<OutboundSession>,
479}
480
481impl PeerDiscovery {
482    fn new(session: Weak<OutboundSession>) -> Arc<Self> {
483        Arc::new(Self { process: StoppableTask::new(), wakeup_self: CondVar::new(), session })
484    }
485}
486
487#[async_trait]
488impl PeerDiscoveryBase for PeerDiscovery {
489    async fn start(self: Arc<Self>) {
490        let ex = self.p2p().executor();
491        self.process.clone().start(
492            async move {
493                self.run().await;
494                unreachable!();
495            },
496            // Ignore stop handler
497            |_| async {},
498            Error::NetworkServiceStopped,
499            ex,
500        );
501    }
502    async fn stop(self: Arc<Self>) {
503        self.process.stop().await;
504    }
505
506    /// Activate peer discovery if not active already. For the first two
507    /// attempts, this will loop through all connected P2P peers and send
508    /// out a `GetAddrs` message to request more peers. Other parts of the
509    /// P2P stack will then handle the incoming addresses and place them in
510    /// the hosts list.  
511    ///
512    /// On the third attempt, and if we still haven't made any connections,
513    /// this function will then call `p2p.seed()` which triggers a
514    /// `SeedSyncSession` that will connect to configured seeds and request
515    /// peers from them.
516    ///
517    /// This function will also sleep `outbound_peer_discovery_attempt_time`
518    /// seconds after broadcasting in order to let the P2P stack receive and
519    /// work through the addresses it is expecting.
520    async fn run(self: Arc<Self>) {
521        let mut current_attempt = 0;
522        loop {
523            dnetev!(self, OutboundPeerDiscovery, {
524                attempt: current_attempt,
525                state: "wait",
526            });
527
528            // wait to be woken up by notify()
529            let sleep_was_instant = self.wait().await;
530
531            // Read the current P2P settings
532            let settings = self.p2p().settings().read_arc().await;
533            let outbound_peer_discovery_cooloff_time =
534                settings.outbound_peer_discovery_cooloff_time;
535            let outbound_peer_discovery_attempt_time =
536                settings.outbound_peer_discovery_attempt_time;
537            let outbound_connections = settings.outbound_connections;
538            let getaddrs_max = settings.getaddrs_max;
539            let allowed_transports = settings.allowed_transports.clone();
540            let seeds = settings.seeds.clone();
541            drop(settings);
542
543            if sleep_was_instant {
544                // Try again
545                current_attempt += 1;
546            } else {
547                // reset back to start
548                current_attempt = 1;
549            }
550
551            if current_attempt >= 4 {
552                info!(
553                    target: "net::outbound_session::peer_discovery()",
554                    "[P2P] [PEER DISCOVERY] Sleeping and trying again. Attempt {current_attempt}"
555                );
556
557                dnetev!(self, OutboundPeerDiscovery, {
558                    attempt: current_attempt,
559                    state: "sleep",
560                });
561
562                sleep(outbound_peer_discovery_cooloff_time).await;
563                current_attempt = 1;
564            }
565
566            // First 2 times try sending GetAddr to the network.
567            // 3rd time do a seed sync (providing we have seeds
568            // configured).
569            if self.p2p().is_connected() && current_attempt <= 2 {
570                // Broadcast the GetAddrs message to all active peers.
571                // If we have no active peers, we will perform a SeedSyncSession instead.
572                info!(
573                    target: "net::outbound_session::peer_discovery()",
574                    "[P2P] [PEER DISCOVERY] Asking peers for new peers to connect to...");
575
576                dnetev!(self, OutboundPeerDiscovery, {
577                    attempt: current_attempt,
578                    state: "getaddr",
579                });
580
581                let get_addrs = GetAddrsMessage {
582                    max: getaddrs_max.unwrap_or(outbound_connections.min(u32::MAX as usize) as u32),
583                    transports: allowed_transports,
584                };
585
586                self.p2p().broadcast(&get_addrs).await;
587
588                // Wait for a hosts store update event
589                let store_sub = self.p2p().hosts().subscribe_store().await;
590
591                let result = timeout(
592                    Duration::from_secs(outbound_peer_discovery_attempt_time),
593                    store_sub.receive(),
594                )
595                .await;
596
597                match result {
598                    Ok(addrs_len) => {
599                        info!(
600                            target: "net::outbound_session::peer_discovery()",
601                            "[P2P] [PEER DISCOVERY] Discovered {addrs_len} peers"
602                        );
603                    }
604                    Err(_) => {
605                        warn!(
606                            target: "net::outbound_session::peer_discovery()",
607                            "[P2P] [PEER DISCOVERY] Waiting for addrs timed out."
608                        );
609                        // Just do seed next time
610                        current_attempt = 3;
611                    }
612                }
613
614                // NOTE: not every call to subscribe() in net/ has a
615                // corresponding unsubscribe(). To do this we need async
616                // Drop. For now it's sufficient for publishers to be
617                // de-allocated when the Session completes.
618                store_sub.unsubscribe().await;
619            } else if !seeds.is_empty() {
620                info!(
621                    target: "net::outbound_session::peer_discovery()",
622                    "[P2P] [PEER DISCOVERY] Asking seeds for new peers to connect to...");
623
624                dnetev!(self, OutboundPeerDiscovery, {
625                    attempt: current_attempt,
626                    state: "seed",
627                });
628
629                self.p2p().seed().await;
630            }
631
632            self.wakeup_self.reset();
633            self.session().wakeup_slots().await;
634
635            // Give some time for new connections to be established
636            sleep(outbound_peer_discovery_attempt_time).await;
637        }
638    }
639
640    /// Blocks execution until we receive a notification from notify().
641    /// `wakeup_self.wait()` resets the condition variable (`CondVar`) and waits
642    /// for a call from `notify()`. Returns `true` if the function completed
643    /// instantly (i.e. no wait occured). Returns false otherwise.
644    async fn wait(&self) -> bool {
645        let wakeup_start = Instant::now();
646        self.wakeup_self.wait().await;
647        let wakeup_end = Instant::now();
648
649        let epsilon = Duration::from_millis(200);
650        wakeup_end - wakeup_start <= epsilon
651    }
652
653    /// Wakeup peer discovery by sending a notification to `wakeup_self`.
654    /// Uses the underlying `CondVar` method `notify()`. Subsequent calls
655    /// to this do nothing until `wait()` is called.
656    fn notify(&self) {
657        self.wakeup_self.notify()
658    }
659
660    fn session(&self) -> OutboundSessionPtr {
661        self.session.upgrade().unwrap()
662    }
663
664    fn p2p(&self) -> P2pPtr {
665        self.session().p2p()
666    }
