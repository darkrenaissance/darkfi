19//! Direct connections session. Manages the creation of direct sessions.
20//! Used to create a direct session and to stop and start the session.
21//!
22//! A direct session is a type of outbound session in which a protocol can
23//! open a temporary channel (stopped after used) to a peer. Direct sessions
24//! do not loop continually, once stopped the session will not try to reopen
25//! a connection.
26//!
27//! [`ChannelBuilder`] is used to create new direct connections.
28//!
29//! If there is no slots in the outbound session, the direct session can
30//! optionnally handle peer discovery.
31

61/// Defines direct connections session.
62pub struct DirectSession {
63    /// Weak pointer to parent p2p object
64    pub(in crate::net) p2p: Weak<P2p>,
65    /// Service to create direct channels
66    channel_builder: Arc<AsyncMutex<ChannelBuilder>>,
67    /// Tasks that are trying to create a direct channel (they retry until they succeed).
68    /// A task is removed once the channel is successfully created.
69    retries_tasks: Arc<AsyncMutex<HashMap<Url, Arc<StoppableTask>>>>,
70    /// Peer discovery task
71    peer_discovery: Arc<PeerDiscovery>,
72    /// Channel ID -> usage count
73    channels_usage: Arc<AsyncMutex<HashMap<u32, u32>>>,
74}
75
76impl DirectSession {
77    /// Create a new direct session.
78    pub fn new(p2p: Weak<P2p>) -> DirectSessionPtr {
79        Arc::new_cyclic(|session| Self {
80            p2p,
81            channel_builder: Arc::new(AsyncMutex::new(ChannelBuilder::new(session.clone()))),
82            retries_tasks: Arc::new(AsyncMutex::new(HashMap::new())),
83            peer_discovery: PeerDiscovery::new(session.clone()),
84            channels_usage: Arc::new(AsyncMutex::new(HashMap::new())),
85        })
86    }
87
88    /// Start the direct session.
89    pub(crate) async fn start(self: Arc<Self>) {
90        self.peer_discovery.clone().start().await;
91    }
92
93    /// Stops the direct session.
94    pub async fn stop(&self) {
95        self.peer_discovery.clone().stop().await;
96
97        for (_, task) in self.retries_tasks.lock().await.iter() {
98            task.stop().await;
99        }
100    }
101
102    /// Notify the peer discovery task to start it.
103    /// The direct session's peer discovery process will not start until this
104    /// method is called.
105    /// If there are outbound slots, peer discovery does not start even if this
106    /// method is called, we let the outbound session take care of it.
107    pub fn start_peer_discovery(&self) {
108        self.peer_discovery.notify();
109    }

111    /// If there is an existing channel to the same address, this method will
112    /// return it (even if the channel was not created by the direct session).
113    /// Otherwise it will create a new channel to `addr` in the direct session.
114    pub async fn get_channel(&self, addr: &Url) -> Result<ChannelPtr> {
115        let channels = self.p2p().hosts().channels();
116        let channel = channels.iter().find(|&chan| chan.address() == addr).cloned();
117        if let Some(channel) = channel {
118            let mut channels_usage = self.channels_usage.lock().await;
119            let usage_count = channels_usage.get_mut(&channel.info.id);
120            if let Some(count) = usage_count {
121                *count += 1;
122            }
123            return Ok(channel)
124        }
125        self.channel_builder.lock().await.new_channel(addr).await
126    }
127
128    /// Try to create a new channel until it succeeds, then notify `channel_pub`.
129    /// If it fails to create a channel, a task will sleep
130    /// `outbound_connect_timeout` seconds and try again.
131    pub async fn get_channel_with_retries(&self, addr: Url, channel_pub: PublisherPtr<ChannelPtr>) {
132        let channel_builder = self.channel_builder.clone();
133        let task = StoppableTask::new();
134        let retries_tasks_lock = self.retries_tasks.clone();
135        let mut retries_tasks = self.retries_tasks.lock().await;
136        let channels_usage = self.channels_usage.clone();
137        let p2p = self.p2p().clone();
138        retries_tasks.insert(addr.clone(), task.clone());
139        drop(retries_tasks);
140
141        task.clone().start(
142            async move {
143                loop {
144                    // Check if there is already a channel to this addr
145                    let channels = p2p.hosts().channels();
146                    let channel = channels.iter().find(|&chan| chan.address() == &addr).cloned();
147                    if let Some(channel) = channel {
148                        let mut chan_usage = channels_usage.lock().await;
149                        let usage_count = chan_usage.get_mut(&channel.info.id);
150                        if let Some(count) = usage_count {
151                            *count += 1;
152                        }
153                        channel_pub.notify(channel).await;
154                        let mut retries_tasks = retries_tasks_lock.lock().await;
155                        retries_tasks.remove(&addr);
156                        break
157                    }
158
159                    // Try to create a new channel
160                    let mut builder = channel_builder.lock().await;
161                    let res = builder.new_channel(&addr).await;
162                    match res {
163                        Ok(channel) => {
164                            channel_pub.notify(channel).await;
165                            let mut retries_tasks = retries_tasks_lock.lock().await;
166                            retries_tasks.remove(&addr);
167                            break
168                        }
169                        Err(Error::HostDoesNotExist) => break,
170                        Err(_) => {
171                            drop(builder);
172                            let settings = p2p.settings().read_arc().await;
173                            sleep(settings.outbound_connect_timeout).await;
174                        }
175                    }
176                }
177
178                Ok(())
179            },
180            |res| async {
181                match res {
182                    Ok(()) | Err(Error::DetachedTaskStopped) => { /* Do nothing */ }
183                    Err(e) => {
184                        error!(target: "net::direct_session::create_channel_with_retries()", "{e}")
185                    }
186                }
187            },
188            Error::DetachedTaskStopped,
189            self.p2p().executor(),
190        );
191    }
192
193    /// Close a direct channel if it's not used by anything.
194    /// `AsyncDrop` would be great here (<https://doc.rust-lang.org/std/future/trait.AsyncDrop.html>)
195    /// but it's still in nightly. For now you must call this method manually
196    /// once you are done with a direct channel.
197    /// Returns `true` if the channel is stopped.
198    pub async fn cleanup_channel(&self, channel: ChannelPtr) -> bool {
199        if channel.session_type_id() & SESSION_DIRECT == 0 {
200            // Do nothing if this is not a channel created by the direct session
201            return false
202        }
203
204        let mut channels_usage = self.channels_usage.lock().await;
205        let usage_count = channels_usage.get_mut(&channel.info.id);
206        if usage_count.is_none() {
207            channel.stop().await;
208            return true
209        }
210        let usage_count = usage_count.unwrap();
211        if *usage_count > 0 {
212            *usage_count -= 1;
213        }
214
215        if *usage_count == 0 {
216            channels_usage.remove(&channel.info.id);
217            channel.stop().await;
218            return true
219        }
220
221        false
222    }
223}

268    /// Create a new channel to `addr` in the direct session.
269    /// The transport is verified before the connection is started.
270    pub async fn new_channel(&mut self, addr: &Url) -> Result<ChannelPtr> {
271        info!(
272            target: "net::direct_session",
273            "[P2P] Connecting to direct outbound [{addr}]",
274        );
275
276        let settings = self.session().p2p().settings().read_arc().await;
277        let seeds = settings.seeds.clone();
278        let allowed_transports = settings.allowed_transports.clone();
279        drop(settings);
280
281        // Do not establish a connection to a host that is also configured as a seed.
282        // This indicates a user misconfiguration.
283        if seeds.contains(addr) {
284            error!(
285                target: "net::direct_session",
286                "[P2P] Suspending direct connection to seed [{}]", addr.clone(),
287            );
288            return Err(Error::ConnectFailed(format!("[{addr}]: Direct connection to seed")))
289        }
290
291        // Abort if we are trying to connect to our own external address.
292        let hosts = self.session().p2p().hosts();
293        let external_addrs = hosts.external_addrs().await;
294        if external_addrs.contains(addr) {
295            warn!(
296                target: "net::hosts::check_addrs",
297                "[P2P] Suspending direct connection to external addr [{}]", addr.clone(),
298            );
299            return Err(Error::ConnectFailed(format!(
300                "[{addr}]: Direct connection to external addr"
301            )))
302        }
303
304        // Abort if we do not support this transport.
305        if !allowed_transports.contains(&addr.scheme().to_string()) {
306            return Err(Error::UnsupportedTransport(addr.scheme().to_string()))
307        }
308
309        // Abort if this peer is IPv6 and we do not support it.
310        if !hosts.ipv6_available.load(Ordering::SeqCst) && hosts.is_ipv6(addr) {
311            return Err(Error::ConnectFailed(format!("[{addr}]: IPv6 is unavailable")))
312        }
313
314        if let Err(e) = hosts.try_register(addr.clone(), HostState::Connect) {
315            debug!(target: "net::direct_session",
316                "[P2P] Cannot connect to direct={addr}, err={e}");
317            return Err(e)
318        }
319
320        dnetev!(self, DirectConnecting, {
321            connect_addr: addr.clone(),
322        });
323
324        match self.connector().connect(addr).await {
325            Ok((_, channel)) => {
326                info!(
327                    target: "net::direct_session",
328                    "[P2P] Direct outbound connected [{}]",
329                    channel.display_address()
330                );
331
332                dnetev!(self, DirectConnected, {
333                    connect_addr: channel.info.connect_addr.clone(),
334                    addr: channel.display_address().clone(),
335                    channel_id: channel.info.id
336                });
337
338                // Register the new channel
339                match self
340                    .session()
341                    .register_channel(channel.clone(), self.session().p2p().executor())
342                    .await
343                {
344                    Ok(()) => {
345                        self.session().channels_usage.lock().await.insert(channel.info.id, 1);
346                        Ok(channel)
347                    }
348                    Err(e) => {
349                        warn!(
350                            target: "net::direct_session",
351                            "[P2P] Unable to connect to direct outbound [{}]: {e}",
352                            channel.display_address(),
353                        );
354
355                        dnetev!(self, DirectDisconnected, {
356                            connect_addr: channel.info.connect_addr.clone(),
357                            err: e.to_string()
358                        });
359
360                        // Free up this addr for future operations.
361                        if let Err(e) = self.session().p2p().hosts().unregister(channel.address()) {
362                            warn!(target: "net::direct_session", "[P2P] Error while unregistering addr={}, err={e}", channel.display_address());
363                        }
364
365                        Err(e)
366                    }
367                }
368            }
369            Err(e) => {
370                warn!(
371                    target: "net::direct_session",
372                    "[P2P] Unable to connect to direct outbound: {e}",
373                );
374
375                dnetev!(self, DirectDisconnected, {
376                    connect_addr: addr.clone(),
377                    err: e.to_string()
378                });
379
380                // Free up this addr for future operations.
381                if let Err(e) = self.session().p2p().hosts().unregister(addr) {
382                    warn!(target: "net::direct_session", "[P2P] Error while unregistering addr={addr}, err={e}");
383                }
384
385                Err(e)
386            }
387        }
388    }

391/// PeerDiscovery process for that sends `GetAddrs` messages to a random
392/// whitelist or greylist host (creating a channel in the direct session).
393/// If it's unsuccessful after two attempts, connect to our seed nodes and
394/// perform `SeedSyncSession`.
395struct PeerDiscovery {
396    process: StoppableTaskPtr,
397    init: CondVar,
398    session: Weak<DirectSession>,
399}
400
401impl PeerDiscovery {
402    fn new(session: Weak<DirectSession>) -> Arc<Self> {
403        Arc::new(Self { process: StoppableTask::new(), init: CondVar::new(), session })
404    }
405}
406
407impl PeerDiscovery {
408    async fn start(self: Arc<Self>) {
409        let ex = self.p2p().executor();
410        self.process.clone().start(
411            async move {
412                self.run().await;
413                Ok(())
414            },
415            // Ignore stop handler
416            |_| async {},
417            Error::NetworkServiceStopped,
418            ex,
419        );
420    }
421    async fn stop(self: Arc<Self>) {
422        self.process.stop().await;
423    }
424
425    /// Peer discovery's main process. For the first two attempts, this will
426    /// broadcast a `GetAddrs` message to request more peers. If we are not
427    /// connected to any peer, we try to create a channel in the direct session
428    /// to a random whitelist or greylist host.
429    /// Other parts of the P2P stack will then handle the incoming addresses
430    /// and place them in the hosts list.
431    ///
432    /// On the third attempt, and if we still haven't made any connections,
433    /// this function will then call `p2p.seed()` which triggers a
434    /// `SeedSyncSession` that will connect to configured seeds and request
435    /// peers from them.
436    ///
437    /// This function will also sleep `outbound_peer_discovery_attempt_time`
438    /// seconds after broadcasting in order to let the P2P stack receive and
439    /// work through the addresses it is expecting.
440    ///
441    /// Peer discovery will only start once `notify()` is called.
442    async fn run(self: Arc<Self>) {
443        // DirectSession can handle peer discovery only if there is no outbound
444        // slot. Otherwise we let the outbound session take care of it.
445        let settings = self.p2p().settings().read_arc().await;
446        if settings.outbound_connections > 0 {
447            return
448        }
449
450        // Wait for the peer discovery to be notified
451        self.init.wait().await;
452
453        let mut current_attempt = 0;
454        loop {
455            dnetev!(self, DirectPeerDiscovery, {
456                attempt: current_attempt,
457                state: "wait",
458            });
459
460            // Read the current P2P settings
461            let settings = self.p2p().settings().read_arc().await;
462            let outbound_peer_discovery_cooloff_time =
463                settings.outbound_peer_discovery_cooloff_time;
464            let outbound_peer_discovery_attempt_time =
465                settings.outbound_peer_discovery_attempt_time;
466            let getaddrs_max = settings.getaddrs_max;
467            let allowed_transports = settings.allowed_transports.clone();
468            let seeds = settings.seeds.clone();
469            drop(settings);
470
471            current_attempt += 1;
472
473            if current_attempt >= 4 {
474                info!(
475                    target: "net::direct_session::peer_discovery()",
476                    "[P2P] [PEER DISCOVERY] Sleeping and trying again. Attempt {current_attempt}"
477                );
478
479                dnetev!(self, DirectPeerDiscovery, {
480                    attempt: current_attempt,
481                    state: "sleep",
482                });
483
484                sleep(outbound_peer_discovery_cooloff_time).await;
485                current_attempt = 1;
486            }
487
488            // If we are not connected to any peer, try to create a channel
489            // (using the direct session) to a random host from the goldlist,
490            // whitelist, or greylist.
491            let mut channel = None;
492            if !self.p2p().is_connected() {
493                dnetev!(self, DirectPeerDiscovery, {
494                    attempt: current_attempt,
495                    state: "newchan",
496                });
497
498                for color in [HostColor::Gold, HostColor::White, HostColor::Grey].iter() {
499                    if let Some((entry, _)) = self
500                        .p2p()
501                        .hosts()
502                        .container
503                        .fetch_random_with_schemes(color.clone(), &allowed_transports)
504                    {
505                        channel = self.p2p().session_direct().get_channel(&entry.0).await.ok();
506                        break;
507                    }
508                }
509            }
510
511            // First 2 times try sending GetAddr to the network.
512            // 3rd time do a seed sync (providing we have seeds configured).
513            if self.p2p().is_connected() && current_attempt <= 2 {
514                // Broadcast the GetAddrs message to all active peers.
515                // If we have no active peers, we will perform a SeedSyncSession instead.
516                info!(
517                    target: "net::direct_session::peer_discovery()",
518                    "[P2P] [PEER DISCOVERY] Asking peers for new peers to connect to...");
519
520                dnetev!(self, DirectPeerDiscovery, {
521                    attempt: current_attempt,
522                    state: "getaddr",
523                });
524
525                let get_addrs = GetAddrsMessage {
526                    max: getaddrs_max.unwrap_or(1),
527                    transports: allowed_transports,
528                };
529
530                self.p2p().broadcast(&get_addrs).await;
531
532                // Wait for a hosts store update event
533                let store_sub = self.p2p().hosts().subscribe_store().await;
534
535                let result = timeout(
536                    Duration::from_secs(outbound_peer_discovery_attempt_time),
537                    store_sub.receive(),
538                )
539                .await;
540
541                match result {
542                    Ok(addrs_len) => {
543                        info!(
544                            target: "net::direct_session::peer_discovery()",
545                            "[P2P] [PEER DISCOVERY] Discovered {addrs_len} peers"
546                        );
547                        // Found some addrs, reset `current_attempt`
548                        if addrs_len > 0 {
549                            current_attempt = 0;
550                        }
551                    }
552                    Err(_) => {
553                        warn!(
554                            target: "net::direct_session::peer_discovery()",
555                            "[P2P] [PEER DISCOVERY] Waiting for addrs timed out."
556                        );
557                        // Just do seed next time
558                        current_attempt = 3;
559                    }
560                }
561
562                // NOTE: not every call to subscribe() in net/ has a
563                // corresponding unsubscribe(). To do this we need async
564                // Drop. For now it's sufficient for publishers to be
565                // de-allocated when the Session completes.
566                store_sub.unsubscribe().await;
567            } else if !seeds.is_empty() {
568                info!(
569                    target: "net::direct_session::peer_discovery()",
570                    "[P2P] [PEER DISCOVERY] Asking seeds for new peers to connect to...");
571
572                dnetev!(self, DirectPeerDiscovery, {
573                    attempt: current_attempt,
574                    state: "seed",
575                });
576
577                self.p2p().seed().await;
578            }
579
580            // Stop the channel we created for peer discovery
581            if let Some(ch) = channel {
582                ch.stop().await;
583            }
584
585            // Give some time for new connections to be established
586            sleep(outbound_peer_discovery_attempt_time).await;
587        }
588    }
589
590    /// Init peer discovery by sending a notification to `init`.
591    /// Uses the underlying `CondVar` method `notify()`.
592    pub fn notify(&self) {
593        self.init.notify()
594    }
595
596    fn session(&self) -> DirectSessionPtr {
597        self.session.upgrade().unwrap()
598    }
599
600    fn p2p(&self) -> P2pPtr {
601        self.session().p2p()
602    }
