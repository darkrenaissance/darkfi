50/// Bitwise selectors for the `protocol_registry`
51pub type SessionBitFlag = u32;
52pub const SESSION_INBOUND: SessionBitFlag = 0b000001;
53pub const SESSION_OUTBOUND: SessionBitFlag = 0b000010;
54pub const SESSION_MANUAL: SessionBitFlag = 0b000100;
55pub const SESSION_SEED: SessionBitFlag = 0b001000;
56pub const SESSION_REFINE: SessionBitFlag = 0b010000;
57pub const SESSION_DIRECT: SessionBitFlag = 0b100000;
58
59pub const SESSION_DEFAULT: SessionBitFlag = 0b100111;
60pub const SESSION_ALL: SessionBitFlag = 0b111111;

64/// Removes channel from the list of connected channels when a stop signal
65/// is received.
66pub async fn remove_sub_on_stop(
67    p2p: P2pPtr,
68    channel: ChannelPtr,
69    type_id: SessionBitFlag,
70    stop_sub: Subscription<Error>,
71) {
72    debug!(target: "net::session::remove_sub_on_stop()", "[START]");
73    let hosts = p2p.hosts();
74    let addr = channel.address();
75
76    stop_sub.receive().await;
77
78    debug!(
79        target: "net::session::remove_sub_on_stop()",
80        "Received stop event. Removing channel {}",
81        channel.display_address()
82    );
83
84    // Downgrade to greylist if this is a outbound session.
85    if type_id & (SESSION_OUTBOUND | SESSION_DIRECT) != 0 {
86        debug!(
87            target: "net::session::remove_sub_on_stop()",
88            "Downgrading {}",
89            channel.display_address()
90        );
91
92        // If the host we are downgrading has been moved to blacklist,
93        // fetch_last_seen(addr) can return None. We simply print an
94        // error in this case.
95        match hosts.fetch_last_seen(addr) {
96            Some(last_seen) => {
97                if let Err(e) = hosts.move_host(addr, last_seen, HostColor::Grey).await {
98                    error!(target: "net::session::remove_sub_on_stop()",
99            "Failed to move host {} to Greylist! Err={e}", channel.display_address());
100                }
101            }
102            None => {
103                error!(target: "net::session::remove_sub_on_stop()",
104               "Failed to fetch last seen for {}", channel.display_address());
105            }
106        }
107    }
108
109    // For all sessions that are not refine sessions, mark this addr as
110    // Free. `unregister()` frees up this addr for any future operation. We
111    // don't call this on refine sessions since the unregister() call
112    // happens in the refinery directly.
113    if type_id & SESSION_REFINE == 0 {
114        if let Err(e) = hosts.unregister(channel.address()) {
115            error!(target: "net::session::remove_sub_on_stop()", "Error while unregistering addr={}, err={e}", channel.display_address());
116        }
117    }
118
119    if type_id & SESSION_DIRECT != 0 {
120        dnetev!(p2p.session_direct(), DirectDisconnected, {
121            connect_addr: channel.info.connect_addr.clone(),
122            err: "Channel stopped".to_string()
123        });
124    }
125
126    if !p2p.is_connected() {
127        hosts.disconnect_publisher.notify(Error::NetworkNotConnected).await;
128    }
129    debug!(target: "net::session::remove_sub_on_stop()", "[END]");
130}

132/// Session trait. Defines methods that are used across sessions.
133/// Implements registering the channel and initializing the channel by
134/// performing a network handshake.
135#[async_trait]
136pub trait Session: Sync {
137    /// Registers a new channel with the session.
138    /// Performs a network handshake and starts the channel.
139    /// If we need to pass `Self` as an `Arc` we can do so like this:
140    /// ```
141    /// pub trait MyTrait: Send + Sync {
142    ///     async fn foo(&self, self_: Arc<dyn MyTrait>) {}
143    /// }
144    /// ```
145    async fn register_channel(
146        &self,
147        channel: ChannelPtr,
148        executor: Arc<Executor<'_>>,
149    ) -> Result<()> {
150        trace!(target: "net::session::register_channel()", "[START]");
151
152        // Protocols should all be initialized but not started.
153        // We do this so that the protocols can begin receiving and buffering
154        // messages while the handshake protocol is ongoing. They are currently
155        // in sleep mode.
156        let p2p = self.p2p();
157        let protocols =
158            p2p.protocol_registry().attach(self.type_id(), channel.clone(), p2p.clone()).await;
159
160        // Perform the handshake protocol
161        let protocol_version = ProtocolVersion::new(channel.clone(), p2p.settings().clone()).await;
162        debug!(
163            target: "net::session::register_channel()",
164            "Performing handshake protocols {}", channel.clone().display_address(),
165        );
166
167        let handshake_task =
168            self.perform_handshake_protocols(protocol_version, channel.clone(), executor.clone());
169
170        // Switch on the channel
171        channel.clone().start(executor.clone());
172
173        // Wait for handshake to finish.
174        match handshake_task.await {
175            Ok(()) => {
176                debug!(target: "net::session::register_channel()",
177                "Handshake successful {}", channel.clone().display_address());
178            }
179            Err(e) => {
180                debug!(target: "net::session::register_channel()",
181                "Handshake error {e} {}", channel.clone().display_address());
182
183                return Err(e)
184            }
185        }
186
187        // Now the channel is ready
188        debug!(target: "net::session::register_channel()", "Session handshake complete");
189        debug!(target: "net::session::register_channel()", "Activating remaining protocols");
190
191        // Now start all the protocols. They are responsible for managing their own
192        // lifetimes and correctly selfdestructing when the channel ends.
193        for protocol in protocols {
194            protocol.start(executor.clone()).await?;
195        }
196
197        trace!(target: "net::session::register_channel()", "[END]");
198
199        Ok(())
200    }
201
202    /// Performs network handshake to initialize channel. Adds the channel to
203    /// the list of connected channels, and prepares to remove the channel when
204    /// a stop signal is received.
205    async fn perform_handshake_protocols(
206        &self,
207        protocol_version: Arc<ProtocolVersion>,
208        channel: ChannelPtr,
209        executor: Arc<Executor<'_>>,
210    ) -> Result<()> {
211        // Subscribe to stop events
212        let stop_sub = channel.clone().subscribe_stop().await?;
213
214        // Perform handshake
215        match protocol_version.run(executor.clone()).await {
216            Ok(()) => {
217                // Upgrade to goldlist if this is a outbound session.
218                if self.type_id() & (SESSION_OUTBOUND | SESSION_DIRECT) != 0 {
219                    debug!(
220                        target: "net::session::perform_handshake_protocols()",
221                        "Upgrading {}", channel.display_address(),
222                    );
223
224                    let last_seen = UNIX_EPOCH.elapsed().unwrap().as_secs();
225
226                    self.p2p()
227                        .hosts()
228                        .move_host(channel.address(), last_seen, HostColor::Gold)
229                        .await?;
230                }
231
232                // Attempt to add channel to registry
233                self.p2p().hosts().register_channel(channel.clone()).await;
234
235                // Subscribe to stop, so we can remove from registry
236                executor
237                    .spawn(remove_sub_on_stop(self.p2p(), channel, self.type_id(), stop_sub))
238                    .detach();
239
240                // Channel is ready for use
241                Ok(())
242            }
243            Err(e) => return Err(e),
244        }
245    }
246
247    /// Returns a pointer to the p2p network interface
248    fn p2p(&self) -> P2pPtr;
249
250    /// Return the session bit flag for the session type
251    fn type_id(&self) -> SessionBitFlag;
252}
