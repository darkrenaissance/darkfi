19use log::warn;
20use std::sync::Arc;
21
22use crate::{
23    dht::{ChannelCacheItem, DhtHandler, DhtNode},
24    net::session::{SESSION_REFINE, SESSION_SEED},
25    Result,
26};
27
28/// Send a DHT ping request when there is a new channel, to know the node id of the new peer,
29/// Then fill the channel cache and the buckets
30pub async fn channel_task<H: DhtHandler>(handler: Arc<H>) -> Result<()> {
31    loop {
32        let channel_sub = handler.dht().p2p.hosts().subscribe_channel().await;
33        let res = channel_sub.receive().await;
34        channel_sub.unsubscribe().await;
35        if res.is_err() {
36            continue;
37        }
38        let channel = res.unwrap();
39        let channel_cache_lock = handler.dht().channel_cache.clone();
40        let mut channel_cache = channel_cache_lock.write().await;
41
42        // Skip this channel if it's stopped or not new.
43        if channel.is_stopped() || channel_cache.keys().any(|&k| k == channel.info.id) {
44            continue;
45        }
46        // Skip this channel if it's a seed or refine session.
47        if channel.session_type_id() & (SESSION_SEED | SESSION_REFINE) != 0 {
48            continue;
49        }
50
51        let ping_res = handler.ping(channel.clone()).await;
52
53        if let Err(e) = ping_res {
54            warn!(target: "dht::channel_task()", "Error while pinging (requesting node id) {}: {e}", channel.display_address());
55            // channel.stop().await;
56            continue;
57        }
58
59        let node = ping_res.unwrap();
60
61        channel_cache.entry(channel.info.id).or_insert_with(|| ChannelCacheItem {
62            node: node.clone(),
63            topic: None,
64            usage_count: 0,
65        });
66        drop(channel_cache);
67
68        if !node.addresses().is_empty() {
69            handler.dht().add_node(node.clone()).await;
70            let _ = handler.on_new_node(&node.clone()).await;
71        }
72    }
73}
