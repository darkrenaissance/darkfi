19use std::collections::HashMap;
20
21use darkfi::rpc::jsonrpc::{ErrorCode, JsonError, JsonResponse, JsonResult};
22use log::error;
23use tinyjson::JsonValue;
24
25use crate::DarkfiNode;
26
27impl DarkfiNode {
28    // RPCAPI:
29    // Gets a unique ID that identifies this merge mined chain and
30    // separates it from other chains.
31    //
32    // `chain_id`: A unique 32-byte hex-encoded value that identifies
33    //             this merge mined chain.
34    //
35    // --> {"jsonrpc":"2.0", "method": "merge_mining_get_chain_id", "id": 1}
36    // <-- {"jsonrpc":"2.0", "result": {"chain_id": "0f28c...7863"}, "id": 1}
37    pub async fn xmr_merge_mining_get_chain_id(&self, id: u16, _params: JsonValue) -> JsonResult {
38        let (_, genesis_hash) = match self.validator.blockchain.genesis() {
39            Ok(v) => v,
40            Err(e) => {
41                error!(
42                    target: "darkfid::rpc::xmr_merge_mining_get_chain_id",
43                    "[RPC] Error fetching genesis block hash: {}", e,
44                );
45                return JsonError::new(ErrorCode::InternalError, None, id).into()
46            }
47        };
48
49        let resp_obj = HashMap::from([("chain_id".to_string(), genesis_hash.to_string().into())]);
50        JsonResponse::new(resp_obj.into(), id).into()
51    }
52}
